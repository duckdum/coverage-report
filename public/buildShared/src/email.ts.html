import sgMail from '@sendgrid/mail'
import Bull from 'bull'
import {JobType} from "./platform";
 
async function send(job: Bull.Job, apkURL: string | Error): Promise<boolean> {
    const apiKEY = process.env.SENDGRID_ANDROID_BUILD_API_KEY
    if (!apiKEY) {
        console.log('NO API KEY AVAILABLE') // FIXME handle error - this should be a throw
        return false
    }
    sgMail.setApiKey(apiKEY)
    const msg = createEmailMessage(job, apkURL)
 
    const sent = await sgMail.send(msg)
    // console.log(sent)
    return true
}
 
function createEmailMessage(job: Bull.Job, apkURL: string | Error) {
    const fromEmail = job.data.hasPriority ? 'pro-support@thunkable.com' : 'support@thunkable.com'
    const fromNameEmail = job.data.hasPriority ? 'Thunkable PRO Support' : 'Thunkable Support'
    const bccAddress = job.data.hasPriority ? 'pro-support@thunkable.com': 'thunkable-ios-build-or-publish-updates@thunkable.com';
 
    const androidAABSuccessTemplate = 'd-f7af3b3eb60f4c0190434257f5cdf64d'
    const androidFailureTemplate = 'd-9fcb6ba3e24f44ff9f6375973ec79b06'
    const androidAPKSuccessTemplate = 'd-782fae6b37bf4ca7bac94ed3ef2de8d4'
    let successTemplate = androidAPKSuccessTemplate
    if (job.data.androidBuildType === JobType.AAB) {
        successTemplate = androidAABSuccessTemplate
    }
    const templateId = (typeof apkURL === 'string') ? successTemplate : androidFailureTemplate
 
    const dynamicTemplateData = (typeof apkURL === 'string') ? {
        appName: job.data.projectname,
        apkURL: apkURL,
    } : {
        appName: job.data.projectname,
        error: apkURL.message,
        referenceId: job.data.uuid
    }
 
    const bcc = process.env.NODE_ENV === 'production' ? bccAddress : '';
    const msg = {
        to: job.data.recipientemail,
        from: {
            name: fromNameEmail,
            email: fromEmail
        },
        bcc,
        templateId,
        dynamicTemplateData
    }
    return msg
}
 
async function sendWarning(apkURL: string): Promise<boolean> {
    const apiKEY = process.env.SENDGRID_ANDROID_BUILD_API_KEY
    if (!apiKEY) {
        console.log('NO API KEY AVAILABLE') // FIXME handle error - this should be a throw
        return false
    }
    sgMail.setApiKey(apiKEY)
 
    const msg = {
        to: 'pro-support@thunkable.com',
        from: 'support@thunkable.com',
        subject: 'Thunkable Warning System: Android Build Server: Large app',
        text: `An app is larger than 200MB ${apkURL}`,
    }
    await sgMail.send(msg)
    return true
}
 
export default {
    send,
    sendWarning
}
