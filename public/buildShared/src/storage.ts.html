async function loadBucket(uploadApk: boolean): Promise<Bucket>  {
 
    const googleCloudStorage = new Storage({
        projectId: process.env.GOOGLE_PROJECT_ID,
        keyFilename: path.join(path.dirname(__dirname), '/config/Thunkable-iOS-5edcdefd37bf.json'),
    });
    const bucketName = uploadApk ? process.env.GOOGLE_GCS_BUCKET_ANDROID_APP || 'dev-bucket' : process.env.GOOGLE_GCS_BUCKET || 'dev-bucket'
 
    async function getBucket() {
        const theB = await googleCloudStorage.bucket(bucketName);
        console.log(`Bucket ${bucketName} found.`);
        return theB
    }
 
    const bucket = getBucket().catch((err) => {
        console.error(err)
        return err
    })
    return bucket
}
 
async function downloadAssets(appFolder: string, state: string, uuid: string): Promise<ExistingAssets | Error> {
    const allState = JSON.parse(state)
    // FIXME localPath depends on Platform; for now it is only working for Android
    const localPath = `${appFolder}/android/app/src/main/assets/thunkable-assets/user/`
    // Make sure to delete everything in assets if something was left behind by a build failing
    let files = fs.readdirSync(localPath)
    const ignored = [ '.DS_Store', '.gitignore']
    files = files.filter(f => !ignored.includes(f))
    if (files.length > 0) {
        try {
            const deleteImages = await executeShellCommand(`cd ${localPath} && rm -rvf ./* ..?* .[!.]*`,
                'DELETE_IMAGES_BUILD', uuid)
        } catch(err: any) {
            return err;
        }
    }
    if (allState && allState.assets) {
        const allAssets = allState.assets
        const assetCollection = Object.values(allAssets).map((a: any) => {
            return {
                externalStorageType: a.externalStorageType,
                externalName: a.externalName,
                name: a.name
            }
        })
        for (const currentAsset of assetCollection) {
            await downloadAsset(currentAsset, localPath + currentAsset.name)
        }
        return ExistingAssets.ASSETS_FOUND
    } else {
        console.log('PROJECT WITH NO ASSETS')
        return ExistingAssets.NO_ASSETS_FOUND
    }
}
 
async function downloadAsset(asset: any, localDestination: string) {
    const options = {
        destination: localDestination,
    };
    if (asset.externalStorageType === ExternalStorageType.local) {
        await endOfStream(fs.createReadStream(asset.externalName).pipe(fs.createWriteStream(localDestination)))
    } else {
        const theBucket = await loadBucket(false)
        await theBucket.file(asset.externalName).download(options)
    }
}

async function downloadIcon(appFolder: string, state: any, isCustomSplashScreenEnabled: boolean): Promise<boolean | Error> {
    const allState = JSON.parse(state)
    if (allState && allState.settings && allState.settings.icon) {
        const icon = allState.settings.icon
        // FIXME localDestination depends on Platform; for now it is only working for Android
        const localDestination = `${appFolder}/android/app/src/main/res/mipmap-`
        const localSplash = `${appFolder}/android/app/src/main/res/drawable/splashscreen_image.png`
        try {
            if (icon.externalStorageType === ExternalStorageType.local) {
                await resizeIcon(icon.externalName, localDestination, isCustomSplashScreenEnabled, localSplash)
            } else {
                const localPath = `${appFolder}/android/app/src/main/assets/thunkable-assets/user/`
                await downloadAsset(icon, localPath + icon.name)
                await resizeIcon(localPath + icon.name, localDestination, isCustomSplashScreenEnabled, localSplash)
            }
        } catch (e: any) {
            return e
        }
    }
    return true
}
 
async function resizeIcon(iconPath: string, localDestination: string, isPro: boolean, localSplash: string) {
    // Test if using resizeAction fails -- this is done first because it will not throw within the enfOfStream pipeline
    try {
        await sharp(iconPath).resize(200).png().toBuffer()
    }
    catch (e) {
        throw new Error(`The Icon on your App is Invalid, Corrupted, or of an Unsupported Format: ${e}`)
    }
    for (const rez of Object.keys(iconSize)) {
        try {
            const resizeAction = sharp().resize(iconSize[rez], iconSize[rez]).png()
            await endOfStream(fs.createReadStream(iconPath).
                pipe(resizeAction).
                pipe(fs.createWriteStream(`${localDestination}${rez}/ic_launcher.png`)))
        }
        catch (e) {
            throw new Error(`The Icon on your App is Invalid, Corrupted, or of an Unsupported Format: ${e}`)
        }
    }
 
    if (isPro) {
        const extendSize = (dimensionSize: number) => (dimensionSize - launchScreenSize.icon) / 2;
        const transforms = sharp()
            .resize(launchScreenSize.icon, launchScreenSize.icon)
            .png()
            .extend({
                left: extendSize(launchScreenSize.width),
                right: extendSize(launchScreenSize.width),
                top: extendSize(launchScreenSize.height),
                bottom: extendSize(launchScreenSize.height),
                background: { r: 255, g: 255, b: 255, alpha: 1 },
            })
        await endOfStream(fs.createReadStream(iconPath).
            pipe(transforms).
            pipe(fs.createWriteStream(localSplash)))
    }
}
 
async function downloadKeystore(appFolder: string, state: any): Promise<string | Error> {
    const localDestination = `${appFolder}/android/upload.keystore`
    const gcsKeystorePath = (userId: string, projectId: string) => `users/${userId}/projects/${projectId}/keystores/android.keystore`;
    const allState = JSON.parse(state)
    // const projectId = allState.settings.build.android.keystore.projectId
    const projectId = allState && allState._id
    // const userId = allState.settings.build.android.keystore.owner
    const userId = allState && allState.owner
    if (!projectId || !userId) {
        return new Error('No userId or Owner; cannot download keystore')
    }
    const gcsPath = gcsKeystorePath(userId, projectId);
 
    const options = {
        destination: localDestination,
    };
    const theBucket = await loadBucket(false)
    await theBucket.file(gcsPath).download(options)
    return gcsPath
}
 
async function uploadBuildFileToGCS(appFolder: string, androidBuildType: string, buildFileName: string, state: any, isPublic: boolean): Promise<string | Error> {
    const buildType = androidBuildType === JobType.APK ? 'apk' : 'bundle'
    const buildFilePath = `${appFolder}/android/app/build/outputs/${buildType}/release/${buildFileName}`
    // Note we use the /apks/ folder for both apk and aab files.
    const gcsKeystorePath = (userId: string, projectId: string) =>
        `users/${userId}/projects/${projectId}/apks/${buildFileName}`;
    const allState = JSON.parse(state)
    // const projectId = allState.settings.build.android.keystore.projectId
    const projectId = allState && allState._id
    // const userId = allState.settings.build.android.keystore.owner
    const userId = allState && allState.owner
    if (!projectId || !userId) {
        return new Error('No userId or Owner; cannot upload APK')
    }
    const buildFileUrl = `https://storage.googleapis.com/${process.env.GOOGLE_GCS_BUCKET_ANDROID_APP}/users/${userId}/projects/${projectId}/apks/${querystring.escape(buildFileName)}`
 
    // TODO if it is a local build, it should keep the file and not send to GCS? Also don't clean up the apk file at the end.
    try {
        const theBucket = await loadBucket(true)
        const uploadResponse = await theBucket.upload(buildFilePath, {
            destination: gcsKeystorePath(userId, projectId)
        })
        const uploadedFile = uploadResponse[0];
        if (isPublic) {
            await uploadedFile.makePublic();
        }
        // If file is larger than 200MB, notify us
        const stats = await statPromise(buildFilePath)
        if (stats.size > 200000000) {
            await email.sendWarning(buildFileUrl)
        }
        return buildFileUrl
    } catch (e: any) {
        return e
    }
}
 
async function uploadErrorLog(fileNameForErrorLog: string, uuid: string, errors: {error: Error | null, stdout: string, stderr: string}) {
    const gcsLogPath = `logs/${uuid}-${fileNameForErrorLog}-${new Date().getTime()}`
    const theBucket = await loadBucket(true)
    const localFilePath = `/tmp/${uuid}-${fileNameForErrorLog}-${new Date().getTime()}_log.json`;
    fs.writeFileSync(localFilePath, JSON.stringify(errors))
    const uploadResponse = await theBucket.upload(localFilePath, {
        destination: gcsLogPath
    })
}
