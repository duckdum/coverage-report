import Bull from 'bull'
import disk from './disk'
import manifestChanges from "./manifestChanges"
import cleanup from "./cleanup"
import storage from "./storage"
import email from './email'
import { ExistingAssets, AMP_Event_Types } from "./platform"
import tracking from './tracking'
import { resolve } from 'path'
 
// This code should be in the Android build server and not here.
async function orchestrateAndroidBuild(appFolderRaw: string, job: Bull.Job): Promise<boolean | Error> {
 
    const appFolder = resolve(appFolderRaw);
    let cleanAssets = ExistingAssets.NO_ASSETS_FOUND
    const androidBuildType = job.data.androidBuildType;
    try {
        const cleanFolder = await disk.cleanBuildFolder(appFolder, androidBuildType, job.data.uuid)
        if (typeof cleanFolder !== 'boolean') {
            await email.send(job, cleanFolder)
            await tracking.trackBuild(androidBuildType, job.data, AMP_Event_Types.ANDROID_BUILD_EMAIL_FAILURE, cleanFolder)
            return cleanFolder
        }
        const state = await disk.prepareState(appFolder, androidBuildType, job.data.uuid)
        if (typeof state !== "string") {
            await email.send(job, state)
            await tracking.trackBuild(androidBuildType, job.data, AMP_Event_Types.ANDROID_BUILD_EMAIL_FAILURE, state)
            return state
        }
        await job.progress(10)
        const buildFileName = await manifestChanges.buildChanges(appFolder, job, state)
        const existingAssets = await storage.downloadAssets(appFolder, state, job.data.uuid)
        if (existingAssets instanceof Error) {
            await email.send(job, existingAssets)
            await tracking.trackBuild(androidBuildType, job.data, AMP_Event_Types.ANDROID_BUILD_EMAIL_FAILURE, existingAssets)
            return existingAssets
        } else {
            cleanAssets = existingAssets
        }
 
        const isCustomSplashScreenEnabled = job.data.planLimitation && job.data.planLimitation.isCustomSplashScreenEnabled;
        const copyIcon = await storage.downloadIcon(appFolder, state, isCustomSplashScreenEnabled)
        if (typeof copyIcon !== 'boolean') {
            await email.send(job, copyIcon)
            await tracking.trackBuild(androidBuildType, job.data, AMP_Event_Types.ANDROID_BUILD_EMAIL_FAILURE, copyIcon)
            return copyIcon
        }
        const keyStorePath = await storage.downloadKeystore(appFolder, state)
 
        if (typeof buildFileName !== 'string') {
            await email.send(job, buildFileName)
            await tracking.trackBuild(androidBuildType, job.data, AMP_Event_Types.ANDROID_BUILD_EMAIL_FAILURE, buildFileName)
            return buildFileName
        }
        await job.progress(20)
        const buildOutput = await disk.runBuild(appFolder, androidBuildType, buildFileName, job.data.uuid)
        // If error we do not have to clean or use storage or sendEmail
        if (typeof buildOutput !== 'boolean') {
            await email.send(job, buildOutput)
            await tracking.trackBuild(androidBuildType, job.data, AMP_Event_Types.ANDROID_BUILD_EMAIL_FAILURE, buildOutput)
            return buildOutput
        }
        await job.progress(70)
        const apkURL = await storage.uploadBuildFileToGCS(appFolder, androidBuildType, buildFileName, state, true)
        if (typeof apkURL !== 'string') {
            await email.send(job, apkURL)
            await tracking.trackBuild(androidBuildType, job.data, AMP_Event_Types.ANDROID_BUILD_EMAIL_FAILURE, apkURL)
            return apkURL
        }
        await job.progress(90)
        const emailSent = await email.send(job, apkURL)
        await tracking.trackBuild(androidBuildType, job.data, AMP_Event_Types.ANDROID_BUILD_EMAIL_SUCCESS, null)
        await tracking.updateUserBuildInfo(job.data.accountemail, 'android')
        await job.progress(100)
    } catch (e: any) {
        console.error('Issues running the build: ', e)
        await tracking.trackBuild(androidBuildType, job.data, AMP_Event_Types.ANDROID_BUILD_EMAIL_FAILURE, e)
        await email.send(job, e)
        return e
    }
    finally {
        await cleanup.cleanup(appFolder, androidBuildType, cleanAssets, job.data.uuid, job.data.componentUsage)
    }
 
    return true
}
export default {
    orchestrateAndroidBuild,
}
