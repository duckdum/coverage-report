export function skipTracking(eventType) {
  if ([TRACKING_EVENT_TYPES.CREATE_BLOCK_FROM_FLYOUT,

    return false;
  }
 
  const lastTrackedTime = parseInt(window.sessionStorage.getItem(eventType)) || 0;
  const newTrackedTime = new Date().getTime();
  // only send the tracking to amplitude once every 6 hours
  if (lastTrackedTime === 0 || (newTrackedTime - lastTrackedTime) >= 6*60*60*1000) {

  }
  return true;

export function trackWhereFromToProjectPage(isOwner, location, project) {
  // navType + action: where this is from:
  // reload + POP: (1) navigate using broswer back and forward within Thunkable (2) reload the page
  // navigate + POP: (1) Click on a link from other places (2) enter URL directly
  // back_forward + POP (1) navigate using broswer back to outside pages and forward back to Thunkable
  const navigations = window.performance.getEntriesByType("navigation");
  const navType = get(navigations, '[0].type');
  const action = get(location, 'action');
  const pathname = get(location, 'pathname');
  let context = get(location, 'state.context');
  if (!context) context = sessionStorage.getItem(SESSION_STORAGE_CONTEXT);
  if (!context && navType === 'navigate' && action === 'POP') {
    context = !isOwner ? "Read-only other's public" : 'View owned projects with project links';
    if (isOwner && project.readOnly) context = "Read-only expired owned private";
  }
  if (context) {
    sessionStorage.removeItem(SESSION_STORAGE_CONTEXT);
    const componentUsage = ComponentUtils.getComponentUsage(project) || {};
    const defaultAppLayout = `${project.defaultDesignerDevice} + ${project.defaultDesignerOrientation}`;
    const eventProperties = {
      context,
      projectId: project.id,
      projectName: project.projectName,
      'Default app layout': defaultAppLayout,
      'New UI': !!project.usesDragDropUi,
      projectCategory: project.categories,
      'Public/Private': project.isPublic ? 'Public' : 'Private',
      ...componentUsage,
    }
    track(TRACKING_EVENT_TYPES.OPEN_PROJECTS, eventProperties);
  }
 
  // Clean up state
  window.history.replaceState(null, '', pathname);
}
