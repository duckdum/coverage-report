import React, { Component } from 'react';
import { postLogin } from '../actions';
import { Button, Input, notification as Notification } from 'antd/lib';
// Comment ReCaptcha until we handle this case https://2captcha.com/blog/google-doesnt-accept-recaptcha-answers
// import ReCaptcha from './ReCaptcha';
 
const emailErrorMessage = 'Please enter a valid email address';
 
class EmailLoginButton extends Component {
  constructor(props) {
    super(props);
    this.state = {
      showEmailInput: false,
      isReCaptchaScored: false,
      isReCaptchaVarified: false,
      email: undefined,
      sso: props.sso,
      sig: props.sig,
      errorMessage: undefined,
    };
  }
 
  componentDidUpdate(prevProps, prevState, snapshot) {
    if (this.state.showEmailInput !== prevState.showEmailInput) {
      this.refs.emailInput && this.refs.emailInput.input.focus();
    }
  }
 
  showEmailInput = () => {
    this.setState({ showEmailInput: true });
  }
 
  setIsReCaptchaScored = () => {
    this.setState({ isReCaptchaScored: true });
  }
 
  setIsReCaptchaVarified = () => {
    this.setState({ isReCaptchaVarified: true });
  }
 
  handleBlur = () => {
    if (this.state.email) {
      this.setState({ showEmailInput: true });
    } else {
      this.setState({
        // showEmailInput: false,
        errorMessage: undefined,
      });
    }
  }
 
  handleChange = (e) => {
    const isEmailValid = this.validateEmail(e.target.value);
    this.setState({
      email: e.target.value,
      errorMessage: isEmailValid ? false : emailErrorMessage,
    });
  }
  
  validateEmail = (email) => {
    const regex = /^(([^<>()[\]\\.,;:\s@"]+(\.[^<>()[\]\\.,;:\s@"]+)*)|(".+"))@((\[[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\])|(([a-zA-Z\-0-9]+\.)+[a-zA-Z]{2,}))$/;
    return regex.test(email);
  }
 
  login = async () => {
    if (this.state.errorMessage !== false) return;
 
    const trackingData = {
      isHoc: localStorage.getItem('thunkable-ios-hoc'),
      path: this.props.path,
      prevPath: this.props.prevPath,
      referrerId: this.props.referrerId,
    };
    const response = await postLogin('/emaillogin', undefined, this.state.email, this.props.sso, this.props.sig, trackingData);
 
    if (response.error === 'ACCOUNT_SUSPENDED') {
      this.props.setIsAccountSuspended(true);
    } else if (response.errorMessage) {
      this.setState({ errorMessage: response.errorMessage });
    } else if (response.email) {
      const description = (
        <>
          We just emailed a link to {response.email}. Click the link, and you’ll be signed in.
          <br/><br/>
          <span className="spam-folder-warning">Check your spam folder if you can't find this email.</span>
        </>
      )
      Notification.success({
        message: 'Check your inbox',
        description,
        placement: 'topRight',
        duration: 0,
      });
    }
  }
 
  render() {
    const action = this.props.path === '/signup' ? 'up' : 'in';
    const elements = this.state.showEmailInput ?
      (<div className={'email-login-button-container has-feedback' + (this.state.errorMessage ? ' has-error' : this.state.errorMessage === false ? ' has-success' : '')}>
        <p>We will send a link to your inbox</p>
        <div className='ant-form-item-children'>
          <Input
            className='email-login-input'
            onPressEnter={this.login}
            onBlur={this.handleBlur}
            onChange={this.handleChange}
            placeholder='Enter your email'
            ref='emailInput'
            type='email' />
          <span className='ant-form-item-children-icon'></span>
        </div>
        { this.state.errorMessage &&
          <div className='ant-form-explain'>{this.state.errorMessage}</div> }
        {/* <ReCaptcha
          action="login"
          isScored={ this.state.isReCaptchaScored }
          setIsScored={ this.setIsReCaptchaScored }
          setIsVarified={ this.setIsReCaptchaVarified }
        /> */}
        <Button
          className='email-link-button'
          disabled={ this.state.errorMessage !== false }
            // || !this.state.isReCaptchaScored
            // || !this.state.isReCaptchaVarified }
          onClick={this.login}
          size='large'
          type='primary'>
          Email me the link
        </Button>
      </div>) :
      (<div className='email-login-button-container'>
        <Button
          className='email-login-button'
          icon='mail'
          onClick={this.showEmailInput}
          size='large'
          type='primary'>
          { `Sign ${action} with email` }
        </Button>
      </div>);
 
    return elements;
  }
}
 
export default EmailLoginButton;
