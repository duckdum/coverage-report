import React, { useState } from 'react';
import i18n, { t } from 'i18next';
import { Avatar, Icon, Input, Table, Tooltip } from 'antd/lib';
 
import ProjectActionMenu from './ProjectActionMenu';
import ProjectStats from './ProjectStats';
import defaultProjectIcon from '../images/defaultProjectIcon.png';
import { browserHistory } from 'react-router';
import '../style/Project.scss';
 
const { Column } = Table;
const options = { year: 'numeric', month: 'long', day: 'numeric', hour12: true, hour: '2-digit', minute: '2-digit' };
 
const ProjectList = ({
  copyProject,
  deleteProject,
  paginationProps,
  projects,
  renameProject,
  showLiveTestModal,
  isProfile
}) => {
  const [hoverId, setHoverId] = useState('');
  const [editId, setEditId] = useState('');
  const [isExitEdit, setIsExitEdit] = useState(false);
  const [newProjectName, setNewProjectName] = useState('');
  const onEdit = (e) => {
    setIsExitEdit(false);
    setEditId(hoverId);
    setHoverId('');
  }
  const onRenameProject = (e) => {
    renameProject(newProjectName, editId);
    setEditId('');
    if (e.type === 'blur') setIsExitEdit(true);
  }
  return (
    <Table
      className="project-list"
      dataSource={ projects }
      pagination={ paginationProps }
      rowClassName="project-list-row"
      rowKey={ record => record.id }
      onRow={(record, rowIndex) => {
        return {
          onClick: event => {
            const redirectPath = record.firstScreenId ? `/projects/${record.id}/${record.firstScreenId}/designer` : `/projects/${record.id}`;
            
            if ((event.target.tagName !== 'TD' && event.target.tagName !== 'DIV')
              || isExitEdit) {
              setIsExitEdit(false);
              return;
            }
 
            copyProject && deleteProject
            ? window.location.href = redirectPath
            : browserHistory.push(`/projectPage/${record.id}`);
          },
          onMouseEnter: event => setHoverId(record.id),
          onMouseLeave: event => setHoverId(''),
        };
      }}
    >
      <Column
        className="project-icon"
        dataIndex="icon"
        key="icon"
        render={(text, record) => <Avatar src={ record.icon || defaultProjectIcon } />}
      />
      <Column
        className="project-name"
        dataIndex="projectName"
        key="projectName"
        sorter={ (a, b) => {
          if (a.projectName < b.projectName) return -1;
          if (a.projectName > b.projectName) return 1;
          return 0;
        }}
        render={(text, record) =>
          renameProject && editId === record.id
          ? (
            <Input
              type="text"
              placeholder="Project Name..."
              defaultValue={ record.projectName }
              onChange={ e => setNewProjectName(e.target.value)}
              onPressEnter={ onRenameProject }
              onBlur={ onRenameProject }
              ref={(input) => input && input.focus() }
            />
          )
          : (
            <div className="project-title-wrapper">
              <Tooltip placement="top" title={ record.projectName }>
                <div className="project-title"> { record.projectName } </div>
              </Tooltip>
            { renameProject &&
              <Icon
                className={ `edit-button ${ hoverId === record.id ? 'visible' : '' }` }
                onClick={ onEdit }
                type="edit"
            /> }
            </div>
          )
        }
        title={ t('projectsPage.myProjects.name') }
      />
      <Column
        key="preview"
        render={(text, record) => {
          const handleLiveTest = () => showLiveTestModal(record.id, record.projectName, record.usesDragDropUi);
          return (
            <div className="live-test-action-container" onClick={ handleLiveTest }>
              <Icon type="play-circle" theme="twoTone" />
            </div>
          )
        }}
        title={ t('projectsPage.myProjects.preview') } 
      />
      <Column
        key="stats"
        render={(text, record) => (
          <div className="project-list-detail-container">
            <ProjectStats
              className="project-stats-container"
              isOwner={ record.isOwner }
              project={ record } />
          </div>
        )}
      />
    { copyProject && deleteProject &&
      <Column
        key="privacy"
        render={(text, record) => {
          let privacyTagName;
          let privacyTagClassName = 'project-privacy-setting';
 
          if (record.readOnly) {
            privacyTagName = t('privacy.readOnly');
            privacyTagClassName += ' private-read-only-label';
          } else if (!record.isPublic) {
            privacyTagName = t('privacy.private');
            privacyTagClassName += ' private-label';
          } else {
            privacyTagName = t('privacy.public');
            privacyTagClassName += ' public-label';
          }
 
          return record.isOwner && privacyTagName && (
            <div className="project-privacy-tag-container">
              <div className={ privacyTagClassName }>{ privacyTagName }</div>
            </div>
          )
        }}
        title={ t('projectsPage.myProjects.privacy') } 
      /> }
      <Column
        dataIndex="updatedAt"
        defaultSortOrder="ascend"
        key="updatedAt"
        render={(text, record) => isProfile ?
          (new Date(Number(record.updatedAt))).toLocaleString(i18n.language, options) :
          (new Date(record.updatedAt)).toLocaleString(i18n.language, options) }
        sorter={ (a, b) => {
          const updatedTimestampA = new Date(isProfile ? Number(a.updatedAt): a.updatedAt).getTime();
          const updatedTimestampB = new Date(isProfile ? Number(b.updatedAt): b.updatedAt).getTime();
          if (updatedTimestampA < updatedTimestampB) return 1;
          if (updatedTimestampA > updatedTimestampB) return -1;
          return 0;
        }}
        title={ t('projectsPage.myProjects.dateModified') } 
      />
    { copyProject && deleteProject &&
      <Column
        key="action"
        render={(text, record) => {
          return copyProject && deleteProject && (
            <ProjectActionMenu
              copyProject={ copyProject }
              deleteProject={ deleteProject }
              projectId={ record.id }
              projectName={ record.projectName }
              webAppLink={ record.webAppLink }
            />
          );
        }}
      /> }
    </Table>
  );
};
 
export default ProjectList;
