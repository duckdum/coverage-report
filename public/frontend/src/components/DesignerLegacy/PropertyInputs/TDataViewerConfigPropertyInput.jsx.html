import React, { useState } from 'react';
import { Form, Select } from 'antd/lib';
import TDataViewerItemLayoutPropertyInput from './TDataViewerItemLayoutPropertyInput';
import { connect } from 'react-redux';
import { get } from 'lodash';
import ComponentUtils from '../../../shared/utils/ComponentUtils';
import * as Actions from '../../../actions';
import TDataViewerSwipeConfigPropertyInput from '../../Designer/PropertyInputs/TDataViewerSwipeConfigPropertyInput';
import TDataViewerDataBindingPropertyInput from './TDataViewerDataBindingPropertyInput';
import { ProjectContext } from '../../ProjectController';
 
const FormItem = Form.Item;
const Option = Select.Option;
 
const BINDING_PROPERTY = 'DataSourceTableBinding';
 
const mapStateToProps = (state, ownProps) => {
  const props = {};
  if (state.projects[ownProps.projectId]) {
    const project = state.projects[ownProps.projectId];
    const component = ComponentUtils.getComponentById(ownProps.componentId, project.components, project.apiComponents.ApiContainer.children);
    props.component = component;
    props.project = project;
  }
  
  return props;
}
 
const mapDispatchToProps = (dispatch, ownProps) => {
  return {
    replaceChildren: (newChildren) => {
      dispatch(Actions.replaceChildren(
        ownProps.projectId,
        ownProps.componentId,
        newChildren
      ));
    },
    bindingChangeDispatcher: (componentId, prop, newBinding) => {
      dispatch(Actions.changeComponentPropertyBinding(ownProps.projectId, componentId, prop, newBinding));
    },
    renameComponent: (newComponentName) => {
      dispatch(Actions.renameComponent(ownProps.projectId, ownProps.componentId, newComponentName));
    },
    valueChangeDispatcher: (componentId, prop, change) => {
      dispatch(Actions.changeComponentProperty(ownProps.projectId, componentId, prop, change));
    },
  }
}
  
function DataViewerConfigPropertyInput(props) {
  const [config, setConfig] = useState(props.value);
  const [labelToBindingMap, setLabelToBindingMap] = useState({});
  const {
    layoutSelection,
  } = config;
  const itemLayoutLabel = `${layoutSelection.toLowerCase()} Item Layout`;
  const { componentId, projectId, readOnly, replaceChildren, component } = props;
  
  const updateConfig = (newConfig) => {
    setConfig(newConfig);
    props.valueChanged(props.name, newConfig);
  }
 
  const handleItemLayoutChange = (itemLayoutSelection, components) => {
    const previousBindings = { ...labelToBindingMap };
    const children = ComponentUtils.getChildren(component);
    children.forEach(({ bindings, bindingLabel }) => {
      if (bindings) {
        Object.keys(bindings).forEach((key) => {
          const binding = bindings[key];
          if (binding && bindingLabel && binding.componentId === component.id
            && binding.property === BINDING_PROPERTY) {
            previousBindings[bindingLabel] = bindings[key];
          }
        });
      }
    });
    const newConfig = { ...config, itemLayoutSelection };
    updateConfig(newConfig);
    ComponentUtils.componentSearch(components, (component) => {
      const { bindingLabel, boundProperty } = component;
      if (bindingLabel && boundProperty) {
        if (!component.bindings) {
          component.bindings = {};
        }
        component.bindings[boundProperty] = previousBindings[bindingLabel];
      }
    });
    replaceChildren(components);
    setLabelToBindingMap(previousBindings);
  }
 
  const handleSwipeUpdate = (direction, swipe) => {
    const newConfig = { ...config };
    if (!newConfig.swipe) {
      newConfig.swipe = {};
    }
    newConfig.swipe[direction] = { ...newConfig.swipe[direction], ...swipe[direction] };
    updateConfig(newConfig);
  };
 
  const handleBindingChange = (componentId, column, boundProperty) => {
    const newBinding = {
      type: 'COMPONENT_PROPERTY_TABLE_ELEMENT',
      componentId: props.componentId,
      property: BINDING_PROPERTY,
      column: column,
      row: 0
    };
    props.bindingChangeDispatcher(componentId, boundProperty, newBinding);
  }
 
  const clearBinding = (componentId, boundProperty) => {
    props.bindingChangeDispatcher(componentId, boundProperty, {});
  }
 
  const components = ComponentUtils.getChildren(props.component);
  return (
    <ProjectContext.Consumer>
      {(project) => {
          if (!project) return null;
          const boundTable = get(component, `bindings.${BINDING_PROPERTY}`, {});
          const isLocal = get(boundTable, 'isLocal', false);
          let boundCollection;
          if (isLocal) {
            const localDataSource = get(props.project, 'localDataSources', []).find(({ id }) => id === boundTable.dataSourceLinkId);
            boundCollection = get(localDataSource, 'dataSource.collections', []).find(({ id }) => id === boundTable.collectionId);
          } else {
            const dataSourceLinks = get(project, 'dataSourceLinks', []);
            const dataSource = get(dataSourceLinks.find((link) => link.id === boundTable.dataSourceLinkId), 'dataSource', {});
            boundCollection = get(dataSource, 'collections', []).find(({ id }) => id === boundTable.collectionId);
          }
          const columns = get(boundCollection, 'fields', []);
          const columnOptions = columns.map(({ id, label }) => (
            <Option key={id} value={id}>
              {label}
            </Option>
          ));
        return (
          <>
            <FormItem key="ItemLayout" label={itemLayoutLabel}>
              <TDataViewerItemLayoutPropertyInput
                componentId={componentId}
                componentChildren={component.children}
                handleItemLayoutChange={handleItemLayoutChange}
                layoutSelection={layoutSelection}
                projectId={projectId}
              />
            </FormItem>
            <FormItem key="Bindings" label="Data Bindings">
              <TDataViewerDataBindingPropertyInput
                components={components}
                clearBinding={clearBinding}
                sourcePanelColumns={columnOptions}
                bindings={{}}
                handleBindingChange={handleBindingChange}
              />
            </FormItem>
            {layoutSelection === 'LIST' &&
              <TDataViewerSwipeConfigPropertyInput
                columns={props.boundColumns}
                swipeConfig={get(config, 'swipe', {})}
                handleSwipeUpdate={handleSwipeUpdate}
                readOnly={readOnly}
              />}
          </>
        );
      }}
    </ProjectContext.Consumer>
  );
}
 
export default connect(mapStateToProps, mapDispatchToProps)(DataViewerConfigPropertyInput);
