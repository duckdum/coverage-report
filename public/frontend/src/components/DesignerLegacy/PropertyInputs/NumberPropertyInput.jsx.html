import React, { Component } from 'react'
import { Input } from 'antd/lib';
import { isString } from 'lodash';
import { TRACKING_EVENT_TYPES, track } from "../../../tracking";
 
class NumberPropertyInput extends Component {
  static getDerivedStateFromProps(nextProps, state) {
    if (nextProps.value !== state.currentPropsValue) {
      return {
        ...state,
        value: nextProps.value
      };
    }
    return null;
  }
 
  constructor(props) {
    super(props);
    this.state = {
      value: props.value
    };
  }
 
  onChange = (e) => {
    const { value } = e.target;
    const reg = /^-?(0|[1-9][0-9]*)(\.[0-9]*)?$/;
    if ((!isNaN(value) && reg.test(value)) || value === '' || value === '-') {
      this.setState({value}, () => this.handleChange(value));
    }
  }
 
  componentDidUpdate = () => {
    if (!isNaN(this.props.value) && this.state.currentPropsValue !== this.props.value) {
      this.setState({ currentPropsValue: this.props.value });
    }
  }
 
  handleChange = (value) => {
    if (isString(value) && (value.charAt(value.length - 1) === '.' || value === '-')) return;
 
    let convertedValue;
    if (value === '' || value === undefined) {
      convertedValue = undefined;
    } else {
      convertedValue = parseFloat(value, 10);
    }
    if (convertedValue === this.props.value) return;
 
    if (this.props.valueChanged !== undefined) {
      this.props.valueChanged(this.props.propertyIdentifier, convertedValue);
      // Temporary tracking for Number of Lines property
      if (this.props.propertyIdentifier === 'NumberOfLines') {
        const eventProperty = { "Project Id": this.props.projectId, "Value": value, "New UI": false, "Project Category": this.props.categories };
        track(TRACKING_EVENT_TYPES.NUMBER_OF_LINES_VALUE_CHANGED, eventProperty);
      }
    }
  }
 
  onMouseUp = () => {
    // Temporary tracking for Number of Lines property
    if (this.props.propertyIdentifier === 'NumberOfLines') {
      const eventProperty = { "Project Id": this.props.projectId, "New UI": false, "Project Category": this.props.categories };
      track(TRACKING_EVENT_TYPES.NUMBER_OF_LINES_MOUSE_UP, eventProperty);
    }
  }
 
  render() {
    let defaultValue;
    let value = this.props.isFromCustomModal ? this.props.value : this.state.value;
    if (value !== undefined && isNaN(value) && value !== '-') value = undefined;
    let onChangeFunc = this.props.isFromCustomModal ? this.props.onChange : this.onChange;
    let styleObject = this.props.isFromCustomModal ? this.props.style : { maxWidth: '76px' };
    const numberInput = 
      (<Input 
        onBlur={this.props.onBlur}
        onMouseUp={this.onMouseUp}
        style={styleObject} 
        placeholder={defaultValue}
        value={value} 
        onChange={onChangeFunc}
        disabled={this.props.readOnly}
      />);
 
    if (this.props.isFromCustomModal) {
      return numberInput;
    } else { 
      let descriptor;
      if (this.props.propType) {
        descriptor = this.props.propType.descriptor;
        defaultValue = descriptor.defaultValue;
        if (defaultValue === undefined) defaultValue = '#';
      }
      return numberInput;
    }
  }
}
 
export default NumberPropertyInput;
