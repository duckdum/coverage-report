import { connect } from 'react-redux';
import ComponentUtils from '../../shared/utils/ComponentUtils';
import { isAdMob, isAdMobBanner } from '../../shared/componentDefinitions';
import { COMPONENTS, SORTABLE_COMPONENTS } from '../../thunkableComponents';
import * as Actions from '../../actions';
import { DropPosition } from './SortableView';
import UIModule from '../../thunkableComponents/UIModule';
import { notification as Notification } from 'antd/lib';

/* In some cases we have components which we do not want
 * to be able to be individaully dragged to reposition.
 * This code examines all of the components within a 
 * screen to find which of those should not be 
 * draggable or sortable and stores their ids in a
 * set which we use at render time to conditionally
 * determine whether to render a sortable or non-
 * sortable version of that component in the phone
 * previewer.
 */ 
const getNonSortableIds = (components) => {
  const nonSortableIds = new Set();
  ComponentUtils.componentSearch(components, (component) => {
    if (COMPONENTS[component.type].ChildrenNotSortableInPreview) {
      const children = ComponentUtils.getChildren(component);
      children.forEach(({ id }) => {
        nonSortableIds.add(id);
      });
    }
  })
  return nonSortableIds;
}
 
const mapStateToProps = (state, ownProps) => {
const props = {
    previewMove: state.projectConfigs[ownProps.projectId]
      && state.projectConfigs[ownProps.projectId].previewMoveNode,
  };
  if (state.projects[ownProps.projectId]) {
    if (ownProps.phonePreviewComponentId !== undefined) {
      const project = state.projects[ownProps.projectId];
      const container = ComponentUtils.getContainerComponentByType(ownProps.phonePreviewComponentId, ownProps.containerType, project.components);
      if (container.type === ownProps.containerType) {
        props.components = container;
        props.assets = project.assets;
        props.readOnly = project.readOnly;
        props.nonSortableIds = getNonSortableIds(props.components);
      }
    }
  }
 
  return props;
}
 
const mapDispatchToProps = (dispatch, ownProps) => {
  return {
    addModuleComponent: (parentId, index, type, properties, isApi, translatedName) => {
      dispatch(Actions.addModuleComponent(
        ownProps.projectId,
        parentId, 
        index, 
        type,
        properties,
        isApi, 
        translatedName
      ));
    },
    addComponent: (component, parentId, index, customProperties) => {
      dispatch(Actions.createNewComponent(
        ownProps.projectId,
        parentId,
        index,
        component, 
        undefined,
        customProperties
      ));
    },
    moveComponent: (componentId, toParentId, toIndex, fromParentId, fromIndex) => {
      dispatch(Actions.moveComponent(ownProps.projectId, componentId, toParentId, toIndex, fromParentId, fromIndex));
    },
    previewMoveNode: (parentId, index, dropPosition) => {
      dispatch(Actions.previewMoveNode(ownProps.projectId, parentId, index, dropPosition));
    },
  }
}
 
class PhonePreviewer extends Component {
 
  previewMoveNode = (componentId, toParentId, toIndex, dropPosition) => {
    let previewMove = this.props.previewMove;
    if (previewMove === undefined) {
      previewMove = {};
    }
 
    if (toParentId === previewMove.parentId &&
        toIndex === previewMove.index &&
        dropPosition === previewMove.dropPosition) {
      return
    }
 
    if (toParentId === undefined && toIndex === undefined && previewMove !== undefined) {
      this.props.previewMoveNode(undefined, undefined, undefined);
      return;
    }
 
    this.props.previewMoveNode(toParentId, toIndex, dropPosition);
  }
 
  moveNode = async (componentId, toParentId, toIndex, fromParentId, fromIndex, type, isApi) => {
    if (componentId !== undefined) {
      this.props.moveComponent(componentId, toParentId, toIndex, fromParentId, fromIndex);
    } else if (type !== undefined) {
      const addingModule = type.includes('module:');
      if (addingModule) {
        const moduleId = type.substring(7);
 
        const moduleFound = ModuleUtils.findModule(this.props.modulesAddedToProject, moduleId);
        if (moduleFound) {
          const translatedName = moduleFound.name;
          const customProperties = moduleFound.customProperties;
          const properties = {};
          if (customProperties) {
            customProperties.forEach(function(property, index) {
              properties[property.uuid] = property.defaultValue;
            });
          }
          const isApi = (moduleFound.isApi);
          this.props.addModuleComponent(toParentId, toIndex, type, properties, isApi, translatedName);
        }
        return;
      }
 
      if (isAdMob(type)) {
        const variables = { projectId: this.props.projectId, type: type };
 
        if (isAdMobBanner(type)) {
          const componentIsScreen = this.props.components.type && this.props.components.type === 'Screen';
          const screen = componentIsScreen ? this.props.components : ComponentUtils.getScreenForComponent(toParentId, this.props.components);
          variables.screenUUID = screen.componentUUID ? screen.componentUUID : screen.id;
        }
 
        const result = (await this.props.addAdmob({ variables })).data.addAdmob;
 
        if (result.errorMessage) {
          Notification.error({
            message: 'Oops!',
            description: `You can only add ${result.errorMessage}`,
            placement: 'topRight',
            duration: 0,
          });
          return;
        }
      }
 
      if (isApi) {
        let variables = { projectId: this.props.projectId, type: type, isAdding: true };
        track(TRACKING_EVENT_TYPES.ADD_API_COMPONENT, variables);
        await this.props.trackAPIComponentUsage({ variables: variables });
      }
 
      this.props.addComponent(type, toParentId, toIndex, this.props.customProperties);
    }
  }
 
  selectComponent = (componentId) => {
    return (event) => {
      event.stopPropagation();
      if (componentId === this.props.selectedComponentId) {
        return;
      }
      this.props.setSelectedComponentId(componentId);
    }
  }
 
  reactCreateElementCallback = (component, childElements, index, parentId, parentType, componentParentIdsById, insideModule=false, moduleComponentId=undefined) => {
    const properties = component.properties !== undefined ? component.properties : {};
    const assets = this.props.assets !== undefined ? this.props.assets : {};
    const style = properties.style;
    const id = component.id;
    const active = id === this.props.selectedComponentId;
    const elements = [];
    const isNotSortable = insideModule || this.props.nonSortableIds.has(id);
    const sortableOrNot = isNotSortable ? COMPONENTS : SORTABLE_COMPONENTS;
    if (component.type && sortableOrNot[component.type]) {
      const TComponent = sortableOrNot[component.type];
      elements.push(
        <TComponent
          style={style}
          key={id}
          name={component.name ? component.name : id} 
          id={id}
          type={component.type}
          parentType={parentType}
          parentId={parentId}
          parentIdsById={componentParentIdsById}
          index={index} 
          previewMoveNode={this.previewMoveNode}
          moveNode={this.moveNode}
          active={active} 
          properties={properties}
          assets={assets}
          handleClick={insideModule ? this.selectComponent(moduleComponentId) : this.selectComponent(id)}
          readOnly={this.props.readOnly}
          thunkableVersion={component.thunkableVersion}
        >
          {childElements}
        </TComponent>
      ); 
    } else if (component.type && component.type.includes('module:')) {
      if (component.isApi) return elements;
      const moduleId = component.type.substring(7);
      const moduleFound = ModuleUtils.findModule(this.props.modulesAddedToProject, moduleId);
      const moduleContainer = moduleFound && moduleFound.components && 
        moduleFound.components.children && moduleFound.components.children.length === 1 &&
        moduleFound.components.children[0];
      
      const moduleElement = ComponentUtils.createReactElement(
        moduleContainer, (component, childElements, index, parentId, parentType) => 
          this.reactCreateElementCallback.call(this, component, childElements, index, parentId, parentType, componentParentIdsById, true, id)
      );
      const UIModuleComponent = SORTABLE_COMPONENTS[UIModule.Type];
      elements.push(
        <UIModuleComponent
            moduleElement={moduleElement}
            style={style}
            key={id}
            name={component.name} 
            id={id}
            type={component.type}
            parentType={parentType}
            parentId={parentId}
            parentIdsById={componentParentIdsById}
            index={index} 
            previewMoveNode={this.previewMoveNode}
            moveNode={this.moveNode}
            active={active} 
            properties={properties}
            assets={assets}
            handleClick={this.selectComponent(id)}
            readOnly={this.props.readOnly}
          ></UIModuleComponent>   
            
      );
    } else {
      // TODO: Do something more reasonable here.
      return elements;
    }
 
    const previewMove = this.props.previewMove;
 
    if (previewMove !== undefined) {
      const isTargetOfMovePreview = parentId === previewMove.parentId && index === previewMove.index;
      if (isTargetOfMovePreview) {
        const movePreviewElement = <div key="move-preview" className="phone-move-preview-element" />
        if(previewMove.dropPosition === DropPosition.BEFORE) {
          elements.unshift(movePreviewElement)
        } else {
          elements.push(movePreviewElement)
        }
      }
    }
    return elements;
  }
 
  render() {
    if (this.props.components !== undefined) {
      const componentParentIdsById = ComponentUtils.getParentIdsById(this.props.components);
      const components = ComponentUtils.createReactElement(this.props.components, 
        (component, childElements, index, parentId, parentType) => this.reactCreateElementCallback.call(this, component, childElements, index, parentId, parentType, componentParentIdsById)
      );
 
    return (
      <div className={ "phone-preview-column " + (this.props.isSavedScreen ? "module-preview-column" : '' )}>
        <div className="phone-preview">
          <div className="phone-screen">
            {components}
          </div>
        </div>
      </div>
    );
    }
 
    return ( 
      <div className="no-screen-selected-text">
        Select a screen in the component tree to preview it here.
      </div>
    );
 
  }
}
