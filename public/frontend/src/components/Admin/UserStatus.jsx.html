import React, { Component } from 'react';
import gql from 'graphql-tag';
import { Mutation, Query } from "react-apollo";
import UserStatusCard from './UserStatusCard';
import { Alert, Button, Card, Modal, Tooltip } from 'antd/lib';
 
const userStatus = `{
  id
  isTrial
  isCancelled
  plan
  planId
  nextPlan
  payment {
    kind
    price
    nextPrice
    trialEnd
    currentSubscriptionEnd
    currentPeriodEnd
    isQualifiedForLegacy
    isLegacy
  }
}`;
 
const GET_USER = gql`
  query getUser($email: String!) {
    getUser(email: $email) ${userStatus}
  }
`;
 
const GRANT_PRO = gql`
  mutation grantPRO($email: String!, $plan: String!, $endDate: String) {
   grantPRO(email: $email, plan: $plan, endDate: $endDate) ${userStatus}
  }
`;
 
const REVOKE_PRO = gql`
  mutation revokePRO($email: String!, $isCurrent: Boolean!) {
    revokePRO(email: $email, isCurrent: $isCurrent) ${userStatus}
  }
`;
 
const REVOKE_TRIAL = gql`
  mutation revokeTrial($email: String!) {
    revokeTrial(email: $email)  ${userStatus}
  }
`;
 
const TOGGLE_IS_LEGACY = gql`
  mutation toggleIsLegacy($email: String!) {
    toggleIsLegacy(email: $email)  ${userStatus}
  }
`;
 
const REACTIVATE = gql`
  mutation reactivate($email: String!) {
    reactivate(email: $email)  ${userStatus}
  }
`;
 
const GrantPRO = (email, actionText, plan, hideModal, endDate) => {
  return (
    <Mutation
      mutation={GRANT_PRO}
      update={(cache, { data: { grantPRO: granted } }) => {
        cache.writeQuery({
          query: GET_USER,
          variables: { email },
          data: { getUser: granted },
        });
      }}>
      {(grantPRO, { data }) => (
        <Button size='large' onClick={() => {
          grantPRO({ variables: { email, plan, endDate } });
          hideModal();
        }}>{ actionText }</Button>
      )}
    </Mutation>
  );
}
 
const RevokePRO = (email, isCurrent, isStripe, nextPlan) => {
  const revokePlansMessage = isCurrent && nextPlan ? 'This will also revoke next plan' : '';
  return (
    <Mutation
      mutation={REVOKE_PRO}
      update={(cache, { data: { revokePRO: revoked } }) => {
        cache.writeQuery({
          query: GET_USER,
          variables: { email: email },
          data: { getUser: revoked },
        });
      }}>
      {(revokePRO, { data }) => (
        <Tooltip title={ isStripe ? '' : revokePlansMessage}>
        <Button size='large' onClick={() => {
          revokePRO({ variables: { email: email, isCurrent: isCurrent } });
        }}>Revoke</Button>
        </Tooltip>
      )}
    </Mutation>
  );
}
 
const ToggleIsLegacy = (email, isLegacy) => {
  return (
    <Mutation
      mutation={TOGGLE_IS_LEGACY}
      update={(cache, { data: { toggleIsLegacy: toogled } }) => {
        cache.writeQuery({
          query: GET_USER,
          variables: { email },
          data: { getUser: toogled },
        });
      }}>
      {(toggleIsLegacy, { data }) => (
        <Button size='large' onClick={() => {
          toggleIsLegacy({ variables: { email } });
        }}>{ isLegacy ? 'Revoke Legacy' : 'Grant Legacy' }</Button>
      )}
    </Mutation>
  );
}
 
const RevokeTrail = (email) => {
  return (
    <Mutation
      mutation={REVOKE_TRIAL}
      update={(cache, { data: { revokeTrial: revoked } }) => {
        cache.writeQuery({
          query: GET_USER,
          variables: { email: email },
          data: { getUser: revoked },
        });
      }}>
      {(revokeTrial, { data }) => (
        <Button size='large' onClick={() => {
          revokeTrial({ variables: { email: email } });
        }}>Charge Now</Button>
      )}
    </Mutation>
  );
}
 
const Reactivate = (email) => {
  return (
    <Mutation
      mutation={REACTIVATE}
      update={(cache, { data: { reactivate: reactivated } }) => {
        cache.writeQuery({
          query: GET_USER,
          variables: { email: email },
          data: { getUser: reactivated },
        });
      }}>
      {(reactivate, { data }) => (
        <Button size='large' onClick={() => {
          Modal.confirm({
            title: 'Do you want to un-cancel this subscription?',
            onOk() {
              reactivate({ variables: { email: email } });
            },
          });
        }}>Un-cancel</Button>
      )}
    </Mutation>
  );
}
 
class UserStatus extends Component {
 
  render() {
    return (
      <Query query={GET_USER} variables={{ email: this.props.findBy }}>
        {({ loading, error, data: { getUser: user } , refetch }) => {
          if (loading) return null;
          if (!error && this.props.shouldRefetch) refetch();
          if (error) return (<Alert message={error.message} type='error' showIcon closable />);
          if (!user) return (<Alert message='This user does not exist.' type='error' showIcon closable />);
          const isStripe = user.payment.kind === 'Stripe';
          const billingCycle = user.plan && user.plan.split(' ')[0];
          const nextBillingCycle = user.nextPlan && user.nextPlan.split(' ')[0];
          const currentPlanActionText = user.plan ? 'Extend' : 'Grant';
 
          return (
            <Card title='User Status' className='user-status-cards'>
              <UserStatusCard
                title='Trial'
                content={user.isTrial && 'Trial'}
                revoker={isStripe && user.isTrial && RevokeTrail(this.props.findBy)}
                trialEnd={user.payment.trialEnd}
              />
              <UserStatusCard
                title={`Current Plan ${ user.plan && user.payment.kind ? `(${user.payment.kind})` : '' }`}
                plan={user.plan}
                planId={user.planId}
                content={billingCycle}
                price={user.payment.price}
                legacyToggler={user.payment.isQualifiedForLegacy && ToggleIsLegacy(this.props.findBy, user.payment.isLegacy)}
                reactivator={user.plan && !user.nextPlan && user.payment.currentSubscriptionEnd && Reactivate(this.props.findBy)}
                revoker={user.plan && RevokePRO(this.props.findBy, true, isStripe, user.nextPlan)}
                proGranter={(!user.payment.kind || !user.plan || (isStripe && !user.nextPlan))
                  && GrantPRO.bind(this, this.props.findBy, currentPlanActionText)}
                proGranterActionText={currentPlanActionText}
                currentPeriodEnd={user.payment.currentPeriodEnd}
              />
              <UserStatusCard
                title={`Next Plan ${ user.plan && user.payment.kind ? `(${user.payment.kind})` : '' }`}
                plan={user.nextPlan}
                content={nextBillingCycle}
                price={user.payment.nextPrice}
                revoker={isStripe && user.nextPlan && RevokePRO(this.props.findBy, false, isStripe)}
                proGranter={isStripe && user.nextPlan && GrantPRO.bind(this, this.props.findBy, 'Change')}
                proGranterActionText='Change'
              />
            </Card>
          );
        }}
      </Query>
    );
  }
}
 
export default UserStatus;
