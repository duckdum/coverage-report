import React, { Component } from 'react';
import Dropzone from 'react-dropzone';
import gql from 'graphql-tag';
 
import { 
  Button,
  Form,
  Icon,
  Input,
  Layout,
} from 'antd/lib';
 
import {
  compose,
  graphql,
} from 'react-apollo';
 
const { Content } = Layout;
const typeInvalid = 'The file type is wrong.';
const expiredErrorMessage = 'The file is either expired or will be expired in one hour.';
const teamIDNotMatchRegex = /The team id in your file \(\w{10}\) does not match/;
const bundleIdNotMatchRegex = /The package name in your file \([\w|.|*]+\) does not match/;
const profileTypeErrorMessage = `Oops! It appears the file you uploaded is not a (.mobileprovision) file.
  Please upload the Provisioning Profile you generated from the Apple developer website.`;
const profileMatchErrorMessage = (errorMessage) => `Oops! ${errorMessage} the one your entered earlier.`;
const profileUUIDErrorMessage = 'There is problem getting the provisioning profile UUID right now. Please try again or contact customer support.';
const generalErrorMessage = 'Oops! There is something wrong. Please try again.';
const profileGeneratedErrorMessage = `The provisioning profile is not generated using the certificate you uploaded previously.`;
const profileNoBundleIdErrorMessage = `There is something wrong with your uploaded provisioning profile file. Please regenerate one from the Apple Developer website.`;
 
const paraOne = (

const paraTwo = ( <p>Upload your Provisioning Profile below:</p> );
 
const uploadProfile = gql`
  mutation uploadProfile($projectId: ID! $file: Upload!) {
    uploadProfile(projectId: $projectId file: $file) {
      filename
      isErrorDisplayable
      errorMessage
    }
  }
`;
 
const hasProfileOnGCS = gql`
  mutation hasProfileOnGCS($projectId: ID!) {
    hasProfileOnGCS(projectId: $projectId) {
      has
      filename
      teamID
      bundleId
      bundleIdRegex
      uuid
      expiration
      matchCert
      errorMessage
      metadataErrorMessage
    }
  }
`
 
class TwoFAProfile extends Component {
  constructor(props) {
    super(props);
    this.state = {
      profileFileName: undefined,
      errorMessage: undefined,
    };
  }
 
  componentDidMount = () => {
    this.getProfileMetadataOnGCS();
  }
 
  onNext = () => {
    this.props.form.validateFields(async (error, values) => {
      if (!error && !this.state.errorMessage) this.props.onNext();
    });
  }
 
  onPressEnter = (event) => {
    event.preventDefault();
    this.onNext();
  }
 
  getProfileMetadataOnGCS = async () => {
    const check = await this.props.hasProfileOnGCS();
    const hasProfileOnGCS = check.data.hasProfileOnGCS.has;
 
    if (hasProfileOnGCS) {
      const profileFileName = check.data.hasProfileOnGCS.filename;
      const expiration = check.data.hasProfileOnGCS.expiration;
      const isExpired = expiration && new Date(expiration) - Date.now() < 60 * 60 * 1000;
      const teamID = check.data.hasProfileOnGCS.teamID;
      const bundleId = check.data.hasProfileOnGCS.bundleId;
      const bundleIdRegexString = check.data.hasProfileOnGCS.bundleIdRegex;
      const bundleIdRegex = bundleIdRegexString.substring(1, bundleIdRegexString.length - 1);
      const matchCert = check.data.hasProfileOnGCS.matchCert;
      const metadataErrorMessage = check.data.hasProfileOnGCS.metadataErrorMessage;
      let errorMessage;
 
      if (isExpired) {
        this.setState({ errorMessage: expiredErrorMessage });
      } else if (!matchCert){
        this.setState({ errorMessage: profileGeneratedErrorMessage });
      } else if (teamID !== this.props.settings.teamId) {
        errorMessage = `The team id in your file ${teamID} does not match`;
        this.setState({ errorMessage: profileMatchErrorMessage(errorMessage) });
      } else if (bundleIdRegex && !this.props.settings.packageName.match(new RegExp(bundleIdRegex))) {
        errorMessage = `The package name in your file ${bundleId} does not match`;
        this.setState({ errorMessage: profileMatchErrorMessage(errorMessage) });
      } else if (metadataErrorMessage) {
        this.setState({ errorMessage: metadataErrorMessage });
      }
 
      this.setState({ profileFileName: profileFileName });
    }
  }
 
  handleDrop = async (files) => {
    this.props.startProcess();
    const result = await this.props.uploadProfile({
      variables: {
        projectId: this.props.projectId,
        file: files[0],
      }
    });
    const errorMessage = result.data.uploadProfile.errorMessage;
    const isErrorDisplayable = result.data.uploadProfile.isErrorDisplayable;
 
    if (!errorMessage) {
      this.setState({
        profileFileName: result.data.uploadProfile.filename,
        errorMessage: undefined,
      });
    } else if (errorMessage === profileGeneratedErrorMessage){
      this.setState({ errorMessage: profileGeneratedErrorMessage });
    }  else if (errorMessage === typeInvalid) {
      this.setState({ errorMessage: profileTypeErrorMessage });
    } else if (errorMessage === expiredErrorMessage) {
      this.setState({ errorMessage: expiredErrorMessage });
    } else if (errorMessage.match(teamIDNotMatchRegex) || errorMessage.match(bundleIdNotMatchRegex)) {
      this.setState({ errorMessage: profileMatchErrorMessage(errorMessage) });
    } else if (errorMessage === profileNoBundleIdErrorMessage) {
      this.setState({ errorMessage: profileNoBundleIdErrorMessage });
    } else if (errorMessage === profileUUIDErrorMessage) {
      this.setState({ errorMessage: profileUUIDErrorMessage });
    } else if (isErrorDisplayable) {
      this.setState({ errorMessage: errorMessage });
    } else {
      this.setState({ errorMessage: generalErrorMessage });
    }
 
    this.props.finishProcess();
  }
 
  render() {
    return (
      <div className='content'>
        <Content>
          {paraOne}
          {paraTwo}
 
          <Form className='profile-form'>
            <Form.Item hasFeedback
              validateStatus={(this.state.errorMessage && 'error')
                || (this.state.profileFileName && 'success')}
              help={this.state.errorMessage && this.state.errorMessage}>
              { this.props.form.getFieldDecorator('profile', {
                rules: [
                  { required: this.state.profileFileName ? false : true,
                    message: 'Please upload your profile.' },
                  ],
                })( <Dropzone
                    style={{ width: 'auto' }}
                    disablePreview={true}
                    onDrop={this.handleDrop}
                    multiple={false}>
                    <Input.Group size='default'>
                    { this.state.profileFileName &&
                      <div>{this.state.profileFileName}</div> }
                      <Button disabled={this.props.readOnly}>
                        <Icon type='upload' /> Upload Profile
                      </Button>
                    </Input.Group>
                  </Dropzone> )}
            </Form.Item>
          </Form>
        </Content>
 
        <div className='footer'>
          <Button key='back' onClick={this.props.onBack}>Back</Button>
          <Button key='next' type='primary' onClick={this.onNext}>Next</Button>
        </div>
      </div>
    );
  }
}
 
const TwoFAProfileWithGraphQL = compose(
  graphql(uploadProfile, { name: 'uploadProfile' }),
  graphql(hasProfileOnGCS, {
    name: 'hasProfileOnGCS',
    poptions: props => ({ variables: { projectId: props.projectId } }),
  }),
)(TwoFAProfile);
 
export default Form.create()(TwoFAProfileWithGraphQL);
