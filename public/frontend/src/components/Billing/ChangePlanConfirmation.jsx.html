import React, { useState, useEffect } from 'react';
import { Checkbox, Divider } from 'antd/lib';
import { t } from 'i18next';
import { isEmpty, isNumber } from 'lodash';
import ProBenefitsV4 from './ProBenefitsV4';
import PaypalButton from './PaypalButton';
import { getPlanForDisplay } from './Billing';
import {
  Spacing,
  ThunkableButton,
  ThunkableText,
} from '../ComponentLibrary';
 
const PLAN_LEVEL = {
  enterprise: 4,
  business: 3,
  pro: 2,
  starter: 1,
  free: 0,
};
 
const getAnnualPrice = (price) => price && Math.floor(price * 12 / 10) * 10;
const ChangePlanConfirmation = (props) => {
  const isPremium = props.plan && props.plan.includes('premium');
  const suffix = props.plan && props.plan === 'legacy' ? '' : `_${props.selectedPlan}`;
  const plan = `${props.selectedCycle}${suffix}`;
  const planName = isPremium ? 'Premium Pro' : 'Pro';
  const planDetail = props.pricing[`${plan}`];
  const price = isNumber(planDetail?.price) ? planDetail?.price : planDetail?.originalPrice;
  const initTotal = props.plan === 'annual' ? getAnnualPrice(price) : price;
  const initActionTitle = props.plan === 'free'
    ? t("billing.confirmation.cancel")
    : t("billing.confirmation.upgrade", { plan: planName });
  const initButtonText = props.plan === 'free'
    ? t("billing.confirmation.continue")
    : t("billing.confirmation.upgrade", { plan: planName });
  const [originalTotal, setOriginalTotal] = useState(initTotal);
  const [promoDurationMessage, setPromoDurationMessage] = useState(undefined);
  const [isAgree, setIsAgree] = useState(false);
  const [isUpgrading, setIsUpgrading] = useState(false);
  const [total, setTotal] = useState(initTotal);
  const [actionTitle, setActionTitle] = useState(initActionTitle);
  const [buttonText, setButtonText] = useState(initButtonText);
 
  const onChangeAgreement = (e) => {
    const checked = e.target.checked;
    setIsAgree(checked);
  }
 
  const getCheckout = () => {
    const suffix = props.selectedPlan && props.selectedPlan === 'legacy' ? '' : `_${props.selectedPlan}`;
    const plan = `${props.selectedCycle}${suffix}`;
    const planDetail = props.pricing[plan];
    const price = isNumber(planDetail?.price) ? planDetail?.price : planDetail?.originalPrice;
    let newTotal = props.selectedCycle.includes('annual') ? getAnnualPrice(price) : price;
    setOriginalTotal(newTotal)
 
    if (!isEmpty(props.promoSetup)) newTotal = getTotalWithPromoSetup(newTotal);
    if (props.selectedPlan === 'free') newTotal = 0;
 
    newTotal = isNumber(newTotal) ? +(newTotal.toFixed(2)) : null;
    setTotal(newTotal);
  }
 
  const checkIfUpgrading = () => {
    const currentLevel = PLAN_LEVEL[props.currentPlan];
    const selectedLevel = PLAN_LEVEL[props.selectedPlan];
    return currentLevel < selectedLevel;
  }
 
  const handleChange = () => {
    const isNowUpgrading = checkIfUpgrading();
    props.handleChangePlan(isNowUpgrading);
  }
 
  const getTotalWithPromoSetup = (currentTotal) => {
    const setup = props.promoSetup || {};
    const isBillingCycleNotValid = setup && setup.billingCycle && setup.billingCycle !== props.selectedCycle;
    const isPlanTypeNotValid = setup && setup.planType && !setup.planType.includes(props.selectedPlan);
 
    if (isEmpty(setup) || isBillingCycleNotValid || isPlanTypeNotValid) {
      setPromoDurationMessage(undefined);
      return currentTotal;
    }
 
    const hasAmmountOff = isNumber(setup.amount_off);
    const percentOff = setup.percent_off && setup.percent_off * 0.01;
    let newTotal;
 
    if (percentOff) {
      newTotal = currentTotal * (1 - percentOff).toFixed(2);
    } else if (hasAmmountOff && currentTotal > setup.amount_off * 0.01) {
      newTotal = currentTotal - (setup.amount_off * 0.01);
    }
 
    const durationMessage = setup.duration === 'forever'
      ? t("billing.confirmation.lifetimePromo", { price: newTotal, cycle: props.selectedCycle })
      : t("billing.confirmation.onetimePromo", { price: currentTotal });
    setPromoDurationMessage(durationMessage);
 
    return newTotal;
  }
 
  useEffect(() => {
    getCheckout();
    const selectedPlan = props.selectedPlan && getPlanForDisplay(props.selectedPlan);
    const formattedPlan = selectedPlan && `${selectedPlan.charAt(0).toUpperCase()}${selectedPlan.slice(1)}`;
    const newActionMessage = props.selectedPlan === 'free'
      ? t("billing.confirmation.cancel")
      : t("billing.confirmation.upgrade", { plan: formattedPlan });
    const newButtonText = props.selectedPlan === 'free'
      ? t("billing.confirmation.continue")
      : t("billing.confirmation.upgrade", { plan: formattedPlan });
    const isNowUpgrading = checkIfUpgrading();
 
    setActionTitle(newActionMessage);
    setButtonText(newButtonText);
    setIsAgree(props.selectedPlan === 'free');
    setIsUpgrading(isNowUpgrading);
  }, [props.selectedPlan, props.selectedCycle, props.promoSetup]);
 
  useEffect(() => {
    getCheckout();
  }, []);
 
  return (
    <div className='payment-confirmation'>
      <Spacing bottomSize="medium">
        <ThunkableText type='title' message={ t("billing.confirmation.title") } />
      </Spacing>
      <Spacing bottomSize="medium">
        <ThunkableText type='largeTitle' message={ actionTitle } />
        <ProBenefitsV4 plan={ props.selectedPlan } icon={ props.selectedPlan === 'free' ? 'minus' : 'check' }/>
      </Spacing>
      <Divider />
      <div className='payment-confirm-container plan-details'>
        <ThunkableText type='title' message={ t("billing.confirmation.nextPayment") } />
        <div className='price-container'>
        { promoDurationMessage &&
          <span className='original-price'>
            <span className='price-prefix'>$</span>
            { originalTotal }
          </span> }
          <span className='price-prefix'>$</span>
          <span className='price'>{ total }{ promoDurationMessage && '*' }</span>
        </div>
        { promoDurationMessage &&
        <Spacing topSize="medium">
          <ThunkableText type='body' message={ promoDurationMessage } />
        </Spacing> }
      </div>
      <Divider />
      { props.selectedPlan === 'free'
        ? <ThunkableText message={ t("billing.cancelWarning") } />
        : (<div className="terms-conditions-container">
            <Checkbox onChange={ onChangeAgreement } checked={ isAgree }>
              <ThunkableText type='body' message={ t("billing.agreement.agree") } />
              <a className="terms-conditions-link"
                href="https://thunkable.com/#/terms"
                target="_blank"
                rel="noopener noreferrer">
                <ThunkableText type='body' message={ t("billing.agreement.termsOfService") } />
              </a>
              <ThunkableText type='body' message={ t("billing.agreement.and") } />
              <a className="terms-conditions-link"
                href="https://thunkable.com/#/privacy"
                target="_blank"
                rel="noopener noreferrer">
                <ThunkableText type='body' message={ t("billing.agreement.privacyPolicy") } />
              </a>
            </Checkbox>
          </div>)
      }
      <div className="upgrade-button-container">
        { props.paymentKind === 'PayPal' && isUpgrading && isAgree
        ? <PaypalButton
            isProcessing={ props.isProcessing }
            isUpgrading={ isUpgrading }
            onHide={ props.onHide }
            plan={ `${props.selectedCycle}_${props.selectedPlan}` }
            price={ total }
            setIsProcessing={ props.setIsProcessing }
          />
        :
        <ThunkableButton
          disabled={ !isAgree
            || !isNumber(total)
            || !!(props.selectedPlan === props.currentPlan
              && props.selectedCycle === props.currentCycle
              && ((props.paymentKind !== 'PayPal' && !props.currentSubscriptionEnd && isEmpty(props.promoSetup))
                || (props.paymentKind === 'PayPal' && !props.nextPlan)))}
          onClick={ handleChange }
          translationKey={ buttonText }
          type='primary'
        /> }
      </div>
    </div>
  );
}
 
export default ChangePlanConfirmation;
