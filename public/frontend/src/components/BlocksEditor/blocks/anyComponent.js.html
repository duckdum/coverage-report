const determineInclusivity = function(componentClass, typeToInclude, excludeScreen, componentType, includeContainerOnly, excludeContainer, excludeCannotDropOnScreen, includeInstancesOnCanvasOnly, componentTypesToInclude, includeHasTablePropertyOnly, includeCloneableComponentsOnly, excludeSTPComponents, excludeDndComponents, isRelativePosition, getPropFunc, workspace) {
  if (!componentClass) return false;
  let toBeAdded = true;
  if (typeToInclude === "VISIBLE") {
      if (componentClass.IsApi !== undefined) toBeAdded = false;
  } else if (typeToInclude === "API") {
    if (componentClass.IsApi === undefined) toBeAdded = false;
  }
  if (excludeScreen && componentType === TScreen.Type) toBeAdded = false;
  if (excludeCannotDropOnScreen && componentClass.CanDropOnScreen === false) toBeAdded = false;
  if (excludeSTPComponents && PropTypeUtils.isSTPComponentContainer(componentClass.Type, { includeLayout: true })) toBeAdded = false;
  if (excludeDndComponents && ((!includeContainerOnly && !isRelativePosition) || (includeContainerOnly && !(PropTypeUtils.isSTPComponentContainer(componentClass.Type, { includeLayout: true }))))) toBeAdded = false;
  if (componentClass.ThunkableVersion > workspace.thunkableVersion) toBeAdded = false;
  if (componentClass.Hidden === true) toBeAdded = false;
  if (includeContainerOnly && !(componentClass.ContainVisibleComponents === true)) toBeAdded = false;
  if (excludeContainer && componentClass.ContainVisibleComponents === true) toBeAdded = false;
  if (PropTypeUtils.isNavigator(componentType)) toBeAdded = false;
  if (getPropFunc && getPropFunc(componentType).size === 0) {
    toBeAdded = false;
  }
  if (includeInstancesOnCanvasOnly && !componentClass.IsInstanceOnCanvas) toBeAdded = false;
  if (componentTypesToInclude && componentTypesToInclude.indexOf(componentClass.Type) === -1) toBeAdded = false;
  if (includeCloneableComponentsOnly && componentClass.IsNotCloneable) toBeAdded = false;
  if (includeHasTablePropertyOnly) {
    let hasTableProperty = componentClass === TLocalTable;
    const customProperties = workspace.customProperties;
    if (customProperties) {
      for (let customProperty of customProperties) {
        if (customProperty.type === 'TABLE' && customProperty.componentType === componentType) {
          hasTableProperty = true;
          break;
        }
      }
    }
    if (!hasTableProperty) toBeAdded = false;
  }
 
  return toBeAdded;
}
 
// given the workspace and the type of component (visible or api), return a list of component
const getComponentList = function(workspace, typeToInclude, componentTypesToInclude) {
  if (typeToInclude === "API" || (componentTypesToInclude && componentTypesToInclude.indexOf("Screen") !== -1)) {
    return workspace.getAllThunkableComponents();
  } else if (typeToInclude === "VISIBLE") return workspace.getThunkableComponentsOnScreen();
  return undefined;
}
 
export const getComponentDropdownOptions = function(workspace, typeToInclude, includeAny, includeClasses, excludeInstances, excludeScreen, includeContainerOnly, excludeContainer, excludeCannotDropOnScreen, includeInstancesOnCanvasOnly, componentTypesToInclude, includeHasTablePropertyOnly, includeCloneableComponentsOnly, excludeSTPComponents, excludeDndComponents, getPropFunc) {
  const options = getComponentDropdownOptionsHelper(workspace, typeToInclude, includeAny, includeClasses, excludeInstances, excludeScreen, includeContainerOnly, excludeContainer, excludeCannotDropOnScreen, includeInstancesOnCanvasOnly, componentTypesToInclude, includeHasTablePropertyOnly, includeCloneableComponentsOnly, excludeSTPComponents, excludeDndComponents, getPropFunc);
 
  if (options.length === 0) return [['', '']];
  return options;
}
 
 
// given different criteria, return a list of list of component options
export const getComponentDropdownOptionsHelper = function(workspace, typeToInclude, includeAny, includeClasses, excludeInstances, excludeScreen, includeContainerOnly, excludeContainer, excludeCannotDropOnScreen, includeInstancesOnCanvasOnly, componentTypesToInclude, includeHasTablePropertyOnly, includeCloneableComponentsOnly, excludeSTPComponents, excludeDndComponents, getPropFunc) {
  if (typeToInclude === "ALL") {
    const visibleComponentOptions = getComponentDropdownOptionsHelper(workspace, "VISIBLE", includeAny, includeClasses, excludeInstances, excludeScreen, includeContainerOnly, excludeContainer, excludeCannotDropOnScreen, includeInstancesOnCanvasOnly, componentTypesToInclude, includeHasTablePropertyOnly, includeCloneableComponentsOnly, excludeSTPComponents, excludeDndComponents, getPropFunc);
    const apiComponentOptions = getComponentDropdownOptionsHelper(workspace, "API", includeAny, includeClasses, excludeInstances, excludeScreen ? excludeScreen : !excludeScreen, includeContainerOnly, excludeContainer, excludeCannotDropOnScreen, includeInstancesOnCanvasOnly, componentTypesToInclude, includeHasTablePropertyOnly, includeCloneableComponentsOnly, excludeSTPComponents, excludeDndComponents, getPropFunc);
    return visibleComponentOptions.concat(apiComponentOptions);
 
  }
  const componentList = getComponentList(workspace, typeToInclude, componentTypesToInclude);
  if (!componentList) return [['','']];
  let options = [];
 
  // Add the component instances
  if (!excludeInstances) {
    for (let component of componentList) {
      const componentType = component.type;
      const componentClass = COMPONENTS[componentType];
      const isRelativePosition = get(component, 'properties.style.position') === "relative";
      const toBeAdded = determineInclusivity(componentClass, typeToInclude, excludeScreen, componentType, includeContainerOnly, excludeContainer, excludeCannotDropOnScreen, includeInstancesOnCanvasOnly, componentTypesToInclude, includeHasTablePropertyOnly, includeCloneableComponentsOnly, excludeSTPComponents, excludeDndComponents, isRelativePosition, getPropFunc, workspace);
      if (toBeAdded) {
        const name = component.name ? component.name : component.id;
        options.push([name, component.id]);
      }
    }
  }
 
  // Add the component classes
  // Visible components: add all componentType regardless whether they exist in the project because we can dynamically create/clone them
  // API components: only add only the componentType that exists in the project because we can't dynamically create/clone API components rn
  const unSortedComponentTypeList = typeToInclude === "API" ? componentList.map((component) => component.type) : Object.keys(COMPONENTS);
  // use component categories to provide a reasonable ordering for the component types
  const componentCategories = Object.keys(CATEGORY_TO_LABEL);
  let componentTypeList = [];
  componentCategories.forEach(componentCategory => {
    const categoryComponents = unSortedComponentTypeList.filter(componentType => COMPONENTS[componentType] && COMPONENTS[componentType].Category === componentCategory);
    const sortedCategoryComponents = categoryComponents.sort((a, b) => COMPONENTS[a].Index - COMPONENTS[b].Index);
    componentTypeList = componentTypeList.concat(sortedCategoryComponents);
  });
  // if there are components that did not have categories, add them at the end
  unSortedComponentTypeList.forEach((componentType) => {
    if (componentTypeList.indexOf(componentType) === -1) {
      componentTypeList.push(componentType);
    }
  })
 
  if (includeClasses) {
    for (let componentType of componentTypeList) {
      const componentClass = COMPONENTS[componentType];
      const toBeAdded = determineInclusivity(componentClass, typeToInclude, excludeScreen, componentType, includeContainerOnly, excludeContainer, excludeCannotDropOnScreen, includeInstancesOnCanvasOnly, componentTypesToInclude, includeHasTablePropertyOnly, includeCloneableComponentsOnly, excludeSTPComponents, excludeDndComponents, undefined, getPropFunc, workspace);
      if (toBeAdded) {
        const componentTypeTranslated = t(COMPONENTS[componentType].TranslateLabelKey) || componentType;
        const componentTypeDisplay = includeAny ?  t('blocks.event.any').concat(' ', componentTypeTranslated) : componentTypeTranslated;
        options.push([componentTypeDisplay, componentType]);
      }
    }
  }
  return options;
};
