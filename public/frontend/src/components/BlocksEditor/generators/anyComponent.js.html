// Wrap the javascript code around a function, so the cloneComp object created by Screen.insertChild() is not shared beyond its necessary scope.
const addAwait = function(codeFunction, cloneCompVariableName, isDoInputAsync) {
  let wrappedCodeFunc = isDoInputAsync ? `await (` : `(`;
  wrappedCodeFunc += codeFunction; 
  wrappedCodeFunc += `).call(this, ${cloneCompVariableName}.id)`;
  return wrappedCodeFunc;
}
 
// Generates the code for the blocks inside the 'Do' input of the clone block.
const genCloneBlockCallbackCode = function(block, cloneCompVariableName) {
  const inputDoBlocks = Blockly.JavaScript.statementToCode(block, 'DO');
  const isDoInputAsync = Blockly.JavaScript.doesBlockInputContainAsyncBlock(block, 'DO');
  const codeFunction = genFunction(inputDoBlocks, block, 'CALLBACK_PARAM', false, isDoInputAsync);
  return addAwait(codeFunction, cloneCompVariableName, isDoInputAsync);
}
 
// Used when cloning a generic component: This creates the new imitation clone component object with MOST of the default properties. 
// Note that this isn't the actual clone component as the actual clone component will be created by Screen.insertChild(). 
// This default property object of this imitation clone component will be passed to Screen.insertChild().
// Note that the design time custom setters for the props have NOT been dealt with inside of initComponentProperties(); 
const createNewComponent = function(cloneComponentType, usesDragDropUi) {
  let newComponent = { properties: getDefaultProps(cloneComponentType, usesDragDropUi), type: cloneComponentType};
  initComponentProperties(newComponent, undefined, true);
  return newComponent;
}

const cloneBlockGenerator = function(block) {
  // For creating generic component
  let targetCloneComponentId;
  let cloneComponentType;
  let cloneComponentReactType;
  let propertiesString;
  let propToRuntimeCustomSetter = ``;
 
  // For cloning component instances
  let targetCloneComponentNameOrId;
 
  //For drag and drop
  let x_pos = ``;
  let y_pos = ``;
 
  if (block.createGeneric) {
    targetCloneComponentId =  block.getFieldValue(block.fieldNames.cloneComponent);
    const componentClass = COMPONENTS[targetCloneComponentId];
    if (componentClass.IsNotCloneable) return '';
    cloneComponentType = componentClass.Type;
    cloneComponentReactType = componentClass.ReactType || cloneComponentType;
    const newComponent = createNewComponent(cloneComponentType, block.dragAndDrop);
 
    //Set up the runtime custom setter func for the props
    propToRuntimeCustomSetter += '{';
    let propertiesKey = Object.keys(newComponent.properties); 
    for (let i = 0; i < propertiesKey.length; i++) {
      const propName = propertiesKey[i]; 
      const runtimeCustomSetter = getRuntimeCustomSetter(cloneComponentType, propName);
      if (runtimeCustomSetter) {
        propToRuntimeCustomSetter += `"${propName}": `;
        propToRuntimeCustomSetter += runtimeCustomSetter.toString(); 
        propToRuntimeCustomSetter += ",\n";
      }
    }
    propToRuntimeCustomSetter += '}';
    propertiesString = JSON.stringify(newComponent.properties);
  } else {
    targetCloneComponentNameOrId = Blockly.JavaScript.valueToCode(block, this.valueNames.cloneComponent, Blockly.JavaScript.ORDER_ATOMIC) || 'undefined';
  }
  let atPositionOrRelativeLocation;
 
  if (block.dragAndDrop) {
    x_pos = Blockly.JavaScript.valueToCode(block, this.valueNames.xPositionComponent, Blockly.JavaScript.ORDER_NONE);
    y_pos = Blockly.JavaScript.valueToCode(block, this.valueNames.yPositionComponent, Blockly.JavaScript.ORDER_NONE);
    atPositionOrRelativeLocation = block.getFieldValue(block.fieldNames.dndRefLocation);
  } else {
    atPositionOrRelativeLocation = block.getFieldValue(block.fieldNames.refLocation);
  }
 
  const referenceComponentNameOrId = Blockly.JavaScript.valueToCode(block, this.valueNames.referenceComponent, Blockly.JavaScript.ORDER_ATOMIC) || 'undefined';
  const cloneCompVariableName = 'cloneCompVariable';
  const isBlockAsync = Blockly.JavaScript.usingAsyncEval;
  const cloneBlockCallbackCode = genCloneBlockCallbackCode(block, cloneCompVariableName);
  Blockly.JavaScript.usingAsyncEval = isBlockAsync;
  return codeGenerator(cloneCompVariableName, targetCloneComponentNameOrId, referenceComponentNameOrId,
    cloneBlockCallbackCode, cloneComponentType, cloneComponentReactType, propertiesString,
    propToRuntimeCustomSetter, block.dragAndDrop, atPositionOrRelativeLocation, x_pos, y_pos);
}

const allComponentsOfTypeGenerator = function(block) {
  const componentType = block.getFieldValue(block.fieldNames.componentType);
  const getAllComponentsOfTypeFuncName = Blockly.JavaScript.provideFunction_(
    'getAllComponentsOfType',
    [`function ${Blockly.JavaScript.FUNCTION_NAME_PLACEHOLDER_}(componentType, components) {
      return this.getComponentUtils().findAllComponentsOfType(componentType, components);
    }`]
  );
 
  let code = ``;
  if (block.apiOnly) {
    code += `${getAllComponentsOfTypeFuncName}.call(this, "${componentType}", this.getApiContainer())`;
  } else {
    const containerComponentNameOrId = Blockly.JavaScript.valueToCode(block, this.valueNames.containerComponentId, Blockly.JavaScript.ORDER_ATOMIC) || 'undefined';
    code += `
      ${Blockly.JavaScript.usingAsyncEval ? `await (async ` : `(` } function() {
        const containerComponentNameOrIdVariable = ${containerComponentNameOrId};
        const containerComponentId = this.mapComponentNameToId(containerComponentNameOrIdVariable, this.state.components); 
        const containerComponent = this.getComponentUtils().getComponentById(containerComponentId, this.state.components, {});
        const matchingComponentIdsIncludingContainer = ${getAllComponentsOfTypeFuncName}.call(this, "${componentType}", containerComponent);
        return matchingComponentIdsIncludingContainer.filter(listComponentId => listComponentId !== containerComponentId);
      }).call(this)
    `;
  }
  return [code, Blockly.JavaScript.ORDER_NONE];
}
 
const removeComponentBlockGenerator = function(block) {
  const componentToRemove = Blockly.JavaScript.valueToCode(block, this.valueNames.componentId, Blockly.JavaScript.ORDER_ATOMIC) || 'undefined';
  
  let code = `
  await (async function() {
    const componentToRemoveVariable = ${componentToRemove};
    const isArray = Array.isArray(componentToRemoveVariable);
  
    this.setState(function(prevState, props) {
      const nextState = this.spreadCopy(prevState);
      const parentIdsById = this.getComponentUtils().getParentIdsById(nextState.components); 
      
      function componentToRemove(componentToRemoveVariable) {
        const idOfComponentToRemove = this.mapComponentNameToId(componentToRemoveVariable, nextState.components);
        const parentIds = parentIdsById.get(idOfComponentToRemove);
        if (!parentIds) return;
        this.removeComponent(idOfComponentToRemove, parentIds[0], nextState.components);
      }
 
      if (isArray) {
        for (let i = 0; i < componentToRemoveVariable.length; i++) {
          componentToRemove.call(this, componentToRemoveVariable[i]);
        }
      } else { 
        componentToRemove.call(this, componentToRemoveVariable);
      }
      return nextState;
    })
  }).call(this);\n
  `;
  return code;
 
}
