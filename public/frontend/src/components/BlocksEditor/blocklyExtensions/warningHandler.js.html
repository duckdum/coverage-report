import { t } from 'i18next';
import { componentBlockTypeToDependantFieldName, blockTypesWithCallbackParams } from '../customBlocks'
import { componentBlockTypes } from '../constants'
 
let Blockly = window.Blockly;
 
Blockly.WarningHandler = function(workspace) {
  this.workspace = workspace;
}
 
Blockly.WarningHandler.prototype.errorCount = 0;
 
Blockly.WarningHandler.prototype.checkWorkspaceForErrors = function() {
  for (var block of this.workspace.getAllBlocks()) {
    this.checkBlockForErrors(block)
  }
  const warningIndicator = this.workspace.getWarningIndicator();
  if (warningIndicator) {
    warningIndicator.updateErrorCount();
  }
}
 
Blockly.WarningHandler.prototype.checkBlockForErrors = function(block) {
  const errorHandlers = [this.checkComponentExists, this.checkComponentProperties, this.checkValidBlockForVariableInit, this.checkValidAsset, this.checkDataSourceExists, this.checkValidVariable, this.checkBreakInLoop, this.checkValidBlockForAsyncReturn, this.checkCallbackVarsInScope];
  let foundError = false;
  for (let i = 0; i < errorHandlers.length; i++) {
    foundError = errorHandlers[i](block);
    if (foundError) {
      if (!block.hasError) {
        block.hasError = true;
        this.errorCount += 1;
      }
      break;
    }
  }
  if (!foundError) {
    if (block.hasError) {
      block.hasError = false;
      this.errorCount -= 1;
    }
    block.setWarningText(null);
  }
  const warningIndicator = this.workspace.getWarningIndicator();
  if (warningIndicator) {
    warningIndicator.updateErrorCount();
  }
  return foundError;
}
 
// Check if component exists and if component type exists
Blockly.WarningHandler.prototype.checkComponentExists = function(block) {
  if (block.componentInfo) {
    let blockInvalid = false;
    for (let componentFieldName of Object.keys(block.componentInfo)) {
      const componentField = block.getField(componentFieldName);
      const componentOptionValues = componentField.getOptions().map(opt => opt[1]);
      const componentFieldValue = componentField.getValue();
      if (componentOptionValues.indexOf(componentFieldValue) === -1) {
        componentField.setValue(block.componentInfo[componentFieldName].savedName);
        blockInvalid = true;
      }
    }
    if (!blockInvalid) return false;
    block.setWarningText('Select a valid component');
    return true;
  } else if (componentBlockTypes.indexOf(block.type) !== -1) {
    // To-do: fix blocks inside componentBlockTypes + remove the if-statement
    const componentField = block.getField(block.fieldNames.componentId);
    if (componentField) {
      const componentValue = block.getFieldValue(block.fieldNames.componentId)
      const componentOptions = componentField.getOptions();
      const componentValues = componentOptions.map(opt => opt[1]);
      if (componentValues.indexOf(componentValue) === -1) {
        componentField.setValue(block.componentName);
        block.setWarningText('Select a valid component');
        return true;
      } else {
        return false;
      }
    }
 
    const componentTypeField = block.getField(block.fieldNames.componentType);
    if (componentTypeField) {
      const componentTypeValue = block.getFieldValue(block.fieldNames.componentType);
      const componentTypeOptions = componentTypeField.getOptions();
      const componentTypeValues = componentTypeOptions.map(opt => opt[1]);
      if (componentTypeValues.indexOf(componentTypeValue) === -1) {
        block.setWarningText('Select a valid component type');
        return true;
      } else {
        return false;
      }
    }
  }
}
 
Blockly.WarningHandler.prototype.checkComponentProperties = function(block) {
  const dependantFieldName = componentBlockTypeToDependantFieldName[block.type];
  if (dependantFieldName) {
    const dependantField = block.getField(dependantFieldName);
    if (dependantField) {
      const propertyOptions = dependantField.getOptions();
      const propertyValues = propertyOptions.map(opt => opt[1]);
      if (propertyValues.indexOf(block.getFieldValue(dependantFieldName)) === -1) {
        block.setWarningText('Select a valid property');
        return true;
      } else {
        return false;
      }
    }
  }
}
 
Blockly.WarningHandler.prototype.checkValidBlockForVariableInit = function(block) {
  if ((componentBlockTypes.indexOf(block.type) !== -1 ||
      block.type === 'variables_get' ||
      block.type === 'variables_set' ||
      block.type === 'math_change' ||
      block.type === 'procedures_callnoreturn' ||
      block.type === 'procedures_callreturn') &&
        block.getRootBlock().type === 'variables_def'
    ) {
    block.setWarningText('This block cannot be in the initialize variable block');
    return true;
  } else {
    return false;
  }
}
 
Blockly.WarningHandler.prototype.checkValidVariable = function(block) {
  if (block.type !== 'variables_get' &&
      block.type !== 'variables_set' &&
      block.type !== 'math_change') {
    return false;
  }
  const variableOptions = block.getField('VAR').getOptions();
  let foundVariable = false;
  variableOptions.forEach(varOption => {
    if (varOption[0] === block.getFieldValue('VAR') || varOption[1] === block.getFieldValue('VAR')) {
      foundVariable = true;
    }
  });
  if (foundVariable) {
    return false;
  } else {
    block.setWarningText('Select a valid variable');
    if (block.savedName) {
      block.setFieldValue('‏‏‎ ‎' + block.savedName, 'VAR');
    }
    return true;
  }
}

    return true;
  } else {
    return false;
  }
}

    }
  }
};
 
Blockly.WarningHandler.prototype.checkDataSourceExists = function(block) {
  if (block.dataSourceInfo) {
    let blockInvalid = false;
    for (let fieldName of Object.keys(block.dataSourceInfo)) {
      const field = block.getField(fieldName);
      const fieldOptionValues = field.getOptions().map(opt => opt[1]);
      const fieldValue = field.getValue();
      if (fieldOptionValues.indexOf(fieldValue) === -1) {
        field.setValue(block.dataSourceInfo[fieldName]);
        blockInvalid = true;
      }
    }
    if (!blockInvalid) return false;
    block.setWarningText('Select a valid data source');
    return true;
  }
}
 
Blockly.WarningHandler.prototype.checkValidBlockForAsyncReturn = function(block) {
  let rootBlock = block.getRootBlock();
  if (block.type === 'procedures_return' &&
    rootBlock.type !== 'procedures_defasyncreturn' &&
    rootBlock.type !== 'procedures_defreturn' &&
    rootBlock.type !== 'procedures_defnoreturn' &&
    rootBlock.previousConnection === null) {
    block.setWarningText('This block can only be used in a procedure block');
    return true;
  } else if (block.type === 'procedures_defasyncreturn') {
    let children = block.getChildren();
    let nextChildren = [];
    while (children.length) {
      for (let child in children) {
        let childBlock = children[child];
        if (childBlock.type === 'procedures_return') {
          return false
        } else {
          nextChildren = nextChildren.concat(childBlock.getChildren());
        }
      }
      children = nextChildren;
      nextChildren = [];
    }
    block.setWarningText('This block is missing a return block');
    return true;
  }
  return false;
}
 
Blockly.WarningHandler.prototype.checkBreakInLoop = function(block) {
  if (block.type === 'controls_flow_statements') {
 
    // if block won't be executed, there shouldn't be an error
    if (block.getRootBlock().previousConnection) {
      return false;
    }
 
    let surroundParentBlock = block.getSurroundParent();
    let inLoop = false;
    while (surroundParentBlock) {
      if (surroundParentBlock.isLoop) {
        inLoop = true;
        break;
      }
      surroundParentBlock = surroundParentBlock.getSurroundParent();
    }
    if (inLoop) {
      return false;
    }
    block.setWarningText('Block must be in a loop');
    return true;
  }
}
 
Blockly.WarningHandler.prototype.checkCallbackVarsInScope = function(block) {
  if (block.type === 'callback_arg' && block.outputConnection && block.outputConnection.targetConnection && block.isDetachedParameter && !block.getRootBlock().previousConnection) {
    let surroundParentBlock = block.getSurroundParent();
    let inScope = false;
    while (surroundParentBlock) {
      if (blockTypesWithCallbackParams.indexOf(surroundParentBlock.type) !== -1) {
        // eslint-disable-next-line no-loop-func
        surroundParentBlock.inputList.forEach((input) => {
          if (input.connection && input.connection.targetConnection && input.connection.targetConnection.getSourceBlock().languageNeutralName === block.languageNeutralName) {
            inScope = true;
          }
        });
      }
      surroundParentBlock = surroundParentBlock.getSurroundParent();
    }
    if (inScope) {
      return false;
    }
    block.setWarningText(t('blocks.warning.variable-out-of-scope'));
    return true;
  }
}
