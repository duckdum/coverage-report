import React, { Component } from 'react';
import { COMPONENTS } from '../../thunkableComponents';
import GalleryItemLegacy from './GalleryItemLegacy';
import { t } from 'i18next';
import { Menu, Input, Button, Tooltip } from 'antd/lib';
import '../../style/ComponentGalleryLegacy.scss';
import ModuleUtils, { TYPES } from '../../shared/utils/ModuleUtils';
import ChooseScreenModal from '../ChooseScreenModal';
import moduleIcon from '../../images/viewIconModule.svg';
import * as PropTypeUtils from '../../shared/componentDefinitions/index';
import { RELEASE_ACCESS } from '../../shared/utils/FeatureUtils';
 
const { SubMenu } = Menu;
const DATA_INDEX = 'data-index';
 
export const CATEGORY_TO_LABEL = {
  'USER_INTERFACE': "design.UserInterface",
  'LAYOUT': "design.Layout",
  'VOICE': "design.Voice",
  'IMAGE': "design.Image",
  'DATA': "design.Data",
  'LOCATION': "design.Location",
  'SENSORS': "design.Sensors",
  'SOCIAL': "design.Social",
  'AUTHENTICATION': "design.Authentication",
  'MONETIZATION': "design.Monetization",
  "MODULES": "design.Modules.categoryLabel"
}
 
class GalleryCategory extends Component {
 
  state = {
    filter: undefined,
    showChooseModulesToAddModal: false,
  }
 
  binaryInsert(galleryItem, category, startIdx, endIdx) {
    const length = category.length;
    const start = typeof(startIdx) !== 'undefined' ? startIdx : 0;
    const end = typeof(endIdx) !== 'undefined' ? endIdx : length - 1;;
    const mid = start + Math.floor((end - start) / 2);
    const dataIndex = galleryItem.props[DATA_INDEX];
 
    if (length === 0) {
      category.push(galleryItem);
    } else if (dataIndex >= category[end].props[DATA_INDEX]) {
      category.splice(end + 1, 0, galleryItem);
    } else if (dataIndex <= category[start].props[DATA_INDEX]) {
      category.splice(start, 0, galleryItem);
    } else if (dataIndex === category[mid].props[DATA_INDEX]) {
      category.splice(mid, 0, galleryItem);
    } else if (dataIndex < category[mid].props[DATA_INDEX]) {
      this.binaryInsert(galleryItem, category, start, mid - 1);
    } else if (dataIndex > category[mid].props[DATA_INDEX]) {
      this.binaryInsert(galleryItem, category, mid + 1, end);
    }
    return;
  }
 
  filterComponentsTextChanged = (event) => {
    this.setState({filter: event.currentTarget.value.toLowerCase()})
  }
 
  filterOutForModules = (isModule, type, isApiModule) => {
    if (!isModule) return false;
    if (PropTypeUtils.isScreen(type) || PropTypeUtils.isNavigator(type)) return true;
    if (isApiModule && !COMPONENTS[type].IsApi) return true;
    return false;
  }
 
  toggleShowChooseModulesToAddModal = () => {
    this.setState({ showChooseModulesToAddModal: !this.state.showChooseModulesToAddModal });
  }
 
  addToModulesCategory = (moduleCategory, modulesAddedToProject, numOfThunkableComponents) => {
    if (!modulesAddedToProject) return;
    for (let i = 0; i < modulesAddedToProject.length; i++) {
      const mod = modulesAddedToProject[i];
      this.addToCategoryArray(mod.id, "module:" + mod.id, numOfThunkableComponents + 1 + i, moduleIcon, mod.name, mod.name, moduleCategory);
    }
  }
 
  addToCategoryArray = (key, type, dataIndex, icon, label, translateLabelKey, category) => {
    this.binaryInsert((
      <div className="gallery-item-wrapper"
           key={key}
           data-index={dataIndex}>
        <GalleryItemLegacy
          icon={icon}
          label={label}
          type={type}
          trackingData={this.props.trackingData}
          translateLabelKey={translateLabelKey}
          thunkableVersion={this.props.thunkableVersion}
        />
      </div>), category
    );
  }
 
  render() {
    const isApiComponentInBlocksFeatureEnabled = this.props.isDndEditor;
    let categories = {};
    let CATEGORIES = Object.keys(CATEGORY_TO_LABEL);
    if (isApiComponentInBlocksFeatureEnabled) CATEGORIES = ['USER_INTERFACE', 'LAYOUT'];
    CATEGORIES.forEach(category => categories[category] = []);
    let numOfThunkableComponents = 0;
    this.props.types.forEach((type, index) => {
      const hasVersioning =
        this.props.thunkableVersion !== null && COMPONENTS[type].ThunkableVersion !== undefined;
      const isSupportedVersion = COMPONENTS[type].ThunkableVersion <= this.props.thunkableVersion;
      let category = categories[COMPONENTS[type].Category];
      if (isApiComponentInBlocksFeatureEnabled && COMPONENTS[type].UIComponentOrderIndex !== undefined) {
        category = categories['USER_INTERFACE'];
      }
      const searchTags = t(COMPONENTS[type].SearchTags) ? t(COMPONENTS[type].SearchTags).split(',') : [];
      const translatedSearchTags = searchTags.map((tag) => t(tag).toLowerCase());
      const lowerCaseTranslateLabelKey = (COMPONENTS[type].TranslateLabelKey
          ? t(COMPONENTS[type].TranslateLabelKey) : type).toLowerCase();
      const shouldNotRender = !translatedSearchTags.some((tag) =>
        tag.includes(this.state.filter)) && !lowerCaseTranslateLabelKey.includes(this.state.filter);
      if (this.state.filter && shouldNotRender) return;
      if (this.filterOutForModules(this.props.isModule, type, this.props.isApiModule)) return;
      if (COMPONENTS[type].ExcludeInDnd && this.props.isDndEditor) return;
      if (COMPONENTS[type].ExcludeInSnapToPlace && !this.props.isDndEditor) return;
      if (isApiComponentInBlocksFeatureEnabled && COMPONENTS[type].ManageInBlocks) return;
      if (
        (!hasVersioning || (hasVersioning && isSupportedVersion))
        && category
      ) {
        let componentIndex;
        if (isApiComponentInBlocksFeatureEnabled && COMPONENTS[type].UIComponentOrderIndex !== undefined) {
          componentIndex = COMPONENTS[type].UIComponentOrderIndex;
        } else {
          componentIndex = COMPONENTS[type].Index;
        }
        const dataIndex = CATEGORIES.indexOf(category) * 20 + componentIndex;
        numOfThunkableComponents += 1;
        this.addToCategoryArray(type, type, dataIndex, COMPONENTS[type].Icon, COMPONENTS[type].Label, COMPONENTS[type].TranslateLabelKey, category);
      }
    });
    this.addToModulesCategory(categories['MODULES'], this.props.modulesAddedToProject, numOfThunkableComponents);
    const populatedCategories = CATEGORIES.filter(category => categories[category].length > 0);
    const items = populatedCategories.map((category, index) => {
      return (<SubMenu title={t(CATEGORY_TO_LABEL[category])} key={category}>
                <Menu.Item key={index}>
                  <div className="gallery-items">
                    {categories[category]}
                  </div>
                </Menu.Item>
              </SubMenu>);
    });
 
    return (
      <div className="component-gallery-container legacy">
        <div className="component-gallery-header-container">
          <div className="component-gallery-header">{t("design.AddComponents")}</div>
          {/* TODO: Add back allowClear prop once we update antd */}
          <Input.Search
            data-testid="search-components"
            placeholder={t("design.SearchComponents")}
            onChange={this.filterComponentsTextChanged}
          />
        </div>
        <div className="component-gallery-panel">
          <Menu mode="inline" defaultOpenKeys={CATEGORIES}>
          {items}
          </Menu>
          {this.props.thunkableVersion >= RELEASE_ACCESS.DEVELOPERS_ONLY && ModuleUtils.hasComponentTypeModules(this.props.modules) && !this.props.isModule &&
          <div className="component-gallery-panel-add-module-container">
            <Tooltip placement="right" title={t('design.Modules.addModuleButtonTooltip')}>
              <Button onClick={this.toggleShowChooseModulesToAddModal}>
                {t('design.Modules.addModuleButtonText')}
              </Button>
            </Tooltip>
            <ChooseScreenModal
              modalTitle={t('projects.chooseModules.modalTitle')}
              showChooseScreenOrModuleModal={this.state.showChooseModulesToAddModal}
              closeChooseScreenOrModuleModal={this.toggleShowChooseModulesToAddModal}
              onClickCancel={this.toggleShowChooseModulesToAddModal}
              modules={ModuleUtils.getModulesOfType(this.props.modules, TYPES.OTHERS)}
              addModulesOrSavedScreens={this.props.addModuleToProject}
              project={this.props.project}
              cancelText={t('projects.chooseScreens.cancelText')}
              okText={t('projects.chooseScreens.okText')}
              typeOfItemToAdd="modules"
              projectId={this.props.projectId}>
            </ChooseScreenModal>
          </div>}
        </div>
 
      </div>
      );
  }
}
 
export default GalleryCategory;
