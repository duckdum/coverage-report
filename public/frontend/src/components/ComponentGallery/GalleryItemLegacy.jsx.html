import React, { Component } from 'react';
import SortableView from '../Designer/SortableView';
import { Popover, Tooltip, Button, Icon } from 'antd/lib';
import viewIconInfo from '../ComponentGallery/viewIconInfoLegacy.svg';
import { COMPONENTS } from '../../thunkableComponents';
import { RELEASE_ACCESS } from '../../shared/utils/FeatureUtils';
import { t } from 'i18next';
import { TRACKING_EVENT_TYPES, track } from '../../tracking';
 
class GalleryItem extends Component {
  constructor(props) {
    super(props);
    this.state = {
      
      viewIconInfoSrc: null,
      
    };
  }
  
  handleMouseOver = (e) => {
    if (!this.props.active) {
      this.setState({ viewIconInfoSrc : viewIconInfo });
    } else {
      this.setState({ viewIconInfoSrc : null });
    }
  }
  handleMouseOut = (e) => {
    this.setState({ viewIconInfoSrc : null });
  }
 
  setRef = (element) => {
    if (element) {
      this.clientRect = element.getBoundingClientRect();
    }
  };
 
  trackClick = () => {
    const properties = {
      component: this.props.type,
      url: t(COMPONENTS[this.props.type].DocumentationURL),
      ...this.props.trackingData
    };
 
    track(TRACKING_EVENT_TYPES.COMPONENT_LEARN_MORE_CLICKED, properties);
  }
 
  render() {
    const isDragging = this.props.isDragging;
    const connectDragSource = this.props.connectDragSource;
    const viewIconInfoSrc = this.state.viewIconInfoSrc;
    const className = 'gallery-item' + (isDragging ? ' dragging':'');
    let name = this.props.type;
 
    if (this.props.translateLabelKey !== undefined) {
      name = t(this.props.translateLabelKey);
    } else if (this.props.label !== undefined) {
      name = this.props.label;
    } 
   
    const popoverContent = (
      <div style={{ maxWidth: '300px' }}>
        {COMPONENTS[this.props.type] ? t(COMPONENTS[this.props.type].TranslateTooltipKey) : ''}
        {
          COMPONENTS[this.props.type] && COMPONENTS[this.props.type].DocumentationURL && (
            <>
              <br />
              <a
                onClick={this.trackClick}
                href={t(COMPONENTS[this.props.type].DocumentationURL)}
                target="_blank" rel="noopener noreferrer">
                Learn more
              </a>
            </>
          )
        }
      </div>)
    
    return connectDragSource((
      <div ref={this.setRef} style={{}} className={className} onMouseOver={this.handleMouseOver} onMouseOut={this.handleMouseOut}>
        <img className="icon" alt="Component Icon" src={this.props.icon} />
        {name}
        { COMPONENTS[this.props.type] && COMPONENTS[this.props.type].ThunkableVersion === RELEASE_ACCESS.PUBLIC_BETA &&
            <Tooltip
              placement="right"
              title="This component is only available to Private Beta users.">
              <Button 
                className='live-test-info-button' 
                shape="round" 
                style={styles.private_beta_button}>
                Private Beta
                <Icon type="info-circle-o" style={styles.private_beta_icon} />
              </Button>
            </Tooltip>
        }  
        {
          <Popover display= {this.state.viewIconInfoSrc ? 'flex' : 'none'} placement= "right"
            title= {name} trigger= "click" content={popoverContent}>
                  <img alt= {viewIconInfoSrc}
                  style={{marginLeft: 'auto', marginRight: 10}}
                  src={viewIconInfoSrc}
                  />
            </Popover>
            }
      </div>
    ), { dropEffect: 'copy' });
  }
}
 
const styles = {
  private_beta_button: {
    marginLeft: 22,
    height: 20, 
    paddingLeft: 8, 
    paddingRight: 8, 
    fontSize: 13, 
    fontWeight: 300, 
    border: 0, 
    backgroundColor: "#f0f1ff", 
    color: "#4D5EAC", 
    boxShadow: "none"
  },
  private_beta_icon: {
    marginRight: 0,
    marginLeft: 4, 
    fontSize: 11, 
    fontWeight: 300, 
    color: "rgba(77, 94, 172, 0.5)"
  }
}
const dndGalleryItem = SortableView(GalleryItem, 'NewComponent', ['ComponentTreeNode', 'MockComponent']);
export const GalleryItemUI = GalleryItem;
export default dndGalleryItem;
