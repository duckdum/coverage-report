import React, { useState, useEffect } from 'react';
import gql from 'graphql-tag';
import {
  Mutation,
  compose,
  graphql,
  withApollo
} from 'react-apollo';
import { get } from 'lodash';
import { Button, Form, Input, Spin, Select, Popover, Notification, Switch } from 'antd/lib';
import { t } from 'i18next';
 
const DEFAULT_VIEW_NAME = 'Grid view';
 
const InputGroup = Input.Group;
const FormItem = Form.Item;
const Option = Select.Option;
 
const CREATE_AIRTABLE_CONFIGURATION = gql`
mutation($input: CreateAirtableDataSourceConfigurationInput!) {
  createAirtableDataSourceConfiguration(input: $input) {
    dataSource {
      id
    }
  }
}
`;
 
const USER_QUERY = gql`
query User {
  user {
    id
    airtable
  }
}
`;
 
const GET_POTENTIAL_DATA_SOURCES = gql`
query getPotentialDataSources($input: GetPotentialDataSourcesInput!) {
  getPotentialDataSources(input: $input) {
    potentialDataSources {
      id
      name
    }
  }
}
`;
 
const SET_USER_AIRTABLE_API_KEY = gql`
mutation UpdateAirtableApiKey($apiKey: String!) {
  updateAirtableApiKey(apiKey: $apiKey) {
    apiKey
  }
}
`;
 
const AirtablePicker = ({
  client,
  createDataSourceAndSync,
  data,
  owner,
}) => {
  const defaultBase = { key: '', value: '' };
  const initialApiKey = get(data, 'user.airtable.apiKey');
  const [apiKey, setApiKey] = useState(initialApiKey);
  const [potentialDataSources, setPotentialDataSources] = useState([]);
  const [selectedBase, setSelectedBase] = useState(defaultBase);
  const [inputValue, setInputValue] = useState(initialApiKey || '');
  const [viewValue, setViewValue] = useState(DEFAULT_VIEW_NAME);
  const [useCustomView, setUseCustomView] = useState(false);
  const [isRefreshing, setIsRefreshing] = useState(false);
 
  useEffect(() => {
    if (apiKey) {
      updatePotentialDataSources();
    }
  },[]);
 
  // The early exit conditions must be under all the hooks
  if (data.error) return <div>{t('errors.generic.tryAgain')}</div>;
  if (data.loading) return <Spin size="large" />;
  const inputsAreValid = () => viewValue && selectedBase.key && selectedBase.label && owner;
 
  const createDataSource = (configMutation) => {
    const { key: baseId, label: name } = selectedBase;
    const dataSource = {
      name,
      owner,
      configuration: {
        airtableDataSourceConfiguration: {
          name,
          owner,
          apiKey,
          baseId,
          viewName: viewValue
        }
      }
    };
    if (inputsAreValid()) {
      setIsRefreshing(true);
      createDataSourceAndSync(dataSource, configMutation);
    } else {
      const message = t('dataSources.airtable.invalidData');
      const description = t('dataSources.airtable.invalidDataDescription');
      handleError(message, description);
    }
  };
 
  const handleError = (message, description) => {
    setIsRefreshing(false);
    Notification.error({
      message,
      description,
      placement: 'topRight',
      duration: 0,
    });
  };
 
  const updatePotentialDataSources = async () => {
    const queryOptions = {
      query: GET_POTENTIAL_DATA_SOURCES,
      variables: {
        input: {
          type: 'AIRTABLE',
        }
      },
      fetchPolicy: 'network-only'
    };
    try {
      const result = await client.query(queryOptions);
      const newPotentialDataSources = get(result, 'data.getPotentialDataSources.potentialDataSources', []);
      setPotentialDataSources(newPotentialDataSources);
      setSelectedBase(defaultBase);
      setIsRefreshing(false);
    } catch (error) {
      handleError(t('dataSources.airtable.invalidApiKey'), t('dataSources.airtable.checkApiKey'));
      setPotentialDataSources([]);
    }
  };
 
  const onClickRefresh = async () => {
    if (apiKey !== inputValue) {
      setIsRefreshing(true);
      setApiKey(inputValue);
      const mutationOptions = {
        mutation: SET_USER_AIRTABLE_API_KEY,
        variables: {
          apiKey: inputValue
        },
      };
      try {
        await client.mutate(mutationOptions);
        updatePotentialDataSources();
      } catch (error) {
        handleError(t('dataSources.airtable.invalidApiKey'), t('dataSources.airtable.checkApiKey'));
        setPotentialDataSources([]);
      }
    }
  };
 
  const onChangeUseCustomView = () => {
    const newValue = !useCustomView;
    setUseCustomView(newValue);
    if (!newValue) {
      setViewValue(DEFAULT_VIEW_NAME);
    }
  }
 
  return (
    <Mutation mutation={CREATE_AIRTABLE_CONFIGURATION} key="config-mutation">
      {createAirtableConfiguration => (
          <div style={{ display: 'flex', flexDirection: 'column' }}>
            <div>
              {t('dataSources.airtable.signUpForFree')}
              <a href="https://airtable.com/thunkable" rel="noopener noreferrer" target="_blank">
                {t('confirmation.here')}
              </a>.
            </div>
            <div>
              {t('dataSources.airtable.getApiKey')}
              <a href="https://airtable.com/account" rel="noopener noreferrer" target="_blank">
                {t('dataSources.airtable.yourAccountPage')}
              </a>.
            </div>
            <FormItem label={t('dataSources.airtable.apiKey')}>
              <InputGroup compact style={{ display: 'flex' }}>
                <Input value={inputValue} onChange={(event) => setInputValue(event.target.value)}/>
                <Popover title="Refresh" content={<div>{t('dataSources.airtable.updateApiKey')}</div>}>
                  <Button style={{ height: 'auto' }}type='primary' onClick={onClickRefresh}>{t('confirmation.refresh')}</Button>
                </Popover>
              </InputGroup>
            </FormItem>
            <div style={{ width: '100%', display: 'flex' }}>
              <FormItem label={t('dataSources.airtable.selectBase')}>
                <Select
                  labelInValue
                  placeholder={t('dataSources.airtable.selectBase')}
                  notFoundContent={t('dataSources.airtable.refreshBases')}
                  onChange={(value) => setSelectedBase(value)}
                  value={selectedBase}
                >
                  {potentialDataSources.map(({ id, name }) => (<Option key={id} value={id}>{name}</Option>))}
                </Select>
              </FormItem>
            </div>
            <FormItem label={t('dataSources.airtable.customView')}>
              <Switch value={useCustomView} onChange={onChangeUseCustomView} />
              <Input value={viewValue} disabled={!useCustomView} onChange={(event) => setViewValue(event.target.value)}/>
            </FormItem>
            {isRefreshing
              ? <Spin size="large" />
              : <FormItem>
                  <div style={{display: 'flex', justifyContent: 'flex-end'}}>
                    <Button
                      type='primary'
                      disabled={!(apiKey && selectedBase)}
                      onClick={() => createDataSource(createAirtableConfiguration)}
                    >
                      {t('confirmation.create')}
                    </Button>
                  </div>
                </FormItem>}
          </div>
      )}
    </Mutation>
  )
};
 
export default compose(
  graphql(USER_QUERY),
  graphql(CREATE_AIRTABLE_CONFIGURATION),
)(withApollo(AirtablePicker));
