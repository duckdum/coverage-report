  componentDidUpdate(prevProps) {
    // This test might cause some unnecessary reloading but for now I think it's ok.
    if (!this.shouldUpdateCompanion) {
      this.shouldUpdateCompanion = !this.shouldUpdateCompanion;
      return;
    }
    if (this.props.screenId !== prevProps.screenId) {
      this.sendProjectDataToCompanion();
    } else if (this.props.project.defaultDesignerDevice !== prevProps.project.defaultDesignerDevice ||
        this.props.project.defaultDesignerOrientation !== prevProps.project.defaultDesignerOrientation) {
          this.sendProjectDataToCompanion();
    } else if (this.props.selectedComponentId !== prevProps.selectedComponentId) {
      // only update if either the previous or current selected component is on the screen
      const previousSelectedComponentScreen = ComponentUtils.getScreenForComponent(prevProps.selectedComponentId, prevProps.components);
      const currentSelectedComponentScreen = ComponentUtils.getScreenForComponent(this.props.selectedComponentId, this.props.components);
      if (previousSelectedComponentScreen.id === this.props.screenId || currentSelectedComponentScreen.id === this.props.screenId) {
        this.sendProjectDataToCompanion();
      }
    } else if (this.props.components !== prevProps.components) {
      // only update if the screen has changed, or if there was a change in the navigators containing the parent
      const previousScreen = ComponentUtils.getScreenForComponent(this.props.screenId, prevProps.components);
      const currentScreen = ComponentUtils.getScreenForComponent(this.props.screenId, this.props.components);
      if (!isEqual(previousScreen, currentScreen)) {
        this.sendProjectDataToCompanion();
      } else {
        let shouldUpdateFromNavigatorChange = false;
        // update if a screen or parent navigator changed in a way that requires re-rendering
        const idToParentId = ComponentUtils.getParentIdsById(this.props.components);
        const screenParentArray = idToParentId.get(this.props.screenId);
        screenParentArray.forEach(navigatorId => {
          const navigatorComponent = ComponentUtils.getComponentById(navigatorId, this.props.components, {});
          navigatorComponent.children.forEach(screen => {
            const oldScreen = ComponentUtils.getComponentById(screen.id, prevProps.components, {});
            const newScreen = ComponentUtils.getComponentById(screen.id, this.props.components, {});
            if (oldScreen.name !== newScreen.name) {
              shouldUpdateFromNavigatorChange = true
            } else {
              // compare navigation options object
              if (get(oldScreen, 'properties.navigationOptions', null) !== get(newScreen, 'properties.navigationOptions', null)
                || get(oldScreen, 'properties.tabNavigationOptions', null) !== get(newScreen, 'properties.tabNavigationOptions', null)) {
                  shouldUpdateFromNavigatorChange = true;
                }
            }
          })
        });
        if (shouldUpdateFromNavigatorChange) this.sendProjectDataToCompanion();
      }
    }
  }
 
  onMessageFromCompanion(message) {
    if (message.screenId === this.props.screenId) {
      if (message.type === "changePosition") {
        // TODO fix bug so group dragging works
        // should update the companion state if there is a container id
        this.shouldUpdateCompanion = true; //TODO new value: !!message.containerId;
        this.changeComponentPosition(message);
      } else if (message.type === 'requestProjectData') {
        this.sendProjectDataToCompanion();
      } else if (message.type === 'requestScreenSwitch') {
        this.shouldUpdateCompanion = true;
        this.props.setSelectedComponentId(message.screenId)
      } else if (message.type === "resize") {
        this.changeComponentPositionAndDimensions(message);
      } else if (message.type === 'wheelInSimplePhone') {
        const event = new CustomEvent('wheelFromSimpleScreen', {
          detail: {
            deltaX: message.deltaX,
            deltaY: message.deltaY,
            screenId: message.screenId,
          }
        });
        const transformWrapper = document.getElementsByClassName('react-transform-component TransformComponent-module_container__3NwNd ')[0];
        transformWrapper.dispatchEvent(event);
      } else if (message.type === 'selectComponent') {
        this.shouldUpdateCompanion = true;
        this.props.setSelectedComponentId(message.componentId);
      } else if (message.type ==="keydownInSimplePhone") {
        const event = new CustomEvent('keydownFromSimpleScreen', {
          detail: {
            ctrlKey: message.ctrlKey,
            key: message.key,
            metaKey: message.metaKey,

const calculateRelativePosition = (y, height, parentHeight, x, width, parentWidth) => {
  /*
   Generally, we want to use relative, absolute positions (by which we mean 'absolute' CSS positioning with percentage
   units) to account for different device sizes.  We also want to account for the fact that the dimensions of the
   component at design time may not match the 'real' dimensions at runtime. So, for example, if the user positions a
   component with the right edge near the right edge of the screen, we assume that the user really wants that component
   near the right edge of the screen, not some relative distance from the left side.
 
   The basic heuristic here is that if the left edge of the component is left of the midpoint of the screen (i.e. the
   parent) then we calculate a relative position with respect to the left side of the screen.  If the right edge of the
   component is to the right of the midpoint then we calculate a position relative to the right edge of the screen.  If
   both are true (i.e. the component straddles the midline) then we give preference to the left edge.
 
   We do a similar calculation with respect to the top and bottom of the component and screen, giving preference to the
   top in the case that the component straddles the vertical midline of the screen.
   */
  const edgeOffsets = {};
  const leftPercentage = Math.max(100 * x / parentWidth, 0);
  edgeOffsets.left = `${leftPercentage}%`;
  const topPercentage = Math.max(100 * y / parentHeight, 0);
  edgeOffsets.top = `${topPercentage}%`;
  return edgeOffsets;
};

const dropExec = (props, monitor, targetComponent) => {
  const targetRect = findDOMNode(targetComponent).getBoundingClientRect();
  const sourceX = monitor.getSourceClientOffset().x - targetRect.left;
  const sourceY = monitor.getSourceClientOffset().y - targetRect.top;
  const sourceComponentClientRect = monitor.getItem().clientRect;
  const sourceWidth = sourceComponentClientRect.right - sourceComponentClientRect.left;
  const sourceHeight = sourceComponentClientRect.bottom - sourceComponentClientRect.top;
  // TODO: We might want to put a check here at some point that the component's drop position is fully within the Screen
  props.addElement(
    monitor.getItem().type,
    props.screenId,
    sourceX,
    sourceY,
    sourceWidth,
    sourceHeight,
    targetRect.width,
    targetRect.height);
  return true;
};
