const mapStateToProps = (state, ownProps) => {
  const project = state.projects[ownProps.projectOrModuleId];
  const screen = ComponentUtils.getScreenForComponent(ownProps.selectedComponentId, project.components);
  const screens = ComponentUtils.getComponentsByType(project, TScreen.Type);
  const screenIds = [];
  const screenIdsWithDimensions = screens.map(screen => {
    screenIds.push(screen.id);
    return ({
      id: screen.id,
      responsiveWidth: screen.properties.ResponsiveWidth || DEFAULT_PHONE_WIDTH,
      designerHeight: screen.properties.DesignerHeight,
      designerWidth: screen.properties.DesignerWidth,
    });
  });
 
  let screenId = screen.id || screenIds[0];
  // when the selected component is a navigator
  if (isBoolean(screen) && screens.length > 0) {
    const navigator = ComponentUtils.getComponentById(ownProps.selectedComponentId, project.components, []);
    screenId = ComponentUtils.findFirstComponentOfType(navigator, TScreen.Type).id || screenIds[0];
  }
 
  const previewMove = state.projectConfigs[ownProps.projectOrModuleId]
    && state.projectConfigs[ownProps.projectOrModuleId].previewMoveNode;
  const isOwner = project.isOwner;
  const webViewer = ComponentUtils.getComponentsByType(project, TWebView.Type);
  project.hasWebViewer = webViewer && webViewer.length > 0;
  return {
    isOwner,
    isPreviewMode: state.setIsPreviewMode.isPreviewMode,
    isTutorialPanelVisible: state.setTutorialPanelVisibility.isTutorialPanelVisible,
    previewMove,
    project,
    screenId,
    screenIds,
    screenIdsWithDimensions,
  }
}
