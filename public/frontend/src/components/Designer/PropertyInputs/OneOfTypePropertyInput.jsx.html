const isNode = (propValue) => {
  const ITERATOR_SYMBOL = typeof Symbol === 'function' && Symbol.iterator;
  const FAUX_ITERATOR_SYMBOL = '@@iterator'; // Before Symbol spec.
 
  /**
   * Returns the iterator method function contained on the iterable object.
   *
   * Be sure to invoke the function with the iterable as context:
   *
   *     var iteratorFn = getIteratorFn(myIterable);
   *     if (iteratorFn) {
   *       var iterator = iteratorFn.call(myIterable);
   *       ...
   *     }
   *
   * @param {?object} maybeIterable
   * @return {?function}
   */
  function getIteratorFn(maybeIterable) {
    // eslint-disable-next-line no-mixed-operators
    const iteratorFn = maybeIterable && (ITERATOR_SYMBOL && maybeIterable[ITERATOR_SYMBOL] || maybeIterable[FAUX_ITERATOR_SYMBOL]);
    if (typeof iteratorFn === 'function') {
      return iteratorFn;
    }
  }
 
  switch (typeof propValue) {
    case 'number':
    case 'string':
    case 'undefined':
      return true;
    case 'boolean':
      return !propValue;
    case 'object':
      if (Array.isArray(propValue)) {
        return propValue.every(isNode);
      }
      if (propValue === null || React.isValidElement(propValue)) {
        return true;
      }
 
      const iteratorFn = getIteratorFn(propValue);
      if (iteratorFn) {
        const iterator = iteratorFn.call(propValue);
        let step;
        if (iteratorFn !== propValue.entries) {
          // eslint-disable-next-line no-cond-assign
          while (!(step = iterator.next()).done) {
            if (!isNode(step.value)) {
              return false;
            }
          }
        } else {
          // Iterator will provide entry [k,v] tuples rather than values.
          // eslint-disable-next-line no-cond-assign
          while (!(step = iterator.next()).done) {
            const entry = step.value;
            if (entry) {
              if (!isNode(entry[1])) {
                return false;
              }
            }
          }
        }
      } else {
        return false;
      }
 
      return true;
    default:
      return false;
  }
};

const propTypePrecedence = (propType) => {
  if (propType.descriptor.fixedValue) {
    return 4;
  }
  switch (propType.descriptor.type){
    case PROPERTY_TYPES.oneOfWithLabel:
    case PROPERTY_TYPES.instanceOf:
    case PROPERTY_TYPES.objectOf:
    case PROPERTY_TYPES.oneOf:
    case PROPERTY_TYPES.shape:
    case PROPERTY_TYPES.symbol:
    case PROPERTY_TYPES.element:
      return 3;
    case PROPERTY_TYPES.oneOfType:
    case PROPERTY_TYPES.arrayOf:
    case PROPERTY_TYPES.node:
    case PROPERTY_TYPES.array:
    case PROPERTY_TYPES.bool:
    case PROPERTY_TYPES.func:
    case PROPERTY_TYPES.funcForEvent:
    case PROPERTY_TYPES.number:
    case PROPERTY_TYPES.TPixelOrPercent:
      return 2;
    case PROPERTY_TYPES.object:
    case PROPERTY_TYPES.string:
    case PROPERTY_TYPES.secret:
    case PROPERTY_TYPES.any:
    default:
      return 1;
  }
};
 
const isOfPropType = (value, propType) => {
  if (propType.descriptor.fixedValue) {
    return value === propType.descriptor.fixedValue;
  }
  let childPropType;
  switch (propType.descriptor.type){
    case PROPERTY_TYPES.arrayOf:
      childPropType = propType.descriptor.params;
      return Array.isArray(value)
          && value.every(val => isOfPropType(val, childPropType));
    case PROPERTY_TYPES.instanceOf:
      childPropType = propType.descriptor.params;
      return value instanceof childPropType;
    case PROPERTY_TYPES.objectOf:
      childPropType = propType.descriptor.params;
      return typeof value === 'object'
          && Object.values(value).every(val => isOfPropType(val, childPropType));
    case PROPERTY_TYPES.oneOf:
      return propType.descriptor.params.find(paramVal => paramVal === value);
    case PROPERTY_TYPES.oneOfType:
      return propType.descriptor.params.find(param => isOfPropType(value, param));
    case PROPERTY_TYPES.oneOfWithLabel:
      //checks if the selected value matches one of the oneOfWithLabel options
      return propType.descriptor.params.find(paramVal => paramVal[1] === value);
    case PROPERTY_TYPES.shape:
      const shapeObj = propType.descriptor.params;
      const shapeKeys = Object.keys(shapeObj);
      return typeof value === 'object'
          && shapeKeys.every(key => value[key] !== undefined && isOfPropType(value[key], shapeObj[key]));
    case PROPERTY_TYPES.array:
      return Array.isArray(value);
    case PROPERTY_TYPES.bool:
      return typeof value === 'boolean';
    case PROPERTY_TYPES.func:
    case PROPERTY_TYPES.funcForEvent:
      return typeof value === 'function';
    case PROPERTY_TYPES.number:
      return typeof value === 'number'
          || (typeof value === 'string' && isDecimalNumber(value));
    case PROPERTY_TYPES.TPixelOrPercent:
      return typeof value === 'number' ||
            (typeof value === 'string' && (isDecimalNumber(value) || isPercentNumber(value)));
    case PROPERTY_TYPES.object:
      return typeof value === 'object';
    case PROPERTY_TYPES.string:
    case PROPERTY_TYPES.secret:
      return typeof value === 'string';
    case PROPERTY_TYPES.symbol:
      return typeof value === 'symbol';
    case PROPERTY_TYPES.any:
      return true;
    case PROPERTY_TYPES.element:
      return React.isValidElement(value);
    case PROPERTY_TYPES.node:
      return isNode(value);
    case PROPERTY_TYPES.fileName:
      return typeof value === 'string';
    default:
      return false;
  }
};
 
class OneOfTypePropertyInput extends Component {
 
  constructor(props) {
    super(props);
    
 
    this.state = {
        choiceIndex: props.value ? this.getChoiceIndex(props.value) : 0,
      };
  }
  
  valueChanged = (propName, newValue) => {
    this.props.valueChanged(this.props.name, newValue);
  }
 
  handleChange = (value, label) => {
    const choiceIndex = parseInt(value, 10);
    this.setState({ choiceIndex });
    // if choiceIndex refers to a value that is fixed, set the value to the fixed value
    const fixedValue = this.props.propType.descriptor.params[choiceIndex].descriptor.fixedValue;
    if (fixedValue) {
      this.props.valueChanged(this.props.name, fixedValue);
    }
  }
 
  getChoiceIndex = (chosenPropValue) => {
    // Loop through propType params until you find either a type match for chosenPropValue
    // or a match to a oneOf param.  The oneOf param will have precedence over a type match,
    // so we need to go through the entire params array  if we find a type match, in
    // order to continue looking for possible oneOf matches.  For oneOf matches we can
    // return immediately.
    const params = this.props.propType.descriptor.params;
    let choiceIndex;
    let currentHighestPrecedence = 0;
    const numParams = params.length;
    for (let i = 0; i < numParams; i++) {
      const precendence = propTypePrecedence(params[i]);
      if (precendence > currentHighestPrecedence && isOfPropType(chosenPropValue, params[i])) {
        choiceIndex = i;
        currentHighestPrecedence = precendence;
      }
    }
    return choiceIndex;
  };
 
  render() {
    const typeOptions = this.props.propType.descriptor.params;
    const optionDescriptions = typeOptions.map((option, index, array) => {
      return getStringDescription(option);
    })
    const typeOptionElements = optionDescriptions.map((description, index, array) => {
      return <Option key={index + description} value={index.toString()}>{description}</Option>
    })
 
    var name = optionDescriptions[this.state.choiceIndex];
    var chosenProptype = this.props.propType.descriptor.params[this.state.choiceIndex];
    var chosenPropTypeValue;
    if (this.props.value !== undefined) {
      chosenPropTypeValue = this.props.value;
    }
    let chosenInput;
    if (!chosenProptype.descriptor.fixedValue) {
      chosenInput =
        getInputComponent(
          name,
          chosenProptype,
          this.valueChanged,
          chosenPropTypeValue,
          null,
          null,
          this.props.componentId,
          true,
          this.props.readOnly,
          null,
          this.props.projectId,
          this.props.thunkableVersion,
          this.props.parentType,
          this.props.handleFileUpload,
          { displayAsFormItem: true }
        );
    }
    return (
      <div className="form-item-container">
        <FormItem>
          <Select
            className="properties-panel-dropdown"
            placeholder="select"
            value={String(this.state.choiceIndex)}
            onChange={this.handleChange}
            disabled={this.props.readOnly}
            suffixIcon={<SelectSuffixIcon />}
          >
            {typeOptionElements}
          </Select>
        </FormItem>
        { chosenInput }
      </div>
    );
  }
}
