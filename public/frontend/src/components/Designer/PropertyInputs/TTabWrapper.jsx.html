import React, { Component } from 'react';
import { Tabs } from 'antd/lib';
import { get } from 'lodash';
import { t } from 'i18next';
import { getInputComponent } from '../PropertyInputs';
import ImagePreview from './ImagePreview';
import { ThunkableText } from '../../ComponentLibrary';
const { TabPane } = Tabs;
 
class TTabWrapper extends Component {
  getTabInput = (prop) => {
    let currentPropTypeValue;
    const propTypeName = get(prop, 'type.descriptor.type');
    const planLimitKey = prop.type.descriptor.planLimitKey;
    if (planLimitKey !== undefined && this.props.planLimitation && !this.props.planLimitation[planLimitKey]) {
      return null;
    }
 
    if (this.props.value !== undefined) currentPropTypeValue = get(this.props.value, prop.name)
    if (['THorizontalGroup', 'TVerticalGroup', 'TPopover', 'TTab'].includes(propTypeName)) {
      currentPropTypeValue = this.props.value;
    }
 
    const input = getInputComponent(
      prop.name,
      prop.type,
      this.props.valueChanged,
      currentPropTypeValue,
      this.props.bindingChanged,
      this.props.binding,
      this.props.componentId,
      true,
      this.props.readOnly,
      this.props.propertiesFilter,
      this.props.projectId,
      this.props.thunkableVersion,
      this.props.parentType,
      this.props.handleFileUploaded,
      {
        componentType: this.props.componentType,
        planLimitation: this.props.planLimitation,
      });
 
    return input;
  }
 
  getTabPans = () => {
    const params = this.props.propType.descriptor.params;
    const inputs = params.map((prop, index) => {
      const translatePropertyKey = get(prop, 'type.descriptor.translatePropertyKey');
      const translatedPropertyLabel = translatePropertyKey ? t(translatePropertyKey) : null;
      const propertyLabel = get(prop, 'type.descriptor.showNoPropertyLabel')
        && (translatedPropertyLabel || prop.type.descriptor.label || prop.name);
      const hasPicture = get(prop, 'name', '').toLowerCase().includes('picture');
      const pictureParams = get(prop, 'type.descriptor.params');
      const pictureProps = hasPicture && pictureParams ? {} : null;
      if (hasPicture && pictureParams) {
        pictureParams.map(param => {
          const name = param.name.toLowerCase().includes('resizemode') ? 'resizeMode' : 'src';
          pictureProps[name] = this.props.value[param.name];
        })
      }
 
      return (
        <TabPane tab={<ThunkableText message={propertyLabel}/>} key={`${prop.name}-${this.props.componentId}-${index}`} >
          { pictureProps && <ImagePreview assets={ this.props.assets } { ...pictureProps } /> }
          { this.getTabInput(prop) }
        </TabPane>
      );
    });
 
    return inputs;
  }
 
  render() {
    const tabPans = this.getTabPans();
 
    return (
      <Tabs animated={false} className='tab-wrapper' size='small'>
        { tabPans }
      </Tabs>
    );
  }
}
 
export default TTabWrapper;
