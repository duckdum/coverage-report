import React, { Component } from 'react';
import { ThunkableIcon, ThunkableText } from '../../ComponentLibrary';
import { getInputComponent } from '../PropertyInputs';
import { t } from 'i18next';
import { get } from 'lodash';
import { Badge, Col, Row } from 'antd/lib';
 
class THorizontalGroupPropertyInput extends Component {
 
  render() {
    const inputs = [];
    const params = this.props.propType.descriptor.params;
    const shouldWrap = params.find(prop => get(prop, 'type.descriptor.lineBreakAfter'));
    params.forEach((prop, index) => {
      var currentPropTypeValue;
      const propName = prop.name;
      const propType = prop.type;
      let translatePropertyKey = propType.descriptor.translatePropertyKey;
      let translatedPropertyLabel = t(translatePropertyKey);
      const iconName = propType.descriptor.showIconPropertyLabel;
      const label = iconName
        ? <ThunkableIcon icon={iconName} />
        : <ThunkableText message={translatedPropertyLabel || propType.descriptor.label || propName}/>;
      if (this.props.value !== undefined) {
        currentPropTypeValue = get(this.props.value, propName);
      }
      const canUseWithParentType = get(propType, 'descriptor.canUseWithParentType');
      if (canUseWithParentType && !canUseWithParentType(this.props.parentType)) {
        return;
      }
      const propTypeName = get(propType, 'descriptor.type');
      const propertyLabel = !propType.descriptor.showNoPropertyLabel
        && (iconName || translatedPropertyLabel || propType.descriptor.label || propName)
        ? <Col span={4} className='horizontal-group-property-label-col'>{ label }</Col>
        : null;
      let warpperClassName = shouldWrap ? `horizontal-group-wrap-${propTypeName}` : 'horizontal-group-wrap';
      if (['THorizontalGroup', 'TVerticalGroup', 'TPopover', 'TTab'].includes(propTypeName)) currentPropTypeValue = this.props.value;
      if (shouldWrap && propTypeName === 'oneOfWithLabel') {
        const optionStyle = get(propType, 'descriptor.optionStyle', 'option').includes('icon');
        const hasIcon = optionStyle || propType.descriptor.params.find(param => param[2]);
        if (hasIcon) warpperClassName += `-iconSize`;
      }
 
      var currentInput = getInputComponent(propName, 
                                          propType, 
                                          this.props.valueChanged, 
                                          currentPropTypeValue, 
                                          this.props.bindingChanged,
                                          this.props.binding,
                                          this.props.componentId, 
                                          true,
                                          this.props.readOnly,
                                          this.props.propertiesFilter,
                                          this.props.projectId,
                                          this.props.thunkableVersion,
                                          this.props.parentType,
                                          this.props.handleFileUploaded,
                                          {
                                            assets: this.props.assets,
                                            componentType: this.props.componentType,
                                            planLimitation: this.props.planLimitation,
                                          });
      const supportPlatforms = propType.descriptor.supportPlatforms;
      let withBadges = currentInput;
      if (supportPlatforms && !propertyLabel) {
        supportPlatforms.forEach((platform, index) => {
          withBadges = <Badge className={`${platform}-badge badge-${index + 1} without-label`} dot>{ withBadges }</Badge>
        });
      }
 
      inputs.push(
        <div key={propName} className={ warpperClassName }>
          <Row align='middle' className='horizontal-group-property-row' type='flex'>
            { propertyLabel }
            <Col className='horizontal-group-property-col'>{ withBadges }</Col>
          </Row>
        </div>
      );
 
      if (propType.descriptor.lineBreakAfter) {
        inputs.push(<div key={`${propName}-line-break`} className='property-line-break'></div>);
      }
    })
 
    return (
      <div className={`horizontal-group-property ${ shouldWrap ? 'horizontal-wrap' : '' }`}>
        { inputs }
      </div>
    );
  }
}
 
export default THorizontalGroupPropertyInput;
