import React from 'react';
import { Menu, Modal } from 'antd/lib';
import { t } from 'i18next';
import TScreen from '../../shared/componentDefinitions/TScreen';
import TLayout from '../../shared/componentDefinitions/TLayout';
import TScrollView from '../../shared/componentDefinitions/TScrollView';
import { RELEASE_ACCESS } from '../../shared/utils/FeatureUtils';
 
const PropertiesPanelHeaderOptionsMenu = ({
  deleteComponent,
  duplicateComponent,
  closeMenu,
  componentName,
  componentType,
  readOnly,
  renameComponent,
  setIsShowingSaveScreenModal,
  setIsShowingDVLayoutModal,
  planLimitation,
  thunkableVersion,
}) => {
  const showDeleteConfirmation = () => {
    closeMenu && closeMenu();
    Modal.confirm({
      title: t("confirmation.deleteConfirmationQuestion", {componentOrAsset: componentName}),
      content: t("confirmation.deleteConfirmationMsg", {componentOrAsset: componentName}),
      okText: t("confirmation.delete"),
      okType: 'danger',
      cancelText: t("confirmation.cancel"),
      onOk() {
        deleteComponent();
      },
      onCancel() {},
    });
  };
  
  const showDuplicateConfirmation = () => {
    closeMenu && closeMenu();
    Modal.confirm({
      title: t("confirmation.duplicateConfirmationQuestion", {componentOrAsset: componentName}),
      content: t("confirmation.duplicateConfirmationMsg", {componentOrAsset: componentName}),
      okText: t("confirmation.duplicate"),
      cancelText: t("confirmation.cancel"),
      onOk() {
        duplicateComponent();
      },
      onCancel() {},
    });
  };
 
  const handleRenameClick = () => {
    closeMenu && closeMenu();
    renameComponent();
  };
 
  const handleSaveScreenClick = () => {
    setIsShowingSaveScreenModal(true);
  }
 
  const handleSaveDataViewerLayoutClick = () => {
    setIsShowingDVLayoutModal(true);
  }
  
  return (
    <>
      <Menu selectable={false}>
        <Menu.Item className="prop-panel-header-menu-item" key="Rename" onClick={handleRenameClick} disabled={readOnly}>
          {t("confirmation.rename")}
        </Menu.Item>
        <Menu.Item className="prop-panel-header-menu-item" key="Duplicate" onClick={showDuplicateConfirmation} disabled={readOnly}>
          {t("confirmation.duplicate")}
        </Menu.Item>
        { thunkableVersion >= RELEASE_ACCESS.PUBLIC_BETA && (componentType === TLayout.Type || componentType === TScrollView.Type) && (
            <Menu.Item className="prop-panel-header-menu-item" key="SaveDataviewLayout" onClick={handleSaveDataViewerLayoutClick} disabled={readOnly}>
              {t("confirmation.saveLayout")}
            </Menu.Item>)
        }
        { planLimitation
            && planLimitation.isSaveScreensEnabled
            && componentType === TScreen.Type
            && <Menu.Item className="prop-panel-header-menu-item" key="Save Screen" onClick={handleSaveScreenClick} disabled={readOnly}>
        {t("confirmation.saveScreen")}
        </Menu.Item> }
        <Menu.Item className="prop-panel-header-menu-item" key="Delete" onClick={showDeleteConfirmation} disabled={readOnly}>
          {t("confirmation.delete")}
        </Menu.Item>
      </Menu>
    </>
  );
}
 
 
export default PropertiesPanelHeaderOptionsMenu;
