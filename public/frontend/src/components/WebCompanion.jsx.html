import React, { useEffect, useState } from 'react';
import { t } from 'i18next';
import { TRACKING_EVENT_TYPES, TRACKING_EVENT_CONTEXTS, track } from "../tracking";
import { Button, Icon, Tooltip } from 'antd/lib';
import MobileDetect from 'mobile-detect';
import { cloneDeep } from 'lodash';
import { getDeviceSizeFromTypeAndOrientation, DEVICE_FRAME_SIZE } from '../shared/utils/DeviceSizes';
import '../style/WebCompanion.scss';
let boundingHeight = 0;
 
const WebCompanionInfo = (betaFeature = "The feature", context, tooltipLink = "https://docs.thunkable.com/v/drag-and-drop/publish-as-a-web-app-pro#unsupported-components") => {
  const trackLinkClick = () => {
    track(TRACKING_EVENT_TYPES.WEB_COMPANION_INFO_LINK_CLICKED, { context });
  }
 
  return (
    <span>
      { t('projectDetailPage.infoPopUp', { betaFeature }) }
      <a href={ tooltipLink }
        onClick={ trackLinkClick }
        target="_blank"
        style={{ padding: '0 3px' }}
        rel="noopener noreferrer">
        { t('projectDetailPage.listOfSupportedComponents') }
      </a>
    </span>
  );
};
 
const InfoBadge = ({ betaFeature, overlayClassName, context, styleClassName, tooltipLink }) => (
  <Tooltip
    overlayClassName={ overlayClassName || '' }
    placement="right"
    title={ WebCompanionInfo(betaFeature, context, tooltipLink) }>
    <Button className={ styleClassName ? `info-badge ${styleClassName}` : 'info-badge' }>
      <Icon type="question-circle-o" />
    </Button>
  </Tooltip>
);
 
const WebCompanion = ({ projectId, project, screenId, shouldReload, refresh, trackLivePreview, defaultDesignerDevice, defaultDesignerOrientation, isEditorPreview, previewOption }) => {
  const initCommunicationWithCompanion = (handler) => {
    window.addEventListener('ForSimplePhoneScreen', (e) => {
      handler(e.detail.value)
    });
  };
 
  const sendProjectDataToCompanion = () => {
    sendToCompanion('projectData', {
      screenId: screenId,
      project: cloneDeep(project),
    });
  }
 
  const sendToCompanion = (type, value) => {
    const myEvent = new CustomEvent('ForCompanion', {
      detail: {
        type,
        value,
      }
    });
    if (document.getElementById("web-companion")) {
      document.getElementById("web-companion").contentWindow.dispatchEvent(myEvent);
    }
  }
 
  useEffect(() => {
    initCommunicationWithCompanion(onMessageFromCompanion);
  }, []);
 
  const onMessageFromCompanion = (message) => {
    if (message.type === 'requestProjectData') {
      sendProjectDataToCompanion();
    }
  }
 
  const [isTracked, setIsTracked] = useState(false);
  const mobileDetector = new MobileDetect(window.navigator.userAgent);
  const onMobileClassName = mobileDetector.mobile() ? ' on-mobile' : '';
  let frameHeight = (getDeviceSizeFromTypeAndOrientation('PHONE', 'PORTRAIT').height + DEVICE_FRAME_SIZE) + 'px';
  let frameWidth = (getDeviceSizeFromTypeAndOrientation('PHONE', 'PORTRAIT').width + DEVICE_FRAME_SIZE) + 'px';
  if (defaultDesignerDevice && defaultDesignerOrientation) {
    let { height, width } = getDeviceSizeFromTypeAndOrientation(defaultDesignerDevice, defaultDesignerOrientation)
    frameHeight = (height + DEVICE_FRAME_SIZE) + 'px';
    frameWidth = (width + DEVICE_FRAME_SIZE) + 'px';
  }
  useEffect(() => {
    if (shouldReload) {
      document.getElementById("web-companion").src = document.getElementById("web-companion").src
    };
  },[ shouldReload ]);
  useEffect(() => {
      document.getElementById("web-companion").src = document.getElementById("web-companion").src
  },[ refresh ]);
  const trackMouseOver = () => {
    if (!isTracked) {
      if (trackLivePreview) trackLivePreview(TRACKING_EVENT_CONTEXTS.MOUSE_OVER_WEB_COMPANION);
      setIsTracked(true);
    }
  }
 
  const getIframeHeightWidth = () => {
    if (defaultDesignerDevice && defaultDesignerOrientation) {
      const { height, width } = getDeviceSizeFromTypeAndOrientation(defaultDesignerDevice, defaultDesignerOrientation);
      return { height: height + DEVICE_FRAME_SIZE, width: width + DEVICE_FRAME_SIZE};
    }
    return getDeviceSizeFromTypeAndOrientation('PHONE', 'PORTRAIT');
  }
  
  const setPreviewSize = () => {
    const { height, width } = getIframeHeightWidth();
    const iframe = document.getElementsByTagName("iframe")[0];
    let scaleCorrection = 0.2;
      if (defaultDesignerDevice === 'PHONE') {
        if (defaultDesignerOrientation === 'PORTRAIT') {
          scaleCorrection = 0.4
        } else {
          scaleCorrection = 1
        }
      }
      const deviceScale = Math.min((window.innerWidth - 200)/width, (window.innerHeight - 48)/height) - scaleCorrection;
      if (previewOption === '50') {
        return {
          "-webkit-transform": "scale(0.5)", 
          "transform": "scale(0.5)",
          "transform-origin": defaultDesignerDevice === 'PHONE' ? "center center" : "top center",
        }
      } else if (previewOption === '100') {
        if (document.getElementsByClassName("previewer-container")[0]) {
          boundingHeight = document.getElementsByClassName("previewer-container")[0].scrollHeight;
        }
        return {
          "-webkit-transform": "scale(1)", 
          "transform": "scale(1)",
          "transform-origin": defaultDesignerDevice === 'PHONE' ? "center center" : "center left",
          "margin-bottom": "30px"
        }
      } else if (previewOption === '200') {
        return {
          "-webkit-transform": "scale(2)", 
          "transform": "scale(2)",
          "transform-origin": (defaultDesignerDevice === 'PHONE' && defaultDesignerOrientation !== 'LANDSCAPE') ? "top center" : "top left",
        }
      } else {
        // Show phone at 100% unless screen is too small
        if (boundingHeight === 0 || boundingHeight >= window.innerHeight) {
          iframe.style['-webkit-transform'] = `scale(${deviceScale})`;
          iframe.style.transform = `scale(${deviceScale})`;
          iframe.style['transform-origin'] = "top center";
          return {
            "-webkit-transform": `scale(${deviceScale})`, 
            "transform": `scale(${deviceScale})`, 
            "transform-origin": "top center"
          }
        }
      } 
    } 
  
  const src = `${window.location.origin}/web-build/index.html?projectId=${projectId}${screenId ? `&screenId=${screenId}&isEditorPreview=${isEditorPreview ? 1 : 0}` : ''}` 
 
  return (
    <iframe
      onMouseOver={ trackMouseOver }
      title="thunkable-web-preview"
      id="web-companion"
      className={ `web-companion-container${onMobileClassName}` }
      style={
        project.usesDragDropUi && !window.location.href.includes('/preview') 
          ? {...getIframeHeightWidth(), ...setPreviewSize()} 
          : {height: frameHeight, width: frameWidth}
      }
      src={src}>
    </iframe>
  );
}
