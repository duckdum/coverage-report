import React, { Component } from 'react'
 
import { Button } from 'antd/lib';
 
class Tutorial extends Component {
 
  componentDidMount() {
    // initialize the tutorial to exist in the tutorial panel state and be on the first page
    this.props.tutorialFunctions.addTutorial(this.props.tutorialId);
    //set the maximum number of pages in the tutorial
    this.props.tutorialFunctions.setMaxPageNum(this.props.tutorialId, React.Children.count(this.props.children));
  }
 
  render() {
    const tutorialId = this.props.tutorialId;
    const tutorialName = this.props.name;
    const tutorialFunctions = this.props.tutorialFunctions;
    const maxPageNum = tutorialFunctions.getMaxPageNum(tutorialId);
    const pageNum = tutorialFunctions.getTutorialPageNum(tutorialId)
    // the back button should be hidden if it's on the first page
    const backButtonClass = pageNum === 0 ? "back-button hidden" : "back-button";
    const backButtonOnClick = () => tutorialFunctions.goToTutorialPreviousPage(tutorialId);
    // the next button should be hidden if it's on the last page
    const nextButtonClass = (pageNum === maxPageNum - 1 ? "next-button hidden" : "next-button");
    
    const nextButtonOnClick = () => tutorialFunctions.goToTutorialNextPage(tutorialId);
    
    const exitTutorialOnClick = () => {
      tutorialFunctions.switchTutorialPageContent("home");
      // if on the last page when exiting a tutorial, reset tutorial to the first page
      if (pageNum === maxPageNum - 1) {
        tutorialFunctions.setTutorialPageNum(tutorialId, 0);
      }
    }
    const tutorialTitleOnClick = () => tutorialFunctions.setTutorialPageNum(tutorialId, 0);
 
    const bottomNextButtonClass = (pageNum === maxPageNum - 1 ? "hidden" : "");
    const bottomHomeButtonClass = (pageNum === maxPageNum - 1 ? "" : "hidden");
 
    // add the current page and the page number to each tutorial page object
    let propPageNum = 0;
    const childrenWithProps = React.Children.map(this.props.children, child => 
      React.cloneElement(child, { currentPage: pageNum, page: propPageNum++, tutorialFunctions: this.props.tutorialFunctions })
    );
    return (
      <div className="tutorial-page">
        <div className="tutorial-header">
          <div>
            <span onClick={exitTutorialOnClick} className="tutorial-home-link">Home</span> / <span className="tutorial-title" onClick={tutorialTitleOnClick}>{tutorialName}</span>
          </div>
        </div>
        <div className="tutorial-nav-buttons">
          <div className="back-button-container">
            <Button className={backButtonClass} onClick={backButtonOnClick}>Back</Button>
          </div>
          <div className="page-indicator-container">
            <span>{pageNum + 1} of {maxPageNum}</span>
          </div>
          <div className="next-button-container">
            <Button className={nextButtonClass} type="primary" onClick={nextButtonOnClick}>Next</Button>
          </div>
        </div>
        <div className="tutorial-content">
          {childrenWithProps}
        </div>
        <div className="tutorial-bottom-buttons">
          <Button className={bottomNextButtonClass} type="primary" onClick={nextButtonOnClick}>Next</Button>
          <Button className={'bottom-home-buttton ' + bottomHomeButtonClass} type="primary" onClick={exitTutorialOnClick}>Home</Button>
        </div>
 
      </div>
   );
  }
}
 
export default Tutorial
