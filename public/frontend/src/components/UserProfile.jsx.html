import { Alert, Button, Col, Icon, Row, Tabs } from 'antd/lib';
import gql from 'graphql-tag';
import React, { Component } from 'react';
import {
  compose,
  graphql,  
} from 'react-apollo';
import { browserHistory } from 'react-router';
import { get } from 'lodash';
import { getCompleteBackendURI } from '../shared/componentDefinitions/SourceProcessor';
 
import TopNavBar from './TopNavBar';
import LiveTest from './LiveTest';
import ProjectCard from './ProjectCard';
import ProjectList from './ProjectList';
import '../style/Project.scss';
import '../style/Account.scss';
import '../style/UserProfile.scss';
import defaultProfilePicture from '../images/beaverCircle.svg';
import { isFeatureEnabled, RELEASE_ACCESS } from '../shared/utils/FeatureUtils';
import { TRACKING_EVENT_TYPES, track, TRACKING_EVENT_CONTEXTS } from "../tracking";
 
const DEFAULT_VIEW_CLASSNAME = "default-view";
const ButtonGroup = Button.Group;
const TabPane = Tabs.TabPane;
const projectInfo = `{
  id
  icon
  isOwner
  projectName
  totalCopy
  totalStar
  updatedAt
  summary
  username
  isPublic
  readOnly
  usesDragDropUi
}`;
 
const UserQuery = gql`
  query User($username: String!)  {
    thunkable {
      version
    }
    user {
      id
      isAnonymous
      isLiveTesting
      isPro
      currentLiveTestProjectId
      currentLiveTestScreenId
      paymentBannerMessage
      profile
      shouldShowPromotionBanner
    }
    userProfile(username: $username) {
      isFeatured
      isPro
      username
      profile
      planLimitation
      joinAt
    }
    projects(username: $username) ${projectInfo}
    starProjects(username: $username) ${projectInfo}
  }
`;
 
const onUserUpdated = gql`
  subscription onUserUpdated {
    userUpdated {
      id
      isLiveTesting
      currentLiveTestProjectId
      currentLiveTestScreenId
    }
  }
`;
 
class UserProfile extends Component {
  constructor(props) {
    super(props);
    const tab = props.location.query.tab;
    this.state = {
      activeKey: tab || "overview",
      showLiveTestModal: false,
      showLiveTestProjectId: null,
      liveTestingProjectName: null,
      liveTestingDndProject: null,
      view: 'list',
      page: 1,
      pageSize: 12,
      profilePicturePath: defaultProfilePicture,
      projectStartIndex: 0,
    };
  }
  
  componentDidMount() {
    this.props.subscribeToUserUpdates();
    const eventProperties = { username: this.props.params.username };
    const userProperties = { hasVisitedProfile: 1 };
    track(TRACKING_EVENT_TYPES.USER_PROFILE_PAGE_VIEWED, eventProperties, userProperties);
  }
 
  componentDidUpdate(prevProps, prevState) {
    const prevUser = prevProps.data && prevProps.data.user;
    const user = this.props.data && this.props.data.user;
    const userStartedLiveTesting = prevUser && prevUser.isLiveTesting === false && user.isLiveTesting;
    const isCurrentLiveTestingProject = user && user.currentLiveTestProjectId === this.state.showLiveTestProjectId;
 
    if (userStartedLiveTesting && isCurrentLiveTestingProject) {
      this.trackLivePreview(TRACKING_EVENT_CONTEXTS.PROFILE_PAGE);
    }
 
    if (prevProps.data.loading !== this.props.data.loading) {
      this.loadProfilePicture();
    }
  }
 
  trackLivePreview = (context, eventProperties) => {
    const email = this.props.data && this.props.data.user && this.props.data.user.email;
    const properties = {
      context,
      email,
      projectId: this.state.showLiveTestProjectId,
      projectName: this.state.liveTestingProjectName,
      view: this.state.view,
      'New UI': this.state.liveTestingDndProject,
      ...eventProperties
    };
    track(TRACKING_EVENT_TYPES.LIVE_PREVIEW, properties, { isLivePreview: 1 });
  }
 
  showLiveTestModal = (projectId, projectName, usesDragDropUi, appLayout) => {
    this.setState({
      showLiveTestModal: true,
      showLiveTestProjectId: projectId,
      liveTestingProjectName: projectName,
      liveTestingDndProject: usesDragDropUi,
    });
    const eventProperties = { projectId, projectName, 'New UI': usesDragDropUi, 'Default app layout': appLayout };
    this.trackLivePreview(TRACKING_EVENT_CONTEXTS.PROFILE_PAGE, eventProperties);
  }
 
  hideLiveTestModal = () => {
    this.setState({ showLiveTestModal: false });
  }
 
  handleTabChange = (activeKey) => {
    browserHistory.replace(`/profiles/${this.props.params.username}?tab=${activeKey}`);
    this.setState({ activeKey });
  }
 
  handleChangeView = (view) => {
    track(TRACKING_EVENT_TYPES.VIEW_SWITCHED, {
      context: "User Profile Page",
      tab: this.state.activeKey,
      view,
    });
 
    this.setState({ view });
  }
 
  handlePaginationChange = (newPage, newPageSize) => {
    if (newPageSize !== this.state.pageSize) {
      track(TRACKING_EVENT_TYPES.PAGE_SIZE_CHANGED, {
        context: "User Profile Page",
        tab: this.state.activeKey,
        view: this.state.view,
        pageSize: newPageSize,
        prevPageSize: this.state.pageSize,
      });
    }
 
    const newStartIndex = (newPage - 1) * newPageSize;
    this.setState({
      page: newPage,
      pageSize: newPageSize,
      projectStartIndex: newStartIndex,
    });
  }
 
  getPaginationProps = (projects) => ({
    showSizeChanger: false,
    pageSize: this.state.pageSize,
    total: projects && projects.length,
    current: this.state.page,
    pageSizeOptions: ['6', '12', '24'],
    onChange: (newPage, pageSize) => {
      this.handlePaginationChange(newPage, pageSize);
    },
    onShowSizeChange: (page, newPageSize) => {
      this.handlePaginationChange(page, newPageSize);
    }
  })
 
  loadProfilePicture = () => {
    const profilePicture = get(this.props, 'data.userProfile.profile.picture');
    if (!profilePicture) return;
 
    const image = new Image();
    image.src = getCompleteBackendURI(profilePicture);
    image.onload = () => this.setState({ profilePicturePath: image.src });
    image.onerror = (err) => this.setState({ profilePicturePath: defaultProfilePicture });
  }
 
  render() {
    const { loading, thunkable, user, userProfile, projects, starProjects } = this.props.data;
    if (loading || !user ) return null;
    if (!userProfile) {
      return (
        <div>
          <TopNavBar
            thunkable={thunkable}
            paymentBannerMessage={user.paymentBannerMessage}
            shouldShowPromotionBanner={user.shouldShowPromotionBanner}
            shouldShowUsernameBanner={ false }
            userAvatar={user.profile.picture}
          />
          <div className='center-flex user-profile-error'>
            <Alert type="error"
              message="This page isnâ€™t available"
              description="The link you followed may be broken, or the page may have been removed." />
          </div>
        </div>
      )
    }
 
    if (thunkable && !(isFeatureEnabled(thunkable.version, 122))) return null;
    const projectSource = this.state.activeKey === 'stars' ? starProjects : projects;
    const hasData = projectSource && projectSource.length !== 0;
    const projectData = hasData && projectSource.slice(this.state.projectStartIndex, this.state.projectStartIndex + this.state.pageSize);
    const projectProps = {
      showLiveTestModal: this.showLiveTestModal,
      paginationProps: this.getPaginationProps(projectSource),
      isProfile: true
    };
 
    const projectCard = (<ProjectCard projects={ projectData } { ...projectProps } />);
    const projectList = (<ProjectList projects={ projectSource } { ...projectProps } />);
    const hasProBadge = userProfile.planLimitation && userProfile.planLimitation.hasProBadge;
 
    return (
      <div>
        <TopNavBar
          thunkable={thunkable}
          paymentBannerMessage={user.paymentBannerMessage}
          shouldShowUsernameBanner={ false }
          userAvatar={user.profile.picture} />
        <Row type="flex">
          <Col span={6} className="user-profile-sidebar">
            <div className="user-profile-photo-container">
              <img src={ this.state.profilePicturePath } alt="user-profile" />
            </div>
            <div className="user-profile-container">
              <div className="user-profile-username-container">
                <h1 className="user-profile-username">{ userProfile.username }</h1>
              </div>
              <div className="badges-container">
                { hasProBadge && <div className="user-profile-pro-badge"><span>PRO</span></div> }
                { userProfile.isFeatured && <div className="user-profile-feature-badge"><span>Featured</span></div> }
              </div>
              <p className="join-date"><Icon type="calendar" />Joined { userProfile.joinAt }</p>
            </div>
          </Col>
          <Col span={18} className="user-profile-content">
            <Tabs className="account user-profile"
              activeKey={this.state.activeKey}
              animated={false}
              onChange={this.handleTabChange}>
              <TabPane tab="OVERVIEW" key="overview">
              </TabPane>
            { isFeatureEnabled(thunkable.version, RELEASE_ACCESS.DEVELOPERS_ONLY) &&
              <TabPane tab="STARS" key="stars">
              </TabPane>
            }
            </Tabs>
            <ButtonGroup>
              <Button
                className={ this.state.view === "list" ? DEFAULT_VIEW_CLASSNAME : "" }
                size="small"
                icon="bars"
                onClick={ this.handleChangeView.bind(this, 'list') }
              />
              <Button
                className={ this.state.view === "card" ? DEFAULT_VIEW_CLASSNAME : "" }
                size="small"
                icon="appstore"
                onClick={ this.handleChangeView.bind(this, 'card') } />
            </ButtonGroup>
            { hasData && this.state.view === 'list' ? projectList : projectCard }
            <LiveTest
              projectId={ this.state.showLiveTestProjectId }
              shouldShowWebCompanion={ true }
              show={ this.state.showLiveTestModal }
              onHide={ this.hideLiveTestModal.bind(this) }
              trackLivePreview={ this.trackLivePreview }
            />
          </Col>
        </Row>
      </div>
    );
  }
}
 
const UserProfileWithGraphQL = compose(
  graphql(UserQuery, {
    options: props => ({ variables: { username: props.params.username } }),  
  }),
  graphql(UserQuery, {
    name: 'userUpdates',
    options: props => ({ variables: { username: props.params.username } }),
    props: props => ({
      subscribeToUserUpdates: params => (
        props.userUpdates.subscribeToMore({ document: onUserUpdated })
      )
    }),
  }),
)(UserProfile);
 
export default UserProfileWithGraphQL;
