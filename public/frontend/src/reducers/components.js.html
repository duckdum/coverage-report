export const properties = (state = {}, action) => {
  switch (action.type) {
    case ACTION_TYPES.CHANGE_COMPONENT_PROPERTY:
      let nestedKeys = action.prop.split('.');
 
      if (nestedKeys.length === 1) {
        return Object.assign({}, state, { [nestedKeys[0]]: action.newValue });
      }
 
      let firstProp = nestedKeys.shift();
      let deepAction = Object.assign({}, action, { prop: nestedKeys.join('.') });
      let deepActionReduction = properties(state[firstProp], deepAction);
      return Object.assign({}, state, { [firstProp]: deepActionReduction });
    default:
      return state;
  }
}

export const initComponentProperties = (component, parent, setMappedProperties = true) => {
  if (!component.properties) return;
  for (let prop in component.properties) {
    if (!component.properties.hasOwnProperty(prop)) {
      continue;
    }
    // Deal with mapped properties
    const mappedProperty = getMappedProperty(component.type, prop);
    // Don't allow properties with default values not set (ie undefined)
    // override existing values
    if (setMappedProperties && mappedProperty && component.properties[prop] !== undefined) {
      const mappedAction =
      {
        type: ACTION_TYPES.CHANGE_COMPONENT_PROPERTY,
        prop: mappedProperty,
        newValue: component.properties[prop]
      };
      const moreProperties = properties(component.properties, mappedAction);
      component.properties = { ...component.properties, ...moreProperties };
    }
    // Deal with design time custom setters
    const designTimeCustomSetter = getDesignTimeCustomSetter(component.type, prop);
    if (designTimeCustomSetter)
      designTimeCustomSetter(component.properties[prop], component, parent);
  }
}
 
const addVisibleComponent = (parentComponent, action, newComponent, setMappedProperties = true) => {
  let children = parentComponent.children;
  // Note: works for top-level component because ids will both be undefined.
  if (parentComponent.id === action.parentId) {
    initComponentProperties(newComponent, parentComponent, setMappedProperties)
    children = update(children || [], { $splice: [[action.index, 0, newComponent]] })
    return update(parentComponent, {
      children: {
        $set: children
      }
    });
  } else if (children !== undefined) {
    return update(parentComponent, {
      children: {
        $set: children.map((child) => addVisibleComponent(child, action, newComponent, setMappedProperties))
      }
    })
  } else {
    return { ...parentComponent };
  }
}

const addCustomPropertiesToNewComponent = (newComponent, customProperties) => {
  if (!customProperties) return;

  }
}
 
const setDefaultValuesFromParent = (newComponent, parentId, components, apiComponents) => {
  const parentComponent = ComponentUtils.getComponentById(parentId, components, apiComponents[API_COMPONENT_CONTAINER_NAME]);
  const propTypes = COMPONENTS[newComponent.type].TPropTypes;
  // TODO move all default value setting code to one place - probably in the reducer
  for (const propTypeKey in propTypes) {
    if (!propTypes.hasOwnProperty(propTypeKey)) { continue; }
    const { descriptor } = propTypes[propTypeKey];
    if (descriptor !== undefined && 'setDefaultValueFromParent' in descriptor) {
      newComponent.properties[propTypeKey] = parentComponent.properties[propTypeKey];
    }
    if (descriptor !== undefined && 'setDefaultValueFromParentFirstListElement' in descriptor) {
      newComponent.properties[propTypeKey] = parentComponent.properties[descriptor.setDefaultValueFromParentFirstListElement.property][0];
    }
  }
}

const addComponent = (components = {}, apiComponents = {}, action) => {
  const newComponent = { ...action.component };
  if (action.type !== ACTION_TYPES.ADD_MODULE_COMPONENT) {
    addCustomPropertiesToNewComponent(newComponent, action.customProperties);
    setDefaultValuesFromParent(newComponent, action.parentId, components, apiComponents);
  }
  if (action.component.isApi) {
    newComponent.name = ComponentUtils.getNewName(apiComponents[API_COMPONENT_CONTAINER_NAME], action.component);
    return [components, addApiComponent(apiComponents, action, newComponent)];
  } else {
    if (!newComponent.name) {
      newComponent.name = ComponentUtils.getNewName(components, action.component);
    }
 
    let visibleComponents = addVisibleComponent(components, action, newComponent)
    // go through components children and give them names if they don't exist
    if (action.shouldNameDefaultChildren) {
      ComponentUtils.componentSearch(newComponent, (childComponent) => {
        if (!childComponent.name || action.shouldForceNameDefaultChildren) {
          childComponent.name = ComponentUtils.getNewName(visibleComponents, childComponent);
        }
        return false;
      })
    }
    return [visibleComponents, apiComponents];
  }
}

const setDesignerDeviceAndOrientation = (components, apiComponents, action) => {
  const newDeviceSize = getDeviceSizeFromTypeAndOrientation(action.designerDevice, action.designerOrientation);
  // get max x/y
  let xMax = 0
  let yMax = 0
  ComponentUtils.componentSearch(components, (childComponent) => {
    if (get(childComponent, 'properties.X', null) !== null && get(childComponent, 'properties.Y', null) !== null) {
      xMax = Math.max(xMax, childComponent.properties.X + childComponent.properties.Width);
      yMax = Math.max(yMax, childComponent.properties.Y + childComponent.properties.Height);
    }
    return false;
  });
  let xScaleFactor = 1;
  let yScaleFactor = 1;
  if (xMax > newDeviceSize.width) {
    xScaleFactor = newDeviceSize.width / xMax;
  }
  if (yMax > newDeviceSize.height) {
    yScaleFactor = newDeviceSize.height / yMax;
  }
  ComponentUtils.componentSearch(components, (childComponent) => {
    if (childComponent.type === 'Screen') {
      childComponent.properties.DesignerHeight = newDeviceSize.height;
      childComponent.properties.DesignerWidth = newDeviceSize.width;
      if (action.designerOrientation === 'LANDSCAPE' && childComponent.properties.ScreenOrientation === 'portrait') {
        childComponent.properties.ScreenOrientation = 'landscape';
      }
      if (action.designerOrientation === 'PORTRAIT' && childComponent.properties.ScreenOrientation === 'landscape') {
        childComponent.properties.ScreenOrientation = 'portrait';
      }
    }
    if (get(childComponent, 'properties.X', null) !== null && get(childComponent, 'properties.Y', null) !== null) {
      childComponent.properties.X = childComponent.properties.X * xScaleFactor;
      childComponent.properties.Y = childComponent.properties.Y * yScaleFactor;
    }
    return false;
  });
  return [components, apiComponents];
}

const setPropertyValueOnComponent = (component, parentComponent, action) => {
  const propToSet = action.prop;
  const value = getClampedValue(component.type, propToSet, action.newValue);
  // Deal with design time custom setters.
  const designTimeCustomSetter = getDesignTimeCustomSetter(component.type, propToSet);
  if (designTimeCustomSetter) {
    designTimeCustomSetter(value, component, parentComponent);
  } else {
    // if there isn't a designTimeCustomSetter, set the properties
    component.properties = properties(component.properties, { ...action, newValue: value });
  }
  // Deal with mapped properties.
  const mappedProperty = getMappedProperty(component.type, propToSet);
  if (mappedProperty) {
    const mappedAction = { ...action, prop: mappedProperty, newValue: value };
    component.properties = properties(component.properties, mappedAction);
  }
  const onChange = getOnChange(component.type, propToSet);
  if (onChange) {
    onChange(component, { ...action }, properties);
  }
}

  }
 
  // Adjust for index shift from removing component.
  if (action.fromIndex < action.index && action.fromParentId === action.parentId) {
    action.index = Math.max(0, action.index - 1);
  }
 
  if (isApi) return [updatedComponents, addApiComponent(updatedApiComponents, action, component, false)];
 
  return [addVisibleComponent(updatedComponents, action, component, false), updatedApiComponents];
}
 
const setDvLayoutComponentPropertyBindable = (components, apiComponents, action) => {
  const { componentId, property, isBindable } = action;
  const isApi = isApiComponent(componentId, apiComponents);
  let newComponents = isApi ? apiComponents[API_COMPONENT_CONTAINER_NAME] : components;
 
  newComponents = ComponentUtils.copy(newComponents,
    (componentCopy, parentComponentCopy) => {
      if (componentCopy.id === componentId) {
        const newValue = isBindable
          ? [ ...(componentCopy.properties.bindableProperties || []), property]
          : componentCopy.properties.bindableProperties.filter((value) => value !== property);
          const mappedAction =
          {
            type: ACTION_TYPES.CHANGE_COMPONENT_PROPERTY,
            prop: 'bindableProperties',
            newValue
          };
          const moreProperties = properties(componentCopy.properties, mappedAction);
          componentCopy.properties = { ...componentCopy.properties, ...moreProperties };
      }
    }
  );
 
  if (isApi) {
    const updatedApiComponents = { ...apiComponents }
    updatedApiComponents[API_COMPONENT_CONTAINER_NAME] = newComponents
    return [components, updatedApiComponents]
  }
 
  return [newComponents, apiComponents];
};
