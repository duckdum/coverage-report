const findScreenById = async function(project, screenId) {
  const screen = ComponentUtils.componentSearch(project.components, function(component) {
    if (component.id === screenId) return true;
  });
 
  if (screen) return screen;
 
  throw new UserInputError(
    `No screen with id '${screenId}' exists in project '${project.projectName}'`
  );
}
 
const findBlocklyByScreenId = async function(project, screenId) {
  const blockly = project.blockly;
  const blocklyForThisScreen = {};
  if (blockly) {
    const ids = Object.keys(blockly);
    for (let id of ids) {
      if (screenId === id) {
        blocklyForThisScreen[screenId] = blockly[id];
        return blocklyForThisScreen;
      }
    }
  }
  return blocklyForThisScreen; 
}
 
const findApiComponentsUsed = async function(project, blockly) {
  const apiComponents = project.apiComponents.ApiContainer.children; 
  if (!apiComponents ||  apiComponents.length === 0 || !blockly || Object.keys(blockly).length === 0) return project.apiComponents;
  const apiComponentArray = [];
  for (let apiComponent of apiComponents) {
    const componentId = apiComponent.id;
    if (blockly.xml.includes(componentId) || blockly.code.includes(componentId) || blockly.appVariableDefCode.includes(componentId)) {
      apiComponentArray.push(apiComponent);
    }
  }
  return { ApiContainer: { 
    id: "ApiContainer",
    children: apiComponentArray } 
  };
}
 
const findAssetsUsed = async function(project, blockly, components) {
  const currentProjectAssets = project.assets;
  if (!currentProjectAssets || Object.keys(currentProjectAssets).length === 0) {
    return currentProjectAssets;
  }
  const assets = {};
  const currentProjectAssetKeys = Object.keys(currentProjectAssets);
  for (let assetKey of currentProjectAssetKeys) {
    const asset = currentProjectAssets[assetKey];
    let assetUsed;
    ComponentUtils.componentSearch(components, function(component) {
      const propertyKeys = component.properties ? Object.keys(component.properties) : [];
      for (let prop of propertyKeys) {
        if (component.properties[prop] === asset.name) assetUsed = true;
      }
      // TODO have a more sustainable way to find assets
      if (component.type === 'Sprite Class' && component.properties && component.properties.PictureList) {
        if (component.properties.PictureList.indexOf(asset.name) !== -1) assetUsed = true;
      }
    });
    if (blockly && Object.keys(blockly).length !== 0 && !assetUsed) {
      if (blockly.xml.includes(asset.name) || blockly.code.includes(asset.name) || blockly.appVariableDefCode.includes(asset.name)) assetUsed = true;
    }
    if (assetUsed) {
      assets[assetKey] = asset;
    }
  }
  return assets;
}

  Mutation: {
    async saveScreen(parent, args, context) {
      const project = await Project.findByIdIfAccessible(args.input.projectId, context.user);
      // TODO move upgrader to when screen is loaded, not saved
      upgradeProject(project, context.user);
      const components = await findScreenById(project, args.input.componentId);
      const blockly = await findBlocklyByScreenId(project, args.input.componentId);
      const apiComponents = await findApiComponentsUsed(project, blockly[args.input.componentId]);
      const assets = await findAssetsUsed(project, blockly[args.input.componentId], components);
      return { 
        module: await new Module({
          type: args.input.moduleType,
          blockly, 
          components: { children: [components] },
          apiComponents,
          name: args.input.name,
          owner: context.user.id,
          projectName: args.input.projectName,
          timeSaved: new Date().toISOString(),
          assets,
          customProperties: [],
          customEvents: [],
          customMethods: [],
          isApi: false,
          settings: {
            isPublic: args.input.isPublic
          }
        }).saveNewModule()
      }
    },
    async addNewModule(parent, args, context) {
      return {

              children: []
            }
          },
          name: args.input.name,
          owner: context.user.id,
          projectName: "N/A",
          timeSaved: new Date().toISOString(),
          assets: {},
          customEvents: [],
          customProperties: [],
          customMethods: [],
          isApi: args.input.isApi,
          settings: {
            isPublic: args.input.isPublic,
          }
        }).saveNewModule()
      }
    },
    async deleteModule(parent, args, context) {
      const mod = await Module.findOne({ _id: args.input.moduleId, owner: context.user.id });
      mod.remove();
      return { modules: await Module.find({ owner: context.user.id })};
    }, 
    async renameModule(parent, args, context) {
      const mod = await Module.findOne({ _id: args.input.moduleId, owner: context.user.id });
      mod.name = args.input.newModuleName;
      return { module: await mod.save() };
    },

      const assets = await findAssetsUsed(project, blockly[screenId], components);
      return {
        module: await new Module({
          type: moduleType,
          blockly,
          components: { children: [components] },
          apiComponents: {
            ApiContainer: {
              id: 'ApiContainer',
              children: []
            }
          },
          name,
          owner: user.id,
          projectName,
          timeSaved: new Date().toISOString(),
          assets,
          customProperties: [],
          customEvents: [],
          customMethods: [],
          isApi: false,
          settings: {
            isPublic
          }
        }).saveNewModule()
      };
    },
  },
