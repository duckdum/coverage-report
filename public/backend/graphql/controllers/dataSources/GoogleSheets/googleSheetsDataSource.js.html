const getDeleteDimension = (collectionName, startIndex, shouldDeleteAllRows, numberOfRowsInSpreadsheet) => {

const getDataFilterForMetadataValue = (metadataValue, firstRowWithData, collectionName, shouldDeleteAllRows, numberOfRowsInSpreadsheet, isDeleteBlock) => {
  const dataFilter = [];
  if (!isUuid(metadataValue)) {
    const index = getIndexFromInput(metadataValue, firstRowWithData);
    const requestObject = isDeleteBlock ? getDeleteDimension(collectionName, index, shouldDeleteAllRows, numberOfRowsInSpreadsheet) : {
      gridRange: {
        sheetId: parseInt(collectionName, 10),
        startRowIndex: index,
        endRowIndex: index + 1,
      }
    };
    dataFilter.push(requestObject);
  } else {
    dataFilter.push({
      developerMetadataLookup: {
        metadataValue
      }
    });
  }
  return dataFilter;
};
 
const handleError = (error, projectOwner) => {
  const noPermissionErrorMessage = NO_PERMISSION_ERRORS.includes(error) ? 'reconsent' : undefined;
  const noRefreshTokenErrorMessage = NO_REFRESH_TOKEN_ERRORS.includes(error) ? 'revoke_and_reconsent' : undefined;
  const shouldResetGoogleSheets = !!(noPermissionErrorMessage || noRefreshTokenErrorMessage);
  const user = projectOwner || { id: 'datasource', email: 'datasource@thunkable.com' };
  const errorMessage = noPermissionErrorMessage
    || noRefreshTokenErrorMessage
    || error.message
    || GENERAL_ERROR_MESSAGE;
 
  if (shouldResetGoogleSheets
    && projectOwner
    && projectOwner.resetGoogleSheets()) {
    projectOwner.save();
  }
 
  trackingController.trackError(error, user, 'Error in Data Source');
  return errorMessage;
};

const generateIdsForIndexes = async (owner, spreadsheetId, sheetId, indexes) => {
  let projectOwner;
 
  try {
    const { auth, user } = await getOauth2Client(owner, spreadsheetId);
    projectOwner = user;
    const sheets = google.sheets('v4');
    const requests = indexes.map(index => ({
      createDeveloperMetadata: {
        developerMetadata: {
          metadataKey: 'thunkableId',
          metadataValue: uuidv4(),
          location: {
            dimensionRange: {
              sheetId,
              dimension: 'ROWS',
              startIndex: index,
              endIndex: index + 1
            }
          },
          visibility: 'PROJECT',
        }
      }
    }));
 
    const addIdRequest = {
      auth,
      spreadsheetId,
      resource: {
        requests
      }
    };
 
    const sheetData = await sheets.spreadsheets.batchUpdate(addIdRequest);
    return sheetData.data.replies.map(({ createDeveloperMetadata: { developerMetadata } }) => (
      {
        id: developerMetadata.metadataValue,
        index: developerMetadata.location.dimensionRange.startIndex,
      }
    ));
  } catch (error) {
    console.error(`Failed to get spreadsheet data from id ${spreadsheetId}`);
    console.error(`The error is  ${error}`);
    const errorMessage = handleError(error, projectOwner);
    return { error: errorMessage };
  }

const getRowValuesFromSpreadsheet = async (owner, collection, spreadsheetId, firstRowWithData, first, after) => {
  let projectOwner;
 
  try {
    const { auth, user } = await getOauth2Client(owner, spreadsheetId);
    projectOwner = user;
    const sheets = google.sheets('v4');
 
    let startIndex = firstRowWithData || 2; // change this based on the config value
    if (after) {
      const request = {
        spreadsheetId,
        auth,
        resource: {
          includeGridData: true,
          dataFilters: getDataFilterForMetadataValue(after, firstRowWithData, collection.name),
        }
      };
      const response = await sheets.spreadsheets.getByDataFilter(request);
      startIndex = Number(get(response, 'data.sheets[0].data[0].startRow', 1)) + 2;
    }
    const endIndex = Number(first) + (Number(startIndex) - 1);
    const range = `${collection.label}!${startIndex}:${endIndex}`;
 
    const spreadsheetData = await sheets.spreadsheets.get(
      {
        auth,
        spreadsheetId,
        includeGridData: true,
        ranges: range,
      });
    const sheetData = spreadsheetData.data.sheets[0].data[0];
    const edges = [];
    if (!sheetData.rowData) {
      return {
        edges,
        pageInfo: {
          hasNextPage: false,
          hasPreviousPage: false,
          startCursor: '',
          endCursor: '',
        }
      };
    }
    const indexesToAddIdsTo = [];
    const sheetId = spreadsheetData.data.sheets[0].properties.sheetId;
    const fieldIds = {};
    collection.fields.forEach(({ id, name }) => {
      const index = name.split('-')[1];
      fieldIds[index] = id;
    });
    const getValidValuesWithOriginalIndexes = (accumulator, current, index) => {
      if (has(fieldIds, `${index}`)) {
        return [...accumulator, { formattedValue: current.formattedValue, originalIndex: index }];
      }
      return accumulator;
    };
    let numberOfRowsWithData = 0;
    for (let i = 0; i < sheetData.rowData.length; i++) {
      const rowMetadata = get(sheetData, `rowMetadata[${i}].developerMetadata`, []);
      const rowWithId = rowMetadata.find(data => data.metadataKey === 'thunkableId' && isUuid(data.metadataValue));
      const hasValues = !!sheetData.rowData[i].values;
      const allEmpty = hasValues && sheetData.rowData[i].values.every(({ formattedValue }) => !formattedValue);
      if (hasValues && !allEmpty) {
        numberOfRowsWithData += 1;
        const fieldValuesWithOriginalIndexes = sheetData.rowData[i].values.reduce(getValidValuesWithOriginalIndexes, []);
        const values = fieldValuesWithOriginalIndexes.map(({ formattedValue, originalIndex }) => ({
          fieldId: fieldIds[originalIndex],
          value: formattedValue || ''
        }));
        const edge = {
          node: {
            values,
          }
        };
        if (rowWithId) {
          edge.cursor = rowWithId.metadataValue;
          edge.node.id = rowWithId.metadataValue;
        } else {
          indexesToAddIdsTo.push(i + sheetData.startRow);
        }
        edges.push(edge);
      }
    }
    if (indexesToAddIdsTo.length) {
      const newIds = await generateIdsForIndexes(owner, spreadsheetId, sheetId, indexesToAddIdsTo);
      newIds.forEach(({ id, index }) => {
        if (edges[index - 1]) {
          edges[index - 1].cursor = id;
          edges[index - 1].node.id = id;
        }
      });
    }
    if (edges.length === 0) {
      return {
        edges,
        pageInfo: {
          hasNextPage: false,
          hasPreviousPage: false,
          startCursor: '',
          endCursor: '',
        }
      };
    }
    const startCursor = edges[0].cursor;
    const endCursor = edges[edges.length - 1].cursor;
    const hasNextPage = numberOfRowsWithData === Number(first);
    const hasPreviousPage = !!after;
    const pageInfo = {
      hasPreviousPage,
      hasNextPage,
      startCursor,
      endCursor,
    };
    return {
      edges,
      pageInfo
    };
  } catch (error) {
    console.error(`Failed to get spreadsheet data from id ${spreadsheetId}`);
    console.error(`The error is  ${error}`);
    const errorMessage = handleError(error, projectOwner);
    return { error: errorMessage };
  }
};

// Cell is of type: https://developers.google.com/sheets/api/reference/rest/v4/spreadsheets/cells
const getCellType = (cell) => {
  if (!cell || !cell.effectiveValue) return null;
  const basicCellType = Object.keys(cell.effectiveValue)[0];
  switch (basicCellType) {
    case 'numberValue': {
      const numberFormat = cell.effectiveFormat.numberFormat;
      const value = cell.effectiveValue.numberValue;
      const stringValue = value.toString();
      if (stringValue.match(PHONE_REGEX)) return 'PHONE';
 
      // https://developers.google.com/sheets/api/reference/rest/v4/spreadsheets/cells#NumberFormat
      return numberFormat ? numberFormat.type : 'NUMBER';
    }
    case 'stringValue': {
      const value = cell.effectiveValue.stringValue;
      // TODO do we want to distinguish link's here? We'd have to extra work to
      // determine what type of file the link points to
      if (cell.hyperlink) {
        const link = cell.hyperlink;
        if (link.match(/\.(jpeg|jpg|png|bmp|gif)$/)) return 'IMAGE';
        if (link.match(/\.(mp4)$/)) return 'VIDEO';
        if (link.match(/\.(mp3)$/)) return 'AUDIO';
        return 'LINK';
      }
      if (value.match(EMAIL_REGEX)) return 'EMAIL';
      if (value.match(PHONE_REGEX)) return 'PHONE';
      if (value.match(LATLNG_REGEX)) return 'LATLNG';
      if (cell.effectiveValue.stringValue.length > 50) return 'PARAGRAPH';
 
      return 'STRING';
    }
    case 'boolValue':
      return 'BOOL';
    case 'formulaValue':
      return 'STRING';
    case 'errorValue':
      return 'STRING';
    default:
      return 'STRING';
  }
};
 
const getCellsType = (cells) => {
  // For now we assume that every cell in a column shares the same type (except for the headerRow)
  // In the future, we can make this smarter by looking at `n` cells and picking the widest type
  // We prune null types which happens if the cell is EMPTY
  const filteredArray = cells.map(getCellType).filter(t => !!t);
  return filteredArray.length ? filteredArray[0] : null;
};
 
const generateFieldsFromRows = (headerRow, rowsForTypeCheck, columnMetadata) => {
  let unnamedColumnCount = 0;
  const fields = [];
  headerRow.values.forEach((cell, i) => {
    // Filter columns which have no data in the first 5 rows
    // and don't have a value for the column title
    let label = cell.formattedValue;
    const filteredRows = rowsForTypeCheck.filter(({ values }) => !!values);
    const type = getCellsType(filteredRows.map(row => row.values[i]));
    if (!label && !type) {
      return;
    }
    const { metadataId } = columnMetadata[i].developerMetadata.find(metadata => metadata.metadataKey === 'thunkableId');
    if (!label) {
      unnamedColumnCount += 1;
      label = `Unnamed Column ${unnamedColumnCount}`;
    }
    fields.push({
      label,
      name: `${metadataId}-${i}`,
      type: type || 'STRING',
    });
  });
  return fields;
};

const syncFromSpreadsheet = async (owner, spreadsheetId, columnHeaderRowNumber, firstDataRowNumber) => {

    const { auth, user } = await getOauth2Client(owner, spreadsheetId);
    projectOwner = user;
    const sheets = google.sheets('v4');
 
    const sheetsProperties = await sheets.spreadsheets.get(
      {
        auth,
        spreadsheetId,
        includeGridData: true,
      });
    let sheetsCopy = sheetsProperties && sheetsProperties.data && sheetsProperties.data.sheets;
    // iterate through all the sheets to find the columns that need ids
    const columnsToAddIdsTo = [];
    sheetsCopy.forEach((sheet) => {
      const sheetId = sheet.properties.sheetId;
      sheet.data[0].columnMetadata.forEach((column, index) => {
        if (!column.developerMetadata) {
          columnsToAddIdsTo.push({
            sheetId,
            index
          });
        }
      });
    });
    // generate request to add ids to the sheets that need them
    if (columnsToAddIdsTo.length > 0) {
      const requests = columnsToAddIdsTo.map(({ sheetId, index }) => ({
        createDeveloperMetadata: {
          developerMetadata: {
            metadataKey: 'thunkableId',
            location: {
              dimensionRange: {
                sheetId,
                dimension: 'COLUMNS',
                startIndex: index,
                endIndex: index + 1
              }
            },
            visibility: 'PROJECT',
          }
        }
      }));
      // send request to add ids
      const batchUpdate = {
        auth,
        spreadsheetId,
        resource: {
          requests
        },
        includeSpreadsheetInResponse: true,
        responseIncludeGridData: true,
      };
      const updatedSheetData = await sheets.spreadsheets.batchUpdate(batchUpdate);
      sheetsCopy = updatedSheetData.data.updatedSpreadsheet.sheets;
    }
    return getCollectionsFromSheets(sheetsCopy, columnHeaderRowNumber, firstDataRowNumber);

const updateValuesInRow = async (owner, collection, spreadsheetId, firstRowWithData, metadataValue, values) => {
  let projectOwner;
 
  try {
    const { auth, user } = await getOauth2Client(owner, spreadsheetId);
    projectOwner = user;
    const sheets = google.sheets('v4');
    const dataFilters = getDataFilterForMetadataValue(metadataValue, firstRowWithData, collection.name);
    const dataFilter = {
      spreadsheetId,
      resource: {
        dataFilters
      },
      auth,
    };
    const response = await sheets.spreadsheets.values.batchGetByDataFilter(dataFilter);
 
    const sheetValues = response.data.valueRanges[0].valueRange.values[0];
    const rowValuesLength = Math.max(sheetValues.length, collection.fields.length);
    // Since we send the entire row back, the initial values need to be null
    // to indicate to Google that no change should be made
    const rowValues = new Array(rowValuesLength).fill(null);
    values.forEach(({ fieldId, value }) => {
      const field = collection.fields.find(({ id }) => id === fieldId);
      const index = field.name.split('-')[1];
      rowValues[index] = value || '';
    });
 
    const resource = {
      data: [
        {
          dataFilter: dataFilters[0],
          values: [rowValues],
          majorDimension: 'ROWS'
        }
      ],
      includeValuesInResponse: true,
      valueInputOption: 'USER_ENTERED',
    };
    const request = {
      auth,
      spreadsheetId,
      resource,
    };
    const result = await sheets.spreadsheets.values.batchUpdateByDataFilter(request);
    const updatedDataValues = result.data.responses[0].updatedData.values[0];
    const updatedValuesWithFieldIds = collection.fields.map(({ id, name }) => ({
      fieldId: id,
      value: updatedDataValues[name.split('-')[1]] || ''
    }));
    return {
      id: metadataValue,
      values: updatedValuesWithFieldIds
    };
  } catch (error) {
    console.error(`Failed to get spreadsheet data from id ${spreadsheetId}`);
    console.error(`The error is  ${error}`);
    const errorMessage = handleError(error, projectOwner);
    return { error: errorMessage };
  }

const getColumnValuesFromSpreadsheet = async (owner, collection, spreadsheetId, fieldId, firstRowWithData = 2) => {
  let projectOwner;
 
  try {
    const { auth, user } = await getOauth2Client(owner, spreadsheetId);
    projectOwner = user;
    const sheets = google.sheets('v4');
 
    const values = [];
 
    const spreadsheetData = await sheets.spreadsheets.get(
      {
        auth,
        spreadsheetId,
        includeGridData: true,
        ranges: collection.label,
      });
    const sheetData = spreadsheetData.data.sheets[0].data[0];
    if (!sheetData.rowData) {
      // values is still an empty array at this point
      return {
        fieldId,
        values
      };
    }
 
    const startIndex = firstRowWithData - 1;
 
    for (let i = startIndex; i < sheetData.rowData.length; i++) {
      const rowData = sheetData.rowData[i];
      const hasValues = !!rowData.values;
      const allEmpty = hasValues && sheetData.rowData[i].values.every(({ formattedValue }) => !formattedValue);
      if (!allEmpty) {
        if (fieldId === 'id') {
          const rowMetadata = get(sheetData, `rowMetadata[${i}].developerMetadata`, []);
          const rowWithId = rowMetadata.find(data => data.metadataKey === 'thunkableId' && isUuid(data.metadataValue));
          if (rowWithId && rowWithId.metadataValue) {
            values.push({ value: rowWithId.metadataValue });
          }
        } else {
          const field = collection.fields.find(({ id }) => id === fieldId);
          const fieldIndex = field.name.split('-')[1];
          const value = get(rowData, `values[${fieldIndex}].formattedValue`, '');
          values.push({ value });
        }
      }
    }
    return {
      fieldId,
      values
    };
  } catch (error) {
    console.error(`Failed to get spreadsheet data from id ${spreadsheetId}`);
    console.error(`The error is  ${error}`);
    const errorMessage = handleError(error, projectOwner);
    return { error: errorMessage };
