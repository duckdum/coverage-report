const updateFieldsForCollection = (newCollection, storedCollection, configType) => {
  const newFieldNames = new Set();
  const storedFieldNames = new Set();
  storedCollection.fields.forEach((field) => {
    const name = dataSourceMap[configType].parseFieldName(field.name);
    storedFieldNames.add(name.toString());
  });
 
  newCollection.fields.forEach((field) => {
    const name = dataSourceMap[configType].parseFieldName(field.name);
    newFieldNames.add(name);
    if (storedFieldNames.has(name)) {
      const exisitingField = storedCollection.fields.find(x => (
        dataSourceMap[configType].parseFieldName(x.name) === name
      ));
      exisitingField.name = field.name;
      exisitingField.type = field.type;
      exisitingField.label = field.label;
    } else {
      const dataSourceField = DataSourceField.create(field, storedCollection);
      storedCollection.fields.push(dataSourceField);
    }
  });
 
  const fieldsToRemove = storedCollection.fields.filter((field) => {
    const name = dataSourceMap[configType].parseFieldName(field.name);
    return !newFieldNames.has(name);
  });
  fieldsToRemove.forEach(field => field.remove());
};

const syncFromDataSource = async (configuration, dataSource) => {
  const configType = configuration.type;
  const metaDataFromDataSource = await dataSourceMap[configType].syncMetaData(configuration);
  const fromDataSourceCollectionNames = new Set();
  const dataSourceCollectionNames = new Set();
 
  dataSource.collections.forEach((collection) => {
    const name = dataSourceMap[configType].parseCollectionName(collection.name);
    dataSourceCollectionNames.add(name);
  });
  // if it exists in the data source
  if (Array.isArray(metaDataFromDataSource)) {
    metaDataFromDataSource.forEach((collection) => {
      const name = dataSourceMap[configType].parseCollectionName(collection.name);
      fromDataSourceCollectionNames.add(name);
      // and also in our database
      if (dataSourceCollectionNames.has(name)) {
        const storedCollection = dataSource.collections.find(x => x.name === name);
        updateCollectionLabel(collection, storedCollection);
        updateFieldsForCollection(collection, storedCollection, configType);
      } else {
        // but not in our database
        const newCollection = DataSourceCollection.create(dataSource, collection);
        dataSource.collections.push(newCollection);
      }
    });
    // if it exists in our database but not in the data source
    const collectionsToRemove = dataSource.collections.filter((collection) => {
      const name = dataSourceMap[configType].parseCollectionName(collection.name);
      return !fromDataSourceCollectionNames.has(name);
    });
    collectionsToRemove.forEach(collection => collection.remove());
  }
  return dataSource;
};

const createDataSourceCopyWithDummyData = (dataSource, owner) => {
  const idMap = {};
  const dataSourceId = uuidv4();
  idMap[dataSource.id] = dataSourceId;
  const oldCollections = dataSource.collections || [];
  const collections = oldCollections.map(({ id, fields, name, label }) => {
    const collectionId = uuidv4();
    idMap[id] = collectionId;
    const newFields = fields.map(({ id, label, name, type }) => {
      const fieldId = uuidv4();
      idMap[id] = fieldId;
      return {
        label,
        name,
        type,
        id: fieldId,
      };
    });
    const edges = [];
    for (let i = 0; i < NUMBER_OF_ITEMS_TO_GENERATE_IN_COPY; i++) {
      edges.push({
        cursor: uuidv4(),
        node: {
          id: uuidv4(),
          values: newFields.map(({ id, type }) => ({
            fieldId: id,
            value: getDefaultValuesByType(type)
          }))
        }
      });
    }
    return {
      fields: newFields,
      id: collectionId,
      data: {
        edges,
      },
      name,
      label,
    };
  });
 
  return {
    dataSource: {
      name: dataSource.name,
      id: dataSourceId,
      owner,
      collections,
      configuration: {
        owner,
        name: dataSource.name,
        id: uuidv4(),
        type: 'LOCAL',
      }
    },
    idMap
  };
};
