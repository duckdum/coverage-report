projectSchema.methods.replaceDataSourceWithLocalCopy = async function (dataSource) {
  if (!dataSource) return;
  const {
    blockly,
    components,
    dataSourceLinks,
    localDataSources,
    owner,
  } = this;
  const dataSourceId = dataSource.id;
  const {
    dataSource: newDataSource,
    idMap
  } = createDataSourceCopyWithDummyData(dataSource, owner);
  let existingDataSourceLink;
  const dataSourceLink = {
    id: uuidv4(),
    dataSource: newDataSource
  };
  localDataSources.push(dataSourceLink);
  for (let i = 0; i < dataSourceLinks.length; i++) {
    const link = dataSourceLinks[i];
    const linkedDataSource = get(link, 'dataSource');
    if (linkedDataSource.toString() === dataSourceId.toString() || linkedDataSource.id === dataSourceId) {
      existingDataSourceLink = link;
      idMap[link.id] = dataSourceLink.id;
    }
  }
  let shouldMarkComponentsModified = false;
  ComponentUtils.modifyComponentInPlace(components, ({ bindings }) => {
    if (bindings) {
      Object.keys(bindings).forEach((key) => {
        const binding = bindings[key];
        if (binding.type === 'COMPONENT_PROPERTY_TABLE_ELEMENT') {
          if (idMap[binding.column]) {
            shouldMarkComponentsModified = true;
            const newColumn = idMap[binding.column];
            binding.column = newColumn;
            binding.row = 0;
          }
        } else if (binding.type === 'DATA_SOURCE_TABLE') {
          if (binding.dataSourceLinkId.toString() === existingDataSourceLink.id.toString()) {
            shouldMarkComponentsModified = true;
            binding.dataSourceLinkId = dataSourceLink.id;
            binding.collectionId = idMap[binding.collectionId];
            binding.isLocal = true;
          }
        }
      });
    }
  });
  if (blockly) {
    Object.keys(blockly).forEach((screenId) => {
      blockly[screenId].xml = replaceAll(blockly[screenId].xml, idMap);
      blockly[screenId].code = replaceAll(blockly[screenId].code, idMap);
    });
    if (shouldMarkComponentsModified) {
      this.markModified('blockly');
    }
  }
  existingDataSourceLink.remove();
  this.markModified('components');
};

projectSchema.pre('save', function (next) {

    this.firstScreenId = firstScreen ? firstScreen.id : null;

  }

  this.updateApiComponentInfo();
  this.updateUIComponentInfo();
 
  setHash(this);
  // Detect corrupted saves and prevent them from happening
  if (Object.keys(this.apiComponents).length === 0) {
    throw new Error('Contact the Thunkable team with your project url. Corrupted file cannot be saved');
  }
  // If the process of storing the project in the filesystem fails,
  // storedInFileSystem will remain false, and the project fields will be stored in the db
  this.storedInFileSystem = false;
  const projectFileName = this._id + '-' + uuidv4();
  const filePath = path.join(__dirname, `../uploads/${projectFileName}.json`);
  modifiedPaths[this._id] = this.modifiedPaths() || [];
  const projectToStore = pick(this, fileStorageFields);
 
  const jsonToStore = JSON.stringify(projectToStore);
  const hasAllFields = every(fileStorageFields, field => Object.keys(projectToStore).includes(field));
  const isValidToStore = jsonToStore !== JSON.stringify({}) && hasAllFields;
  const eventProperties = { projectId: this.id, filePath, jsonToStore, projectFileName };
  const projectOwner = { id: this.owner, email: this.email };
 
  if (!isValidToStore) {
    const errorMessage = 'Issues with project.json';
    trackingController.trackError(Error(errorMessage), projectOwner, errorMessage, eventProperties, true);
    next();
  }
 
  fs.writeFile(filePath, jsonToStore, (error) => {
    if (error) {
      trackingController.trackError(error, projectOwner, 'Error Writing Project File', eventProperties, true);
      next();
    }
    storage.storeBackedUpProjectFile(this, filePath).then(() => {
      this.$locals.storedInFileSystemProperties = {};
      fileStorageFields.forEach(fileStorageField => {
        this.$locals.storedInFileSystemProperties[fileStorageField] = this[fileStorageField];
        this[fileStorageField] = {};
      });
      this.storedInFileSystem = true;
      next();
    }).catch((err) => {
      trackingController.trackError(err, projectOwner, 'Error Storing Project File', eventProperties, true);
      next();
    });
  });
});
