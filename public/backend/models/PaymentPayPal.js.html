const fetchPaypalAPI = async (route, method, data) => {
  const paypalURL = `${process.env.PAYPAL_API_URL}/${route}`;
  const auth = `${process.env.PAYPAL_CLIENT_ID}:${process.env.PAYPAL_CLIENT_SECRET}`;
  const auth64 = Buffer.from(auth).toString('base64');
  const options = {
    method,
    headers: {
      'Content-Type': 'application/json',
      Authorization: `Basic ${auth64}`,
    }
  };
 
  if (data) options.body = JSON.stringify(data);
 
  const response = await fetch(paypalURL, options);
  let result = await response.text();
 
  try {
    const json = JSON.parse(result);
    if (json) result = json;
  } catch (err) {
    // This probably means the response is text
    if (!result) result = response.ok;
  }
 
  return result;
};

paypalSchema.methods.requestUpdate = async function requestUpdate(plan) {
  if (this.currentSubscriptionEnd) await this.reactivateSubscription();
 
  const planMap = GRANDFATHER_PLAN_IDS.includes(this.paypal.planId)
    ? { ...PLAN_TO_PAYPAL_PLAN_ID, ...GRANDFATHER_PLAN_TO_PAYPAL_PLAN_ID }
    : PLAN_TO_PAYPAL_PLAN_ID;
  const data = { plan_id: planMap[plan] };
  const route = `v1/billing/subscriptions/${this.paypal.subscriptionId}/revise`;
  const updated = await fetchPaypalAPI(route, 'POST', data);
 
  if (updated) {
    const approveLink = updated.links && updated.links.find(link => link.rel === 'approve');
    this.paypal.approveLink = approveLink && approveLink.href;
    this.paypal.nextPlanIdPending = updated && updated.plan_id;
  }
 
  return this;
};
 
paypalSchema.methods.update = function update() {
  if (!this.paypal.nextPlanIdPending) return this;
 
  const nextPlan = Object.keys(PLANNAME_TO_PAYPAL_PLAN_ID)
    .find(planName => PLANNAME_TO_PAYPAL_PLAN_ID[planName] === this.paypal.nextPlanIdPending);
  const isResumingCurrentPlan = nextPlan === this.plan;
  this.nextPlan = isResumingCurrentPlan ? null : nextPlan;
  this.paypal.nextPlanId = isResumingCurrentPlan ? null : this.paypal.nextPlanIdPending;
  this.paypal.nextPlanIdPending = null;
  this.paypal.approveLink = null;
 
  if (this.currentSubscriptionEnd) this.currentSubscriptionEnd = null;
 
  return this;
};

paypalSchema.methods.handleUserCancelFromPayPal = async function handleUserCancelFromPayPal() {

  return this;
};

paypalSchema.methods.suspend = async function suspend() {

  const subscription = await fetchPaypalAPI(route, 'GET');
  const nextBillingTime = subscription.billing_info && subscription.billing_info.next_billing_time;
  const suspended = await fetchPaypalAPI(`${route}/suspend`, 'POST');
  const suspendedSubscription = await fetchPaypalAPI(route, 'GET');

      this.nextPlan = null;
      this.paypal.nextPlanId = null;
    }
  }
 
  return this;
};
 
paypalSchema.methods.cancel = async function cancel() {
  const route = `v1/billing/subscriptions/${this.paypal.subscriptionId}`;
  const cancelled = await fetchPaypalAPI(`${route}/cancel`, 'POST');
  const cancelledSubscription = await fetchPaypalAPI(route, 'GET');

    if (this.nextPlan) {

    }
  }
 
  return this;
};

paypalSchema.methods.listAllTransactions = async function listAllTransactions() {
  if (!this.paypal.subscriptionIds || (this.paypal.subscriptionIds && this.paypal.subscriptionIds.length === 0)) return [];
 
  const endTime = new Date().toISOString();
  const startTime = moment(new Date()).subtract(2, 'years').toISOString();
  const promises = await allSettled(
    this.paypal.subscriptionIds.map(async (subscriptionId) => {
      try {
        const route = `/v1/billing/subscriptions/${subscriptionId}/transactions?start_time=${startTime}&end_time=${endTime}`;
        const result = await fetchPaypalAPI(route, 'GET');
        const formattedTransactions = result.transactions.map((transaction) => {
          if (!transaction || (transaction && transaction.status !== 'COMPLETED')) return null;
 
          const amountPaid = transaction.amount_with_breakdown && transaction.amount_with_breakdown.gross_amount.value;
          const formattedAmountPaid = amountPaid && amountPaid.substring(0, amountPaid.indexOf('.'));
          const formatted = {
            id: transaction.id,
            amount_paid: formattedAmountPaid,
            date: moment(transaction.time).format('ll'),
            plan_nickname: 'Thunkable membership',
          };
 
          return formatted;
        });
 
        const filtered = formattedTransactions.filter(transaction => !!transaction);
        return filtered;
      } catch (err) {
        return null;
      }
    })
  );
 
  const settled = await promises;
  const allTransactions = [];
  settled.forEach((data) => {
    if (data && data.status === 'fulfilled' && data.value && data.value[0]) {
      allTransactions.push(data.value[0]);
    }
  });
 
  return allTransactions;
};
