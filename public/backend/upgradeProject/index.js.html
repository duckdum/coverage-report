  function handleVariableUuidToBlocks(currentBlock) {
    if (isVariableDef(currentBlock)) {
      if (!currentBlock.mutation) {
        currentBlock = { 
          $:currentBlock['$'],
          mutation: [{$:{}}],
          ...currentBlock
        }
      }
      if (!currentBlock.mutation[0].$.variable_uuid) {
        currentBlock.mutation[0].$.variable_uuid = uuidv4();
      }
      let scopeIndex = currentBlock.field[0].$.name === 'SCOPE' ? 0 : 1;
      const variableScope = currentBlock.field[scopeIndex]._;
      const variableName = currentBlock.field[(scopeIndex + 1) % 2]._;
      const variableUuid = currentBlock.mutation[0].$.variable_uuid;
      variableKeyToUuid[variableScope + ':' + variableName] = variableUuid;
      project.variables[variableUuid] = {scope: variableScope, name: variableName};
    }
    return currentBlock;
  };
  if (!project.variables) {
    project.variables = {};
  }
  traverseBlocks(project, handleVariableUuidToBlocks);

const traverseBlocks = (project, testFxn) => {
  const parseString = xml2js.parseString;
 
  try {
    for (const screenId in project.blockly) {
      const xmlString = project.blockly[screenId].xml;
      let xmlDoc;
      // NOTE: using thunkable/node-xml2js#48abcfb343c0971dfa3290f59b15300799a690bf to include empty top level nodes
      parseString(xmlString, { includeWhiteChars: true }, function (err, result) {
        xmlDoc = result;
        if (!xmlDoc || !xmlDoc.xml || !xmlDoc.xml.block) return;
        const blockElem = xmlDoc['xml']['block'];
        for (let i = 0; i < blockElem.length; i++) {
          blockElem[i] = processBlock(blockElem[i], testFxn);
        }
        const builder = new xml2js.Builder({ headless: true, renderOpts: { pretty: false }, allowSurrogateChars: true });
        project.blockly[screenId].xml = builder.buildObject(xmlDoc);
      });
    }
  } catch (e) {
    bugsnag.notify(e);
  }
};
 
const processBlock = (currentBlock, testFxn) => {
  currentBlock = testFxn(currentBlock);
  if (currentBlock['statement']) {
    const stateLength = currentBlock['statement'].length;
    for (let j = 0; j < stateLength; j++) {
      if (currentBlock['statement'][j]['block']) {
        currentBlock['statement'][j]['block'][0] = processBlock(currentBlock['statement'][j]['block'][0], testFxn);
      }
    }
  }
  if (currentBlock['value']) {
    const stateLength = currentBlock['value'].length;
    for (let k = 0; k < stateLength; k++) {
      if (currentBlock['value'][k]['shadow']) {
        if (!currentBlock['value'][k]['block']) {
          currentBlock['value'][k]['shadow'][0] = processBlock(currentBlock['value'][k]['shadow'][0], testFxn);
        }
      }
      if (currentBlock['value'][k]['block']) {
        currentBlock['value'][k]['block'][0] = processBlock(currentBlock['value'][k]['block'][0], testFxn);
      }
    }
  }
  if (currentBlock['next']) {
    currentBlock['next'][0]['block'][0] = processBlock(currentBlock['next'][0]['block'][0], testFxn);
  }
  return currentBlock;
};

  try {

      projectName: project.displayName,
      projectOwner: project.email,
    };
    trackingController.trackError({ ...error, ...eventProperties }, user, 'Error Running Project Upgrader', eventProperties);
  }

// This adds the relativeX and relativeY properties to drag and drop components
const upgradeRelativeXAndY = (project) => {
  // ignore this upgrader if the project isn't dnd
  if (!project.usesDragDropUi) return;
  // for every component id (key), store the list of parents ids (values)
  const componentIdToParentIds = ComponentUtils.getParentIdsByIdForComponents(project.components);
 
  const childrenUseSubComponentRenderingIdSet = new Set();
  const parentIdToPosition = {};
  const screenIds =  new Set();
  ComponentUtils.componentSearch(project.components, (component) => {
    if (component.children &&
        get(component, 'properties.X', undefined) !== undefined &&
        get(component, 'properties.Y', undefined) !== undefined
    ) {
      // store the x/y position of every parent for quick lookup access later
      parentIdToPosition[component.id] = {
        X: component.properties.X,
        Y: component.properties.Y,
      };
    }
    // we need to handle components that have a screen as a direct parent
    // differently since they are at the top level. Here, we just store
    // the id of every screen we encounter
    if (component.type === 'Screen') {
      screenIds.add(component.id);
    }
    // keep track of the components that we should not look at the children of
    // to add relativeX and relativeY properties, like data viewers, and canvas
    if (ComponentUtils.shouldChildrenUseSubComponentRendering(component.type)) {
      childrenUseSubComponentRenderingIdSet.add(component.id);
    }
  });
  ComponentUtils.modifyComponentInPlace(project.components, (component) => {
    // check that we are setting the relative x/y value on components that
    // don't have it already set, and that have x/y values
    if (get(component, 'properties.X', undefined) !== undefined &&
        get(component, 'properties.Y', undefined) !== undefined &&
        get(component, 'properties.RelativeX', undefined) === undefined &&
        get(component, 'properties.RelativeY', undefined) === undefined
      ) {
        parentList = componentIdToParentIds[component.id];
        // check to make sure that no parent is a canvas or data viewer
        const hasDataViewerOrCanvasArray = parentList.map(
          parentId => childrenUseSubComponentRenderingIdSet.has(parentId));
        if (hasDataViewerOrCanvasArray.indexOf(true) === -1) {
          const parentId = parentList[0];
          let parentPosition = parentIdToPosition[parentId];
          // set the relative x/y position either based on the parent's position
          // or equal to x/y if the screen is the direct parent component
          if (parentPosition) {
            component.properties.RelativeX = component.properties.X - parentPosition.X;
            component.properties.RelativeY = component.properties.Y - parentPosition.Y;
          } else if (screenIds.has(parentId)) {
            component.properties.RelativeX = component.properties.X;
            component.properties.RelativeY = component.properties.Y;
          }
        }
      }
  });
}
