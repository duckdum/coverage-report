const { exec } = require('child_process');
const fastlanePath = `${__dirname}/../mis/2fa/fastlane`;
const has2FAError = 'Please enter the 6 digit code:';
const has2FAMessage = 'Your Apple ID two-factor authentication is enabled.'; 
const invalidError = 'Invalid username and password combination';
const invalidMessage = 'Your Apple ID username/password is invalid.';
 
class AppleAccount {
  static async check(id, password) {
    const fastlaneCommand = `cd ${fastlanePath} && bundle exec fastlane checkAppleLogIn id:'${id}' password:'${password}'`;
    await AppleAccount.customizeExecError(fastlaneCommand);
  }
 
  static customizeExecError(command) {
    return new Promise((resolve, reject) => {
      exec(command, { maxBuffer: 20000 * 1024 }, (error, stdout, stderr) => {
        if (error !== null) {
          error.stdout = stdout;
          error.stderr = stderr;
          
          if (AppleAccount.is2FAEnable(stdout)) {
            error.message = has2FAMessage;
          } else if (AppleAccount.isLoginInvalid(stdout)) {
            error.message = invalidMessage;
          }
 
          reject(error);
        } else {
          resolve({ stdout, stderr });
        }
      });
    });
  }
  
  static is2FAEnable(stdout) {
    return stdout && stdout.includes(has2FAError);
  }
  
  static isLoginInvalid(stdout) {
    return stdout && stdout.includes(invalidError);
  }
}
 
module.exports = AppleAccount;
