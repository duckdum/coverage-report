  static find(filePath, regex, group) {
    let stream = fs.createReadStream(`${filePath}`);
    return new Promise((resolve, reject) => {
 
      stream.on('data', (data) => {
        const found = ('' + data).match(regex);
        const target = found && found[group];
        resolve(target);
      });
      stream.on('error', (error) => {
        reject(error);
      });
    });
  }

  static encryptFile(filePath) {
    const cipher = crypto.createCipher('aes-256-cbc', encryptKey);
 
    let input = fs.createReadStream(`${filePath}`);
    let output = fs.createWriteStream(`${filePath}.enc`);
    input.pipe(cipher).pipe(output);
    return new Promise((resolve, reject) => {
      output.on('finish', resolve);
      output.on('error', (error) => {
        reject(error);
      });
    });
  }
 
  static decryptFile(filePath) {
    const cipher = crypto.createDecipher('aes-256-cbc', encryptKey);
    const decryptFilePath = filePath.substring(0, filePath.indexOf('.enc'));
 
    let input = fs.createReadStream(`${filePath}`);
    let output = fs.createWriteStream(`${decryptFilePath}`);
    input.pipe(cipher).pipe(output);
    return new Promise((resolve, reject) => {
      output.on('finish', resolve);
      output.on('error', (error) => {
        reject(error);
      });
    });
  }
 
  static async downloadAndDecryptFromGCS(tempDir, gcsFilePath) {
    const file = bucket.file(gcsFilePath);
    const metadata = (await file.getMetadata())[0].metadata;
    await file.download({ destination: `${tempDir}/${metadata.fileName}` });
 
    if (metadata.fileName.endsWith('enc')) {
      await AppleFile.decryptFile(`${tempDir}/${metadata.fileName}`);
    }
  }
 
  static async uploadToGCS(destination, localPath, makePublic, metadata) {
    const fileName = path.basename(localPath);
    const mimeType = mimetype.lookup(localPath) || 'application/octet-stream';
    const options = {
      destination: destination,
      metadata: {
        contentType: mimeType,
        metadata: {
          fileName: fileName,
          ...metadata,
        }
      }
    };
 
    const uploadResponse = await bucket.upload(localPath, options);
    const uploadedFile = uploadResponse[0];
 
    if (makePublic) {
      await uploadedFile.makePublic();
    }
 
    return uploadedFile;
  };
