import React, { Component } from 'react';
import { View, Platform } from 'react-native';
import { Video as ExpoVideo } from 'expo-av';
import { genSourceProp, getAssetPromise } from './SourceProcessor';
import { makeCancelable } from '../utils/PromiseUtils';
 
class Video extends Component {
  constructor(props) {
    super(props);
    this.state = {
      loadingAsset: true,
      isPlaying: props.Play,
    };
  }
 
  componentDidMount() {
    this.processSource(this.props);
  }
 
  componentDidUpdate(prevProps, prevState) {
    if (prevProps.File !== this.props.File) {
      this.setState({ loadingAsset: true });
      this.processSource(this.props);
    }
 
    if (((prevState.loadingAsset !== this.state.loadingAsset && this.state.loadingAsset)
      || (prevState.isLoaded !== this.state.isLoaded && this.state.isLoaded)
      || (prevState.assetUri !== this.state.assetUri && this.state.assetUri)
      || prevProps.Play !== this.props.Play)
      && this.playbackObject) {
 
      if (this.props.Play) {
        if (this.state.positionMillis && this.state.durationMillis && this.state.positionMillis === this.state.durationMillis) {
          this.playbackObject.playFromPositionAsync(0);
        } else {
          this.playbackObject.playAsync();
        }
      } else {
        this.playbackObject.pauseAsync();
      }
    }
 
    if (prevState.isPlaying !== this.state.isPlaying) {
      const newPlayValue = this.state.isPlaying;
      const props = this.props;
      props.stateSetter(function(prevState) {
        const nextState = { ...prevState };
        props.propSetter(props.id, 'Play', newPlayValue, nextState);
        return nextState;
      });
    }
  }
 
  async componentWillUnmount() {
    await this.assetUriPromise && this.assetUriPromise.cancel();
  }
 
  async processSource(props) {
    this.assetUriPromise && this.assetUriPromise.cancel();
    this.assetUriPromise = makeCancelable(getAssetPromise('File', props));
    let uri;
    try {
      uri = await this.assetUriPromise.promise;
      this.setState({ loadingAsset: false, assetUri: uri });
    } catch (e) {
      this.setState({ loadingAsset: false, assetUri: '' });
    }
  }
 
  _onPlaybackStatusUpdate = playbackStatus => {
    this.setState(playbackStatus);
  };
 
  _handleVideoRef = component => {
    if (component) {
      this.playbackObject = component;
      this.playbackObject.setOnPlaybackStatusUpdate(this._onPlaybackStatusUpdate);
    }
  }
 
  render() {
    if (this.state.loadingAsset || !this.props.File || !this.props.Visible) {
      return null;
    } else {
      const sourceProp = genSourceProp(this.props, this.state.assetUri);
      let style = this.props.style;
      if (Platform.OS === 'web') {
        if (!this.props.style.alignSelf) {
          style = {
            height: this.props.style.height,
            width: this.props.style.width,
            position: this.props.style.position,
            left: this.props.style.left,
            top: this.props.style.top,
            zIndex: this.props.style.zIndex
          };
        }
      }
      return (
        <View style={style} onLayout={this.props.onLayout}>
          <ExpoVideo
            ref={this._handleVideoRef}
            resizeMode='contain'
            source={{ uri: sourceProp.uri }}
            style={{ flex: 1 }}
            useNativeControls={true}
          />
        </View>
      );
    }
  }
}
 
export default Video;
