import React, { Component } from 'react';
import { View } from 'react-native';
import { WebView as RNWebView } from 'react-native-webview';
import { get } from 'lodash';
import { genSourceProp, getAssetPromise, isUriLocal } from './SourceProcessor';
import { makeCancelable } from '../utils/PromiseUtils';
import {v4 as uuidv4} from "uuid";
import { requestPermissionsAsync } from 'expo-ads-admob';
 
export default class WebView extends Component {
 
  constructor(props) {
    super(props);
    this.state = {loadingAsset: true, renderedOnce: false};
    this.webViewRef = React.createRef();
    this.navigationChange = false;
    this.onWebViewMessage = this.onWebViewMessage.bind(this);
    this.onNavStateChange = this.onNavStateChange.bind(this);
    this.postMessageReturnUuidToResolver = {};
  }
 
  componentDidMount() {
    // This is a temp fix. We should use an independent package, not expo-ads-admob.
    if (this.props.userTrackingUsageDescription) {
      requestPermissionsAsync();
    }
  }
 
  componentWillMount() {
    this.assetUriPromise = makeCancelable(getAssetPromise('URL', this.props));
    this.assetUriPromise.promise
    .then((uri) => {
      if (uri) {
        this.setState({loadingAsset: false, assetUri: uri});
      }
    })
    .catch(() => {}); // Ignore the rejection for now.
  }
 
  componentWillUnmount() {
    this.assetUriPromise.cancel();
  }
 
  componentDidUpdate(prevProps) {
    if (!(this.navigationChange) && this.props.URL !== prevProps.URL) {
      const assetUriPromise = getAssetPromise('URL', this.props);
      assetUriPromise.then((uri) => {
        if (uri) {
          this.setState({loadingAsset: false, assetUri: uri});
        }
      });
    }
    this.navigationChange = false;
  }
 
  reload() {
    if (this.webViewRef && this.webViewRef.reload && this.props.URL) {
      this.webViewRef.reload();
    }
  }
 
  Back() {
    if (this.webViewRef && this.webViewRef.goBack && this.props.URL) {
      this.webViewRef.goBack();
    }
  }
 
  Forward() {
    if (this.webViewRef && this.webViewRef.goForward && this.props.URL) {
      this.webViewRef.goForward();
    }
  }
 
  PostMessage(message) {
    if (this.webViewRef && this.webViewRef.postMessage) {
      this.webViewRef.postMessage(message);
    }
  }
 
  async ValueFromPostMessage(message) {
    const messageToPost = { type: 'ThunkablePostMessage', uuid: uuidv4(), message };
    return (await new Promise((resolve, reject) => {
      this.postMessageReturnUuidToResolver[messageToPost.uuid] = resolve;
      this.PostMessage(JSON.stringify(messageToPost));
    }));
  }
 
  onWebViewMessage(event) {
    try {
      let dataObject;
      try {
        dataObject = JSON.parse(get(event, 'nativeEvent.data', null));
      } catch (e) {
        // message is not valid json
      }
      if (dataObject && dataObject.type === 'ThunkablePostMessageReturnValue') {
        if (this.postMessageReturnUuidToResolver[dataObject.uuid]) {
          this.postMessageReturnUuidToResolver[dataObject.uuid](dataObject.returnValue);
          delete this.postMessageReturnUuidToResolver[dataObject.uuid];
        }
      } else if (this.props.ReceiveMessage) {
        this.props.ReceiveMessage(event.nativeEvent.data);
      }
    } catch(err) {
      return;
    }
  }
 
  onNavStateChange = (webViewState) => {
    this.navigationChange = true;
    const props = this.props;
    // This changes the URL property of this component in its parent, so that it is available to blocks calling
    // getProp()
    props.stateSetter(function(prevState) {
      const nextState = { ...prevState };
      props.propSetter(props.id, 'URL', webViewState.url, nextState);
      return nextState;
    }, () => {});
  }
 
  // On Android there is a potential race condition where the url gets set 
  // before webview file system permission and as a result local html file fails to load within the webview.
  // https://github.com/react-native-webview/react-native-webview/issues/656#issuecomment-561067913
  updateSource = () => {
    this.setState({ ...this.state, renderedOnce: true })
	};
  
  render() {
    if (this.state.loadingAsset) {
      return null;
    } else {
      const source = genSourceProp(this.props, this.state.assetUri);
      const wvProps = {
        ...this.props,
        ref: (webview) => { this.webViewRef = webview; },
        onMessage: this.onWebViewMessage,
        source: this.state.renderedOnce ? source : undefined,
        style: { flex: 1, minWidth: 1, minHeight: 1, opacity: 0.99 },
      };
      if (this.props.isDndScreen) {
        wvProps.style.width = this.props.style.width;
        wvProps.style.height = this.props.style.height;
      }
      // TODO: Update this if/when the below issue is resolved
      // https://github.com/react-native-community/react-native-webview/issues/767
      if (isUriLocal(source.uri)) {
        wvProps.originWhitelist = ['*'];
      }
      return (
        <View style={this.props.style} onLayout={this.props.onLayout}>
          <RNWebView { ...wvProps }
            allowFileAccess={true}
            allowFileAccessFromFileURLs={true}
            allowsFullScreenVideo={true}
            dataDetectorTypes="all"
            onNavigationStateChange={this.onNavStateChange}
            onLoadStart={this.updateSource}/>
        </View>
      );
    }
  }
}
