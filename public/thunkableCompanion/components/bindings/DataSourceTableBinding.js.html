import { get } from 'lodash';
import ComponentUtils from '../../shared/utils/ComponentUtils';
 
const DELETE = 'DELETE';
const DELETE_ALL = 'DELETE_ALL';
const UPDATE = 'UPDATE';
const CREATE = 'CREATE';
 
export default class DataSourceTableBinding {
 
  static createBindingFromPropObject(propObject, dataSourceManager, screen) {
    const { collectionId, dataSourceLinkId, lazyRetrieval, isLocal, } = propObject;
    return new DataSourceTableBinding(dataSourceLinkId, collectionId, lazyRetrieval, isLocal, dataSourceManager, screen);
  }
 
  // TODO handle updates (writes, etc)
  static updateComponentPropertyBindings(dataSourceLinkId, collectionId, value, screen) {
    const bindingForKeyCheck = new DataSourceTableBinding(dataSourceLinkId, collectionId);
    screen.getBindingManager().updateBoundValues(bindingForKeyCheck, value, screen);
  }
 
  constructor(dataSourceLinkId, collectionId, lazyRetrieval, isLocal, dataSourceManager, screen) {
    this.type = 'DATA_SOURCE_TABLE';
    this.dataSourceLinkId = dataSourceLinkId;
    this.collectionId = collectionId;
    this.lazyRetrieval = lazyRetrieval;
    this.isLocal = isLocal;
    this.dataSourceManager = dataSourceManager;
    this.getScreen = screen && screen.getScreen.bind(screen);
    // TODO store filter information, number of previous rows retreived, etc
  }
 
  getBindingKey() {
    return `${this.type}:${this.dataSourceLinkId}:${this.collectionId}`;
  }
 
  setInitialBoundValue(component, property, screen, nextState) {
    const dataTableValue = screen.getProp(component.id, property, nextState);
    const columns = dataTableValue ? dataTableValue.columns : [];
    this.columnOrder = columns.map(columnObject => columnObject.name);
    if (!this.lazyRetrieval) {
      this.setBoundPropertyValue(component, property);
    }
  }
 
  transformBoundInputToDerivedOutput(input, componentId, property, currentScreen) {
    let component;
    const shouldUpdateInPlace = currentScreen !== this.getScreen();
    if (shouldUpdateInPlace) {
      component = currentScreen.getComponentOnHiddenScreen(componentId)
    } else {
      component = ComponentUtils.getComponentById(componentId, this.getScreen().state.components, []);
    }
    const oldValue = component.properties[property];
    if (!get(oldValue, 'data.edges')) {
      // if it doesn't yet have the data it will automatically fetch it
      // when it mounts so we can safely avoid updating the props
      return;
    }
    const edges = [...oldValue.data.edges];
    if (input.type === DELETE) {
      const indexToDelete = edges.findIndex(({ node: { id }}) => id === input.node.id);
      if (indexToDelete > -1) {
        edges.splice(indexToDelete, 1);
      }
    } else if (input.type === UPDATE) {
      const edge = edges.find(({ node: { id }}) => id === input.node.id);
      if (edge) {
        edge.node.values = input.node.values;
      }
    } else if (input.type === CREATE) {
      // TODO: get the cursor somehow and update the pageinfo as well
      edges.push({ cursor: '', node: input.node });
    } else if (input.type === DELETE_ALL) {
      return {
        ...oldValue,
        data: {
          ...oldValue.data,
          edges: [],
        },
      };
    }
    
    const newValue = {
      ...oldValue,
      data: {
        ...oldValue.data,
        edges,
      },
    }
 
    if (shouldUpdateInPlace) {
      component.properties[property] = newValue;
    } else {
      return newValue;
    }
  }
 
  // component api functions
 
  // get next rows (number)
  async getNextRows(component, property, numberOfRows, after) {
    const oldValue = get(component, `props.${property}`, {}); 
    const oldEdges = get(oldValue, `data.edges`, []); 
    this.setPropOnScreen(component, property, { ...oldValue, isRefreshing: true });
    const tableDataValue = await this.getTableData(numberOfRows, after, false);
    if (!tableDataValue) {
      this.setPropOnScreen(component, property, { ...oldValue, isRefreshing: false });
      return;
    }
    const newValue = {
      data: {
        edges: [ ...oldEdges, ...tableDataValue.data.edges],
        pageInfo: tableDataValue.data.pageInfo
      },
      fields: tableDataValue.fields,
      isRefreshing: false,
    };
    this.setPropOnScreen(component, property, newValue);
  }
 
  async setBoundPropertyValue(component, property) {
    this.setPropOnScreen(component, property, { isRefreshing: true });
    const tableDataValue = await this.getTableData(100, null, true);
    if (!tableDataValue) return;
    this.setPropOnScreen(component, property, { ...tableDataValue, isRefreshing: false });
  }
 
  setPropOnScreen(component, property, tableDataValue) {
    this.getScreen().setState((prevState) => {
      const nextState = this.getScreen().spreadCopy(prevState);
      const componentId = component.id ? component.id : component.props.id;
      this.getScreen().setComponentProp(componentId, property, tableDataValue, nextState);
      return nextState;
    });
  }
 
  async getTableData(numberOfRows = 100, after, forceRefetch) {
    return await this.dataSourceManager.getDataSourceData(this.dataSourceLinkId, this.collectionId, numberOfRows, after, forceRefetch);
  }
}
