  static copy(component, updateCallback, parentComponentCopy) {
    const componentCopy = { ...component };
 
    if (this.hasChildren(componentCopy)) {
      const childComponentCopies = [];
 
      for (const childComponent of componentCopy.children)
        childComponentCopies.push(this.copy(childComponent, updateCallback, componentCopy));
 
      componentCopy.children = childComponentCopies;
    }
 
    updateCallback(componentCopy, parentComponentCopy);
 
    return componentCopy;
  }
 
  static copyWithValidators(component, updateCallback, componentValidator, childrenValidator, parentComponentCopy) {

        for (const childComponent of componentCopy.children)
          childrenCopy.push(this.copyWithValidators(childComponent, updateCallback, componentValidator, childrenValidator, parentComponentCopy));
      }
      componentCopy.children = childrenCopy;
      updateCallback(componentCopy, parentComponentCopy);
      return componentCopy;
    } else {
      return null;
    }
  }

  static componentSearch(component, testFxn) {
    // Breadth-first search of the component tree for the first Screen.
    let componentQueue = [component];
    if (Array.isArray(component)) {
      componentQueue = component.map((child) => child);
    }
    let nextComponent;
 
    while(componentQueue.length > 0) {
      nextComponent = componentQueue.shift();
      if (testFxn(nextComponent)) {
        return nextComponent;
      }
      if (!nextComponent.children) {
        continue;
      }
      nextComponent.children.forEach((child) => {
        componentQueue.push(child);
      });
    }
    return false;
  }

  static makeSizeCustomSetter(dimensionProp) {
    return (newValue, newComponent, newComponentParent, runtimeScreen) => {
      // if STP layout
      if (newComponentParent && ((newComponentParent.type === 'Screen' && !newComponentParent.isSimple) || newComponentParent.type !== 'Group')) {
        const styleDimensionProp = dimensionProp === 'Height' ? 'height' : 'width';
        const isParentLayoutRow = newComponentParent !== undefined &&
                                  newComponentParent.properties !== undefined &&
                                  newComponentParent.properties.Layout === 'row';
        let flexDimensionProp;
        let flexDimensionPropFillValue;
        const resetFlexDimensionProp = () => {
          if (flexDimensionProp === 'flex')
            newComponent.properties.style[flexDimensionProp] = 0;
          else
            delete newComponent.properties.style[flexDimensionProp];
        };
 
        newComponent.properties = { ...newComponent.properties };
        newComponent.properties.style = { ...newComponent.properties.style };
        if (dimensionProp === 'Height') {
          flexDimensionProp = isParentLayoutRow ? 'alignSelf' : 'flex';
        } else {
          flexDimensionProp = isParentLayoutRow ? 'flex' : 'alignSelf';
        }
        flexDimensionPropFillValue = flexDimensionProp === 'flex' ? 1 : 'stretch';
 
        newComponent.properties[dimensionProp] = newValue;
        if (newValue === this.Size.Automatic) {
          resetFlexDimensionProp();
          delete newComponent.properties.style[styleDimensionProp];
        } else if (newValue === this.Size.Fill) {
          newComponent.properties.style[flexDimensionProp] = flexDimensionPropFillValue;
          delete newComponent.properties.style[styleDimensionProp];
        } else if (typeof newValue === 'string') {
          const regex = /^(\d+(\.(\d+)?)?|(\.\d+))\%?$/;
          if (regex.exec(newValue)) {
            newValue += newValue.endsWith('%') ? '' : '%';
          } else {
            newValue = undefined;
          }
          newComponent.properties.style[styleDimensionProp] = newValue;
          newComponent.properties[dimensionProp] = newValue;
          resetFlexDimensionProp();
        } else {
          newComponent.properties.style[styleDimensionProp] = newValue;
          resetFlexDimensionProp();
        }
      } else {
        newComponent.properties[dimensionProp] = newValue;
        if (runtimeScreen) {
          let zIndex = newComponent.properties.style.zIndex
          runtimeScreen.fixZOrder(newComponent, zIndex, newComponentParent, runtimeScreen.getScreenComponent());
        }
      }
 
    };
  }

      });

    const uiComponentCounts = countBy(uiComponentTypes);

    const webApiLinks = {};

    const apiComponentTypes = apiComponents && apiComponents.map(component => {
      const URL = get(component, 'properties.URL', false);
      if (component.type === 'Web API' && URL) {
        totalOfWebApiLinks += 1;
        webApiLinks[`Web API url ${totalOfWebApiLinks}`] = URL;
      }
      return component.type;
    }).concat(apiComponentInBlockTypes);
    const apiComponentCounts = countBy(apiComponentTypes);
 
    return Object.assign({}, uiComponentCounts, apiComponentCounts, webApiLinks);
