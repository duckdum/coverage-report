import ViewStylePropTypes from './TPropTypes/TViewStylePropTypes';
import ColorPropType from './TPropTypes/TColorPropType';
import { TPropTypes as PropTypes } from './TPropTypes';
import TVisibleComponent from './TVisibleComponent';
 
const TSlider = {
  Type: 'Slider',
  ReactType: 'Slider',
  Category: 'USER_INTERFACE',
  Index: 7,
  SupportsWeb: true,
  ThunkableVersion: 35,
  TranslateLabelKey: "components.slider.name",
  TranslateTooltipKey: "components.slider.tooltip",
  DocumentationURL: "components.slider.documentation-url",
  TPropTypes: {
    /**
     * Initial minimum value of the slider. Default value is 0.
     */
    minimumValue: PropTypes.number.core().defaultValue(0)
                           .translatePropertyLabel("components.properties.minimumValue"),
 
    /**
     * Initial maximum value of the slider. Default value is 1.
     */
    maximumValue: PropTypes.number.core().defaultValue(10)
                           .translatePropertyLabel("components.properties.maximumValue"),
 
    /**
     * Initial value of the slider. The value should be between minimumValue
     * and maximumValue, which default to 0 and 10 respectively.
     * Default value is 5.
     *
     * *This is not a controlled component*, you don't need to update the
     * value during dragging.
     */
    value: PropTypes.number.core().defaultValue(5)
                    .translatePropertyLabel("components.properties.value"),
 
    /**
     * Step value of the slider. The value should be
     * between 0 and (maximumValue - minimumValue).
     * Default value is 0.
     */
    step: PropTypes.number.core().defaultValue(1)
                   .translatePropertyLabel("components.properties.step"),
 
    ...TVisibleComponent.TPropTypes,
    Width: TVisibleComponent.TPropTypes.Width.defaultValue('90%').dragDropDefaultValue(135),
    Height: TVisibleComponent.TPropTypes.Height.dragDropDefaultValue(25),
    /**
     * Used to style and layout the `Slider`.  See `StyleSheet.js` and
     * `ViewStylePropTypes.js` for more info.
     */
    style: PropTypes.shape(ViewStylePropTypes).containsCore(),
 
    /**
     * The color used for the track to the left of the button.
     * Overrides the default blue gradient image on iOS.
     */
    minimumTrackTintColor: ColorPropType.core().translatePropertyLabel("components.properties.minimumTrackTintColor"),
 
    /**
     * The color used for the track to the right of the button.
     * Overrides the default blue gradient image on iOS.
     */
    maximumTrackTintColor: ColorPropType.core().translatePropertyLabel("components.properties.maximumTrackTintColor"),
 
    /**
     * If true the user won't be able to move the slider.
     * Default value is false.
     */
    disabled: PropTypes.bool.core().translatePropertyLabel("components.properties.disabled"),
 
    /**
     * Assigns a single image for the track. Only static images are supported.
     * The center pixel of the image will be stretched to fill the track.
     * @platform ios
     */
    // trackImage: PropTypes.fileName.label('trackImage (iOS only)'),
 
    /**
     * Assigns a minimum track image. Only static images are supported. The
     * rightmost pixel of the image will be stretched to fill the track.
     * @platform ios
     */
    // minimumTrackImage: PropTypes.fileName.label('minimumTrackImage (iOS only)'),
 
    /**
     * Assigns a maximum track image. Only static images are supported. The
     * leftmost pixel of the image will be stretched to fill the track.
     * @platform ios
     */
    // maximumTrackImage: PropTypes.fileName.label('maximumTrackImage (iOS only)'),
 
    /**
     * Sets an image for the thumb. Only static images are supported.
     * @platform ios
     */
    // thumbImage: PropTypes.fileName.label('thumbImage (iOS only)'),
 
    /**
     * Color of the foreground switch grip.
     * @platform android
     */
    thumbTintColor: ColorPropType.core().label('thumbTintColor (Android only)')
                                 .translatePropertyLabel("components.properties.thumbTintColor"),
 
    /**
     * Callback continuously called while the user is dragging the slider.
     */
    onValueChange: PropTypes.funcForEvent({
        args: [
            {name: 'value', type: 'Number' , translateKey:'blocks.Slider.value' },
        ]
    }).core().tooltip('Whenever the value of the slider changes, trigger the following block(s)'),
 
    /**
     * Callback that is called when the user releases the slider,
     * regardless if the value has changed. The current value is passed
     * as an argument to the callback handler.
     */
    onSlidingComplete: PropTypes.funcForEvent({
        args: [
            {name: 'value', type: 'Number',translateKey:'blocks.Slider.value' },
        ]
    }).core().tooltip('When the sliding is complete, trigger the following block(s)'),
  }
 
}

    TVisibleComponent.getPositionGroup(),
    TVisibleComponent.getSpacingGroup(),
    TVisibleComponent.getStyle(['BackgroundColor']),
  ];
}
