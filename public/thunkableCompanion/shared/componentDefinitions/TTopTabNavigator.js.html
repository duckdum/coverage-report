import { TPropTypes } from './TPropTypes';
import ColorPropType from './TPropTypes/TColorPropType';
import { TAB_NAVIGATOR_TYPES, TOP_NAVIGATOR_TYPE } from './NavigatorList'
import { NAVIGAOTR_LAYOUT } from './DataViewerComponents';
import { getDeviceSizeFromTypeAndOrientation } from '../utils/DeviceSizes';
 
const TBO = 'config.tabBarOptions';
const scaledPressOpacityCustomSetter = (newValue, newComponent) => {
  const pressOpacity = newValue / 100;
  var tabBarOptions  = Object.assign({}, newComponent.properties.config.tabBarOptions);
  tabBarOptions.scaledPressOpacity = newValue;
  tabBarOptions.pressOpacity = pressOpacity;
  newComponent.properties = Object.assign({}, newComponent.properties, {config: {tabBarOptions: tabBarOptions}, pressOpacity: pressOpacity, scaledPressOpacity: newValue});
};
 
const TTopTabNavigator = {
  Type: TOP_NAVIGATOR_TYPE,
  ReactType: 'TopTabNavigator',
  Category: 'LAYOUT',
  Index: 1,
  ThunkableVersion: 50,
  SupportsWeb: true,
  TranslateLabelKey: "components.top-tab-navigator.name",
  TranslateTooltipKey: "components.top-tab-navigator.tooltip",
  DocumentationURL: "components.top-tab-navigator.documentation-url",
  TPropTypes: {
    SwipeEnabled: TPropTypes.bool.core().reactProperty('config.swipeEnabled').translatePropertyLabel("components.properties.SwipeEnabled").defaultValue(true),
    AnimationEnabled:
      TPropTypes.bool.core().reactProperty('config.animationEnabled').defaultValue(true).translatePropertyLabel("components.properties.AnimationEnabled"),
    Lazy: TPropTypes.bool.core().reactProperty('config.lazy').translatePropertyLabel("components.properties.Lazy"),
    backgroundColor: ColorPropType.core().reactProperty(`${TBO}.style.backgroundColor`).defaultValue('rgba(0, 0, 0, 0)')
                                  .translatePropertyLabel("components.properties.styling.backgroundColor"),
    tabIndicatorColor: ColorPropType.core().reactProperty(`${TBO}.indicatorStyle.backgroundColor`).defaultValue('rgba(0, 0, 0, 1)')
                                    .translatePropertyLabel("components.properties.tabIndicatorColor"),
    activeTintColor: ColorPropType.core().reactProperty(`${TBO}.activeTintColor`).defaultValue('rgba(0, 0, 0, 1)')
                                  .translatePropertyLabel("components.properties.activeTintColor"),
    inactiveTintColor: ColorPropType.core().reactProperty(`${TBO}.inactiveTintColor`).defaultValue('rgba(155, 155, 155, 1)')
                                    .translatePropertyLabel("components.properties.inactiveTintColor"),
    showLabelIcon: TPropTypes.oneOfWithLabel([
      ['components.properties.showLabel', "showLabel"],
      ['components.properties.showIcon', "showIcon"],
      ['components.properties.showLabelAndIcon', "showLabelAndIcon"],
    ]).defaultValue("showLabelAndIcon").core().translatePropertyLabel("components.properties.showLabelIcon"),
 
    /* Advanced */
    upperCaseLabel: TPropTypes.bool.reactProperty(`${TBO}.upperCaseLabel`).defaultValue(true)
                              .translatePropertyLabel("components.properties.upperCaseLabel"),
    pressColor: ColorPropType.reactProperty(`${TBO}.pressColor`).defaultValue('rgba(0, 0, 0, 0)')
                             .translatePropertyLabel("components.properties.scaledPressOpacityAndroidOnly"),
    pressOpacity: TPropTypes.number.reactProperty(`${TBO}.pressOpacity`).defaultValue(0)
                            .translatePropertyLabel("components.properties.pressOpacity"),
    scaledPressOpacity: TPropTypes.number.reactProperty(`${TBO}.scaledPressOpacity`).defaultValue(0)
                            .minValue(0).maxValue(100)
                            .designTimeCustomSetter(scaledPressOpacityCustomSetter)
                            .runtimeCustomSetter(scaledPressOpacityCustomSetter)
                            .translatePropertyLabel("components.properties.scaledPressOpacityAndroidOnly")
                            .dndOnly(),
    scrollEnabled: TPropTypes.bool.reactProperty(`${TBO}.scrollEnabled`).defaultValue(false)
                             .translatePropertyLabel("components.properties.scrollEnabled"),
    tabIndicatorHeight: TPropTypes.number.reactProperty(`${TBO}.indicatorStyle.height`).defaultValue(2)
                                  .translatePropertyLabel("components.properties.tabIndicatorHeight"),
 
    // Tab Navigation Options
    TabNavigationOptions: TPropTypes.shape({
      title: TPropTypes.string,
      tabBarVisible: TPropTypes.bool.defaultValue(true).canUseWithParentTypes(undefined, [TOP_NAVIGATOR_TYPE]),
      tabBarLabel: TPropTypes.string,
    }).core().reactProperty('navigationOptions').canUseWithParentTypes(TAB_NAVIGATOR_TYPES),
    
    TabNavigatorIcon: TPropTypes.shape({
      activeTabIcon: TPropTypes.fileName.core().reactProperty('tabNavigationOptions.activeTabIcon'),
      inactiveTabIcon: TPropTypes.fileName.core().reactProperty('tabNavigationOptions.inactiveTabIcon'),
    }).core().reactProperty('tabNavigationOptions').canUseWithParentTypes(TAB_NAVIGATOR_TYPES),
  },
  getDefaultChildren: ({ usesDragDropUi, defaultDesignerDevice, defaultDesignerOrientation }) => {
    let defaultChildren = NAVIGAOTR_LAYOUT.ICON;
    if (usesDragDropUi) {
      const designerHeightWidthProperties = {};
      if (defaultDesignerDevice && defaultDesignerOrientation) {
        const { height, width } = getDeviceSizeFromTypeAndOrientation(defaultDesignerDevice, defaultDesignerOrientation);
        designerHeightWidthProperties.DesignerHeight = height;
        designerHeightWidthProperties.DesignerWidth = width;
      }
      defaultChildren = defaultChildren.map((screen) => {
        return { ...screen, properties: { ...screen.properties, isSimple: true, ...designerHeightWidthProperties }};
      })
    }
    return defaultChildren;
  },
}
